This batch uses that combination: server-side OIDC code exchange into an HTTP-only app session in web, then bearer-token verification in gov-api, while preserving the current membership flows behind authenticated server functions.

import { randomUUID } from 'node:crypto'
import { decodeJwt } from 'jose'
import type { AppRole } from '@ardtire/authz'
import { appRoles } from '@ardtire/authz'
import type { AuthenticatedUser } from './session'

interface KeycloakDiscoveryDocument {
  authorization_endpoint: string
  token_endpoint: string
  end_session_endpoint?: string
}

interface TokenResponse {
  access_token: string
  expires_in: number
  token_type: string
}

let discoveryCache: KeycloakDiscoveryDocument | null = null

function getIssuer(): string {
  const issuer = process.env.KEYCLOAK_ISSUER
  if (!issuer) {
    throw new Error('KEYCLOAK_ISSUER is required.')
  }
  return issuer
}

function getClientId(): string {
  const clientId = process.env.KEYCLOAK_CLIENT_ID
  if (!clientId) {
    throw new Error('KEYCLOAK_CLIENT_ID is required.')
  }
  return clientId
}

function getClientSecret(): string {
  const secret = process.env.KEYCLOAK_WEB_CLIENT_SECRET
  if (!secret) {
    throw new Error('KEYCLOAK_WEB_CLIENT_SECRET is required.')
  }
  return secret
}

function getWebBaseUrl(): string {
  const webBaseUrl = process.env.WEB_BASE_URL
  if (!webBaseUrl) {
    throw new Error('WEB_BASE_URL is required.')
  }
  return webBaseUrl
}

export function getKeycloakCallbackUrl(): string {
  return `${getWebBaseUrl()}/auth/callback`
}

async function getDiscoveryDocument(): Promise<KeycloakDiscoveryDocument> {
  if (discoveryCache) {
    return discoveryCache
  }

  const response = await fetch(`${getIssuer()}/.well-known/openid-configuration`)
  if (!response.ok) {
    throw new Error(`Failed to load Keycloak discovery document: ${response.status}`)
  }

  const json = (await response.json()) as KeycloakDiscoveryDocument
  discoveryCache = json
  return json
}

export async function createKeycloakAuthorizationUrl(input: {
  state: string
  redirectTo?: string
}): Promise<string> {
  const discovery = await getDiscoveryDocument()
  const url = new URL(discovery.authorization_endpoint)

  url.searchParams.set('client_id', getClientId())
  url.searchParams.set('response_type', 'code')
  url.searchParams.set('scope', 'openid profile email')
  url.searchParams.set('redirect_uri', getKeycloakCallbackUrl())
  url.searchParams.set('state', input.state)

  return url.toString()
}

export async function exchangeAuthorizationCodeForToken(code: string): Promise<{
  accessToken: string
  expiresAt: number
  user: AuthenticatedUser
}> {
  const discovery = await getDiscoveryDocument()

  const body = new URLSearchParams()
  body.set('grant_type', 'authorization_code')
  body.set('code', code)
  body.set('client_id', getClientId())
  body.set('client_secret', getClientSecret())
  body.set('redirect_uri', getKeycloakCallbackUrl())

  const response = await fetch(discovery.token_endpoint, {
    method: 'POST',
    headers: {
      'content-type': 'application/x-www-form-urlencoded',
    },
    body,
  })

  if (!response.ok) {
    const text = await response.text()
    throw new Error(`Failed to exchange Keycloak authorization code: ${text}`)
  }

  const tokenResponse = (await response.json()) as TokenResponse
  const claims = decodeJwt(tokenResponse.access_token)

  return {
    accessToken: tokenResponse.access_token,
    expiresAt: Date.now() + tokenResponse.expires_in * 1000,
    user: mapClaimsToAuthenticatedUser(claims),
  }
}

export function createLoginState(): string {
  return randomUUID()
}

function mapClaimsToAuthenticatedUser(claims: Record<string, unknown>): AuthenticatedUser {
  const validRoles = new Set<string>(appRoles)

  const realmRoles = Array.isArray((claims.realm_access as { roles?: unknown[] } | undefined)?.roles)
    ? ((claims.realm_access as { roles?: unknown[] }).roles ?? [])
    : []

  const resourceRolesObject =
    (claims.resource_access as Record<string, { roles?: unknown[] }> | undefined) ?? {}
  const resourceRoles = Object.values(resourceRolesObject)
    .flatMap((entry) => (Array.isArray(entry.roles) ? entry.roles : []))

  const mergedRoles = [...realmRoles, ...resourceRoles]
    .filter((value): value is string => typeof value === 'string' && validRoles.has(value))

  return {
    sub: typeof claims.sub === 'string' ? claims.sub : '',
    email: typeof claims.email === 'string' ? claims.email : null,
    name: typeof claims.name === 'string' ? claims.name : null,
    preferredUsername:
      typeof claims.preferred_username === 'string' ? claims.preferred_username : null,
    roles: Array.from(new Set(mergedRoles)) as AppRole[],
  }
}

import { createServerFn } from '@tanstack/solid-start'
import { z } from 'zod'
import {
  createKeycloakAuthorizationUrl,
  createLoginState,
  exchangeAuthorizationCodeForToken,
} from '../../lib/auth/keycloak'
import { useArdtireAuthSession } from '../../lib/auth/session'

export const getCurrentAuth = createServerFn({ method: 'GET' }).handler(async () => {
  const session = await useArdtireAuthSession()
  const user = session.data.user
  const accessToken = session.data.accessToken
  const expiresAt = session.data.accessTokenExpiresAt

  if (!user || !accessToken || !expiresAt) {
    return null
  }

  if (Date.now() >= expiresAt) {
    await session.clear()
    return null
  }

  return user
})

export const beginKeycloakLogin = createServerFn({ method: 'POST' })
  .inputValidator(
    z.object({
      redirectTo: z.string().optional(),
    }),
  )
  .handler(async ({ data }) => {
    const session = await useArdtireAuthSession()
    const state = createLoginState()

    await session.update({
      ...session.data,
      loginState: state,
      postLoginRedirect: data.redirectTo ?? '/',
    })

    const authorizationUrl = await createKeycloakAuthorizationUrl({
      state,
      redirectTo: data.redirectTo,
    })

    return {
      authorizationUrl,
    }
  })

export const completeKeycloakLogin = createServerFn({ method: 'POST' })
  .inputValidator(
    z.object({
      code: z.string().min(1),
      state: z.string().min(1),
    }),
  )
  .handler(async ({ data }) => {
    const session = await useArdtireAuthSession()

    if (!session.data.loginState || session.data.loginState !== data.state) {
      throw new Error('Invalid OIDC login state.')
    }

    const tokenResult = await exchangeAuthorizationCodeForToken(data.code)
    const redirectTo = session.data.postLoginRedirect || '/'

    await session.update({
      accessToken: tokenResult.accessToken,
      accessTokenExpiresAt: tokenResult.expiresAt,
      user: tokenResult.user,
      loginState: undefined,
      postLoginRedirect: undefined,
    })

    return {
      redirectTo,
      user: tokenResult.user,
    }
  })

export const clearAuthSession = createServerFn({ method: 'POST' }).handler(async () => {
  const session = await useArdtireAuthSession()
  await session.clear()

  return {
    success: true as const,
  }
})

function getGovApiBaseUrl(): string {

    process.env.GOV_API_BASE_URL || process.env.VITE_GOV_API_BASE_URL || 'http://localhost:3002'

  return baseUrl
}

export async function createAuthenticatedGovClient() {
  const session = await useArdtireAuthSession()

  if (!session.data.accessToken || !session.data.accessTokenExpiresAt) {
    throw new Error('Authentication is required.')
  }

  if (Date.now() >= session.data.accessTokenExpiresAt) {
    await session.clear()
    throw new Error('Your session has expired. Please sign in again.')
  }

  return createGovClient({
    baseUrl: getGovApiBaseUrl(),
    getHeaders: () => ({
      authorization: `Bearer ${session.data.accessToken}`,
    }),
  })
}

import { createRemoteJWKSet, jwtVerify } from 'jose'
import type { JWTPayload } from 'jose'
import { appRoles, type AppRole } from '@ardtire/authz'

const jwks = createRemoteJWKSet(new URL(process.env.KEYCLOAK_JWKS_URL || 'http://localhost:8080'))

function getAcceptedAudiences(): string[] {
  return [
    process.env.KEYCLOAK_CLIENT_ID,
    process.env.KEYCLOAK_GOV_API_CLIENT_ID,
  ].filter((value): value is string => Boolean(value))
}

export async function verifyKeycloakAccessToken(token: string): Promise<JWTPayload> {
  const issuer = process.env.KEYCLOAK_ISSUER
  if (!issuer) {
    throw new Error('KEYCLOAK_ISSUER is required.')
  }

  const { payload } = await jwtVerify(token, jwks, {
    issuer,
  })

  const acceptedAudiences = getAcceptedAudiences()
  const tokenAudiences = Array.isArray(payload.aud)
    ? payload.aud
    : typeof payload.aud === 'string'
      ? [payload.aud]
      : []

  const azp = typeof payload.azp === 'string' ? payload.azp : null

  const audienceAccepted =
    acceptedAudiences.length === 0 ||
    tokenAudiences.some((audience) => acceptedAudiences.includes(audience)) ||
    (azp !== null && acceptedAudiences.includes(azp))

  if (!audienceAccepted) {
    throw new Error('Token audience was not accepted by gov-api.')
  }

  return payload
}

export function extractAppRolesFromJwtPayload(payload: JWTPayload): AppRole[] {
  const validRoles = new Set<string>(appRoles)

  const realmRoles =
    Array.isArray((payload.realm_access as { roles?: unknown[] } | undefined)?.roles)
      ? (((payload.realm_access as { roles?: unknown[] })?.roles ?? []) as unknown[])
      : []

  const resourceAccess =
    (payload.resource_access as Record<string, { roles?: unknown[] }> | undefined) ?? {}

  const resourceRoles = Object.values(resourceAccess).flatMap((entry) =>
    Array.isArray(entry.roles) ? entry.roles : [],
  )

  const merged = [...realmRoles, ...resourceRoles].filter(
    (value): value is string => typeof value === 'string' && validRoles.has(value),
  )

  return Array.from(new Set(merged)) as AppRole[]
}
