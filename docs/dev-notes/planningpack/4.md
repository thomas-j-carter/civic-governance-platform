packages/gov-client/src/http.ts

import { z } from 'zod'

export class GovClientError extends Error {
  readonly status: number
  readonly body: unknown

  constructor(message: string, status: number, body: unknown) {
    super(message)
    this.name = 'GovClientError'
    this.status = status
    this.body = body
  }
}

export interface GovClientOptions {
  baseUrl: string
  getHeaders?: () => HeadersInit | Promise<HeadersInit>
}

export async function requestJson<TSchema extends z.ZodTypeAny>(input: {
  options: GovClientOptions
  path: string
  method?: 'GET' | 'POST' | 'PUT' | 'PATCH' | 'DELETE'
  body?: unknown
  schema: TSchema
}): Promise<z.infer<TSchema>> {
  const headers = new Headers(await input.options.getHeaders?.())

  if (input.body !== undefined) {
    headers.set('content-type', 'application/json')
  }

  const response = await fetch(`${input.options.baseUrl}${input.path}`, {
    method: input.method ?? 'GET',
    headers,
    body: input.body !== undefined ? JSON.stringify(input.body) : undefined,
  })

  const rawBody = await response.text()
  const parsedBody = rawBody ? safeJsonParse(rawBody) : null

  if (!response.ok) {
    throw new GovClientError(
      `Request failed with status ${response.status}`,
      response.status,
      parsedBody,
    )
  }

  return input.schema.parse(parsedBody)
}

function safeJsonParse(input: string): unknown {
  try {
    return JSON.parse(input)
  } catch {
    return input
  }
}

  membershipApplicationSchema,
} from '@ardtire/contracts'

import type { GovClientOptions } from './http'
import { requestJson } from './http'

const membershipApplicationListSchema = z.array(membershipApplicationSchema)

export function createMembershipApplicationsClient(options: GovClientOptions) {
  return {
    list() {
      return requestJson({
        options,
        path: '/membership-applications',
        schema: membershipApplicationListSchema,
      })
    },

    createDraft(input: unknown) {
      const body = createMembershipApplicationDraftSchema.parse(input)

      return requestJson({
        options,
        path: '/membership-applications',
        method: 'POST',
        body,
        schema: membershipApplicationSchema,
      })
    },

    read(applicationId: string) {
      return requestJson({
        options,
        path: `/membership-applications/${applicationId}`,
        schema: membershipApplicationSchema,
      })
    },

    submit(applicationId: string) {
      return requestJson({
        options,
        path: `/membership-applications/${applicationId}/submit`,
        method: 'POST',
        schema: membershipApplicationSchema,
      })
    },
  }
}

This batch keeps the repo in-memory on purpose. Prisma 7’s current setup docs require a driver adapter in addition to @prisma/client, so the next batch is the right place to do the real DB wiring cleanly instead of half-wiring it here. 

apps/gov-api/src/membership-applications/repository.ts

import { randomUUID } from 'node:crypto'
import type { MembershipApplicationDto } from '@ardtire/contracts'

export interface MembershipApplicationRepository {
  create(input: {
    applicantUserId: string
    membershipClassRequested?: string
    summary?: string
  }): Promise<MembershipApplicationDto>

  findById(id: string): Promise<MembershipApplicationDto | null>

  listByApplicant(applicantUserId: string): Promise<MembershipApplicationDto[]>

  update(
    id: string,
    patch: Partial<MembershipApplicationDto>,
  ): Promise<MembershipApplicationDto | null>
}

type StoredApplication = MembershipApplicationDto

export class InMemoryMembershipApplicationRepository
  implements MembershipApplicationRepository
{
  private readonly items = new Map<string, StoredApplication>()

  async create(input: {
    applicantUserId: string
    membershipClassRequested?: string
    summary?: string
  }): Promise<MembershipApplicationDto> {
    const now = new Date().toISOString()
    const item: StoredApplication = {
      id: randomUUID(),
      applicantUserId: input.applicantUserId,
      state: 'draft',
      membershipClassRequested: input.membershipClassRequested ?? null,
      summary: input.summary ?? null,
      createdAt: now,
      updatedAt: now,
    }

    this.items.set(item.id, item)
    return structuredClone(item)
  }

  async findById(id: string): Promise<MembershipApplicationDto | null> {
    const item = this.items.get(id)
    return item ? structuredClone(item) : null
  }

  async listByApplicant(applicantUserId: string): Promise<MembershipApplicationDto[]> {
    return [...this.items.values()]
      .filter((item) => item.applicantUserId === applicantUserId)
      .sort((a, b) => a.createdAt.localeCompare(b.createdAt))
      .map((item) => structuredClone(item))
  }

  async update(
    id: string,
    patch: Partial<MembershipApplicationDto>,
  ): Promise<MembershipApplicationDto | null> {
    const existing = this.items.get(id)
    if (!existing) {
      return null
    }

    const next: StoredApplication = {
      ...existing,
      ...patch,
      updatedAt: new Date().toISOString(),
    }

    this.items.set(id, next)
    return structuredClone(next)
  }
}

export const membershipApplicationRepository =
  new InMemoryMembershipApplicationRepository()

apps/gov-api/src/membership-applications/service.ts

import { createAuditEvent } from '@ardtire/audit'
import { hasRole, type ActorContext } from '@ardtire/authz'
import type {
  CreateMembershipApplicationDraftInput,
  MembershipApplicationDto,
} from '@ardtire/contracts'
import { canTransitionMembershipApplication } from '@ardtire/workflows'
import {
  membershipApplicationRepository,
  type MembershipApplicationRepository,
} from './repository'

function assertAuthenticated(actor: ActorContext): asserts actor is ActorContext & {
  userId: string
} {
  if (!actor.isAuthenticated || !actor.userId) {
    throw new Error('Authentication is required.')
  }
}

function canReadApplication(actor: ActorContext, application: MembershipApplicationDto): boolean {
  if (actor.userId === application.applicantUserId) return true
  return hasRole(actor, 'admin') || hasRole(actor, 'founder') || hasRole(actor, 'reviewer')
}

export class MembershipApplicationService {
  constructor(
    private readonly repository: MembershipApplicationRepository = membershipApplicationRepository,
  ) {}

  async createDraft(input: {
    actor: ActorContext
    data: CreateMembershipApplicationDraftInput
    correlationId?: string
  }): Promise<MembershipApplicationDto> {
    assertAuthenticated(input.actor)

    const created = await this.repository.create({
      applicantUserId: input.actor.userId,
      membershipClassRequested: input.data.membershipClassRequested,
      summary: input.data.summary,
    })

    console.info(
      createAuditEvent({
        correlationId: input.correlationId,
        actorUserId: input.actor.userId,
        actionType: 'membership_application.create_draft',
        targetType: 'membership_application',
        targetId: created.id,
        result: 'success',
      }),
    )

    return created
  }

  async listForActor(input: {
    actor: ActorContext
  }): Promise<MembershipApplicationDto[]> {
    assertAuthenticated(input.actor)

    return this.repository.listByApplicant(input.actor.userId)
  }

  async readById(input: {
    actor: ActorContext
    applicationId: string
  }): Promise<MembershipApplicationDto | null> {
    const application = await this.repository.findById(input.applicationId)

    if (!application) {
      return null
    }

    if (!canReadApplication(input.actor, application)) {
      throw new Error('Not authorized to read this application.')
    }

    return application
  }

  async submit(input: {
    actor: ActorContext
    applicationId: string
    correlationId?: string
  }): Promise<MembershipApplicationDto | null> {
    assertAuthenticated(input.actor)

    const application = await this.repository.findById(input.applicationId)
    if (!application) {
      return null
    }

    if (application.applicantUserId !== input.actor.userId) {
      throw new Error('Only the applicant may submit this application.')
    }

    if (!canTransitionMembershipApplication(application.state, 'submitted')) {
      throw new Error(
        `Cannot transition membership application from ${application.state} to submitted.`,
      )
    }

    const updated = await this.repository.update(application.id, {
      state: 'submitted',
    })

    if (!updated) {
      return null
    }

    console.info(

        actionType: 'membership_application.submit',
        targetType: 'membership_application',
        targetId: updated.id,
        result: 'success',
      }),
    )

    return updated
  }

import { Hono } from 'hono'
import {
  createMembershipApplicationDraftSchema,
} from '@ardtire/contracts'
import { membershipApplicationService } from '../../membership-applications/service'
import type { AppBindings } from '../types'

export function createMembershipApplicationRoutes() {
  const app = new Hono<AppBindings>()

  app.get('/membership-applications', async (c) => {
    try {
      const actor = c.get('actor')
      const items = await membershipApplicationService.listForActor({ actor })
      return c.json(items)
    } catch (error) {
      return c.json(
        {
          code: 'UNAUTHORIZED',
          message: error instanceof Error ? error.message : 'Unauthorized',
        },
        401,
      )
    }
  })

  app.post('/membership-applications', async (c) => {
    try {
      const actor = c.get('actor')
      const requestId = c.get('requestId')
      const json = await c.req.json().catch(() => ({}))
      const data = createMembershipApplicationDraftSchema.parse(json)

      const created = await membershipApplicationService.createDraft({
        actor,
        data,
        correlationId: requestId,
      })

      return c.json(created, 201)
    } catch (error) {
      if (error instanceof Error && error.name === 'ZodError') {
        return c.json(
          {
            code: 'VALIDATION_ERROR',
            message: 'Invalid membership application payload.',
          },
          400,
        )
      }

      return c.json(
        {
          code: 'UNAUTHORIZED',
          message: error instanceof Error ? error.message : 'Unauthorized',
        },
        401,
      )
    }
  })

  app.get('/membership-applications/:applicationId', async (c) => {
    try {
      const actor = c.get('actor')
      const applicationId = c.req.param('applicationId')

      const item = await membershipApplicationService.readById({
        actor,
        applicationId,
      })

      if (!item) {
        return c.json(
          {
            code: 'NOT_FOUND',
            message: 'Membership application not found.',
          },
          404,
        )
      }

      return c.json(item)
    } catch (error) {
      return c.json(
        {
          code: 'FORBIDDEN',
          message: error instanceof Error ? error.message : 'Forbidden',
        },
        403,
      )
    }
  })

  app.post('/membership-applications/:applicationId/submit', async (c) => {
    try {
      const actor = c.get('actor')
      const requestId = c.get('requestId')
      const applicationId = c.req.param('applicationId')

      const updated = await membershipApplicationService.submit({
        actor,
        applicationId,
        correlationId: requestId,
      })

      if (!updated) {
        return c.json(
          {
            code: 'NOT_FOUND',
            message: 'Membership application not found.',
          },
          404,
        )
      }

      return c.json(updated)
    } catch (error) {
      const message =
        error instanceof Error ? error.message : 'Unable to submit application.'

      const status =
        message.includes('Authentication') ? 401 : message.includes('Only the applicant') ? 403 : 400

      return c.json(
        {
          code: 'STATE_TRANSITION_DENIED',
          message,
        },
        status,
      )
    }
  })

  return app
}

import { Hono } from 'hono'
import { authContextMiddleware } from './http/middleware/auth-context'
import { requestIdMiddleware } from './http/middleware/request-id'
import { createAuthContextRoutes } from './http/routes/auth-context'
import { createHealthRoutes } from './http/routes/health'
import { createMembershipApplicationRoutes } from './http/routes/membership-applications'
import type { AppBindings } from './http/types'

export function createApp() {
  const app = new Hono<AppBindings>()

  app.use('*', requestIdMiddleware)
  app.use('*', authContextMiddleware)

  app.onError((error, c) => {
    console.error('Unhandled request error', {
      requestId: c.get('requestId'),
      message: error.message,
      stack: error.stack,
    })

    return c.json(
      {
        code: 'INTERNAL_SERVER_ERROR',
        message: 'An unexpected error occurred.',
      },
      500,
    )
  })

  app.route('/', createHealthRoutes())
  app.route('/', createAuthContextRoutes())
  app.route('/', createMembershipApplicationRoutes())

  return app
}
