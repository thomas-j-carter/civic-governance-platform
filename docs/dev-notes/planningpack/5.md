import type { MembershipApplicationState } from '@ardtire/domain'
import type { PrismaClient } from '../../../../generated/prisma/client'
import { prisma } from '../lib/prisma'

export class PrismaMembershipApplicationRepository
  implements MembershipApplicationRepository
{
  constructor(private readonly db: PrismaClient = prisma) {}

  async createDraft(input: {
    applicantUserId: string
    membershipClassRequested?: string
    summary?: string
  }): Promise<MembershipApplicationDto> {
    const created = await this.db.membershipApplication.create({
      data: {
        applicantUserId: input.applicantUserId,
        state: 'draft',
        membershipClassRequested: input.membershipClassRequested,
        summary: input.summary,
      },
    })

    return toMembershipApplicationDto(created)
  }

  async findById(id: string): Promise<MembershipApplicationDto | null> {
    const found = await this.db.membershipApplication.findUnique({
      where: { id },
    })

    return found ? toMembershipApplicationDto(found) : null
  }

  async listAll(): Promise<MembershipApplicationDto[]> {
    const rows = await this.db.membershipApplication.findMany({
      orderBy: {
        createdAt: 'desc',
      },
    })

    return rows.map(toMembershipApplicationDto)
  }

  async listByApplicant(applicantUserId: string): Promise<MembershipApplicationDto[]> {
    const rows = await this.db.membershipApplication.findMany({
      where: {
        applicantUserId,
      },
      orderBy: {
        createdAt: 'desc',
      },
    })

    return rows.map(toMembershipApplicationDto)
  }

  async updateState(input: {
    applicationId: string
    nextState: MembershipApplicationState
  }): Promise<MembershipApplicationDto> {
    const updated = await this.db.membershipApplication.update({
      where: {
        id: input.applicationId,
      },
      data: {
        state: input.nextState,
        submittedAt: input.nextState === 'submitted' ? new Date() : undefined,
      },
    })

    return toMembershipApplicationDto(updated)
  }

  async startReview(input: {
    application: MembershipApplicationDto
    reviewerUserId: string
  }): Promise<MembershipApplicationDto> {
    const updated = await this.db.$transaction(async (tx) => {
      await tx.membershipApplicationReview.create({
        data: {
          applicationId: input.application.id,
          reviewerUserId: input.reviewerUserId,
          reviewState: 'under_review',
        },
      })

      return tx.membershipApplication.update({
        where: {
          id: input.application.id,
        },
        data: {
          state: 'under_review',
          reviewedAt: new Date(),
        },
      })
    })

    return toMembershipApplicationDto(updated)
  }

  async returnForRevision(input: {
    application: MembershipApplicationDto
    reviewerUserId: string
    reason: string
  }): Promise<MembershipApplicationDto> {
    const updated = await this.db.$transaction(async (tx) => {
      await tx.membershipApplicationReview.create({
        data: {
          applicationId: input.application.id,
          reviewerUserId: input.reviewerUserId,
          reviewState: 'returned_for_revision',
          notes: input.reason,
        },
      })

      return tx.membershipApplication.update({
        where: {
          id: input.application.id,
        },
        data: {
          state: 'returned_for_revision',
          reviewedAt: new Date(),
        },
      })
    })

    return toMembershipApplicationDto(updated)
  }

  async approve(input: {
    application: MembershipApplicationDto
    reviewerUserId: string
    reason?: string
  }): Promise<MembershipApplicationDto> {
    const now = new Date()

    const updated = await this.db.$transaction(async (tx) => {
      const decision = await tx.membershipDecision.create({
        data: {
          applicationId: input.application.id,
          decidedByUserId: input.reviewerUserId,
          decisionCode: 'approved',
          reason: input.reason,
          decidedAt: now,
        },
      })

      await tx.membershipRecord.create({
        data: {
          userId: input.application.applicantUserId,
          decisionId: decision.id,
          state: 'active',
          membershipClass: input.application.membershipClassRequested ?? undefined,
          effectiveAt: now,
        },
      })

      await tx.membershipStatusEvent.create({
        data: {
          membershipId: (
            await tx.membershipRecord.findUniqueOrThrow({
              where: {
                decisionId: decision.id,
              },
              select: {
                id: true,
              },
            })
          ).id,
          toState: 'active',
          actedByUserId: input.reviewerUserId,
          reason: input.reason,
        },
      })

      await tx.membershipApplicationReview.create({
        data: {
          applicationId: input.application.id,
          reviewerUserId: input.reviewerUserId,
          reviewState: 'approved',
          notes: input.reason,
        },
      })

      return tx.membershipApplication.update({
        where: {
          id: input.application.id,
        },
        data: {
          state: 'approved',
          reviewedAt: now,
          decisionAt: now,
        },
      })
    })

    return toMembershipApplicationDto(updated)
  }

  async reject(input: {
    application: MembershipApplicationDto
    reviewerUserId: string
    reason: string
  }): Promise<MembershipApplicationDto> {
    const now = new Date()

    const updated = await this.db.$transaction(async (tx) => {
      await tx.membershipDecision.create({
        data: {
          applicationId: input.application.id,
          decidedByUserId: input.reviewerUserId,
          decisionCode: 'rejected',
          reason: input.reason,
          decidedAt: now,
        },
      })

      await tx.membershipApplicationReview.create({
        data: {
          applicationId: input.application.id,
          reviewerUserId: input.reviewerUserId,
          reviewState: 'rejected',
          notes: input.reason,
        },
      })

      return tx.membershipApplication.update({
        where: {
          id: input.application.id,
        },
        data: {
          state: 'rejected',
          reviewedAt: now,
          decisionAt: now,
        },
      })
    })

    return toMembershipApplicationDto(updated)
  }
}

import { createAuditEvent } from '@ardtire/audit'
import { hasPermission, hasRole, type ActorContext } from '@ardtire/authz'
import type {
  CreateMembershipApplicationDraftInput,
  MembershipApplicationDto,
} from '@ardtire/contracts'
import { canTransitionMembershipApplication } from '@ardtire/workflows'
import {
  membershipApplicationRepository,
  type MembershipApplicationRepository,
} from './repository'

export class MembershipApplicationServiceError extends Error {
  readonly status: number
  readonly code: string

  constructor(input: { status: number; code: string; message: string }) {
    super(input.message)
    this.name = 'MembershipApplicationServiceError'
    this.status = input.status
    this.code = input.code
  }
}

function assertAuthenticated(actor: ActorContext): asserts actor is ActorContext & {
  userId: string
} {
  if (!actor.isAuthenticated || !actor.userId) {
    throw new MembershipApplicationServiceError({
      status: 401,
      code: 'UNAUTHORIZED',
      message: 'Authentication is required.',
    })
  }
}

function canReview(actor: ActorContext): boolean {
  return (
    hasPermission(actor, 'review_membership_application') ||
    hasRole(actor, 'founder')
  )
}

function canReadApplication(actor: ActorContext, application: MembershipApplicationDto): boolean {
  if (actor.userId === application.applicantUserId) return true
  return canReview(actor)
}

function assertCanTransition(
  application: MembershipApplicationDto,
  nextState: MembershipApplicationDto['state'],
) {
  if (!canTransitionMembershipApplication(application.state, nextState)) {
    throw new MembershipApplicationServiceError({
      status: 400,
      code: 'STATE_TRANSITION_DENIED',
      message: `Cannot transition membership application from ${application.state} to ${nextState}.`,
    })
  }
}

export class MembershipApplicationService {
  constructor(
    private readonly repository: MembershipApplicationRepository = membershipApplicationRepository,
  ) {}

  async createDraft(input: {
    actor: ActorContext
    data: CreateMembershipApplicationDraftInput
    correlationId?: string
  }): Promise<MembershipApplicationDto> {
    assertAuthenticated(input.actor)

    const created = await this.repository.createDraft({
      applicantUserId: input.actor.userId,
      membershipClassRequested: input.data.membershipClassRequested,
      summary: input.data.summary,
    })

    console.info(
      createAuditEvent({
        correlationId: input.correlationId,
        actorUserId: input.actor.userId,
        actionType: 'membership_application.create_draft',
        targetType: 'membership_application',
        targetId: created.id,
        result: 'success',
      }),
    )

    return created
  }

  async listForActor(input: {
    actor: ActorContext
  }): Promise<MembershipApplicationDto[]> {
    assertAuthenticated(input.actor)

    if (canReview(input.actor)) {
      return this.repository.listAll()
    }

    return this.repository.listByApplicant(input.actor.userId)
  }

  async readById(input: {
    actor: ActorContext
    applicationId: string
  }): Promise<MembershipApplicationDto | null> {
    assertAuthenticated(input.actor)

    const application = await this.repository.findById(input.applicationId)

    if (!application) {
      return null
    }

    if (!canReadApplication(input.actor, application)) {
      throw new MembershipApplicationServiceError({
        status: 403,
        code: 'FORBIDDEN',
        message: 'Not authorized to read this application.',
      })
    }

    return application
  }

  async submit(input: {
    actor: ActorContext
    applicationId: string
    correlationId?: string
  }): Promise<MembershipApplicationDto | null> {
    assertAuthenticated(input.actor)

    const application = await this.repository.findById(input.applicationId)

    if (!application) {
      return null
    }

    if (application.applicantUserId !== input.actor.userId) {
      throw new MembershipApplicationServiceError({
        status: 403,
        code: 'FORBIDDEN',
        message: 'Only the applicant may submit this application.',
      })
    }

    assertCanTransition(application, 'submitted')

    const updated = await this.repository.updateState({
      applicationId: application.id,
      nextState: 'submitted',
    })

    console.info(
      createAuditEvent({
        correlationId: input.correlationId,
        actorUserId: input.actor.userId,
        actionType: 'membership_application.submit',
        targetType: 'membership_application',
        targetId: updated.id,
        result: 'success',
      }),
    )

    return updated
  }

  async startReview(input: {
    actor: ActorContext
    applicationId: string
    correlationId?: string
  }): Promise<MembershipApplicationDto | null> {
    assertAuthenticated(input.actor)

    if (!canReview(input.actor)) {
      throw new MembershipApplicationServiceError({
        status: 403,
        code: 'FORBIDDEN',
        message: 'Not authorized to review membership applications.',
      })
    }

    const application = await this.repository.findById(input.applicationId)

    if (!application) {
      return null
    }

    assertCanTransition(application, 'under_review')

    const updated = await this.repository.startReview({
      application,
      reviewerUserId: input.actor.userId,
    })

    console.info(
      createAuditEvent({
        correlationId: input.correlationId,
        actorUserId: input.actor.userId,
        actionType: 'membership_application.start_review',
        targetType: 'membership_application',
        targetId: updated.id,
        result: 'success',
      }),
    )

    return updated
  }

  async returnForRevision(input: {
    actor: ActorContext
    applicationId: string
    reason: string
    correlationId?: string
  }): Promise<MembershipApplicationDto | null> {
    assertAuthenticated(input.actor)

    if (!canReview(input.actor)) {
      throw new MembershipApplicationServiceError({
        status: 403,
        code: 'FORBIDDEN',
        message: 'Not authorized to review membership applications.',
      })
    }

    const application = await this.repository.findById(input.applicationId)

    if (!application) {
      return null
    }

    assertCanTransition(application, 'returned_for_revision')

    const updated = await this.repository.returnForRevision({
      application,
      reviewerUserId: input.actor.userId,
      reason: input.reason,
    })

    console.info(
      createAuditEvent({
        correlationId: input.correlationId,
        actorUserId: input.actor.userId,
        actionType: 'membership_application.return_for_revision',
        targetType: 'membership_application',
        targetId: updated.id,
        result: 'success',
        reason: input.reason,
      }),
    )

    return updated
  }

  async approve(input: {
    actor: ActorContext
    applicationId: string
    correlationId?: string
  }): Promise<MembershipApplicationDto | null> {
    assertAuthenticated(input.actor)

    if (
      !hasPermission(input.actor, 'approve_membership_application') &&
      !hasRole(input.actor, 'founder')
    ) {
      throw new MembershipApplicationServiceError({
        status: 403,
        code: 'FORBIDDEN',
        message: 'Not authorized to approve membership applications.',
      })
    }

    const application = await this.repository.findById(input.applicationId)

    if (!application) {
      return null
    }

    assertCanTransition(application, 'approved')

    const updated = await this.repository.approve({
      application,
      reviewerUserId: input.actor.userId,
    })

    console.info(
      createAuditEvent({
        correlationId: input.correlationId,
        actorUserId: input.actor.userId,
        actionType: 'membership_application.approve',
        targetType: 'membership_application',
        targetId: updated.id,
        result: 'success',
      }),
    )

    return updated
  }

  async reject(input: {
    actor: ActorContext
    applicationId: string
    reason: string
    correlationId?: string
  }): Promise<MembershipApplicationDto | null> {
    assertAuthenticated(input.actor)

    if (
      !hasPermission(input.actor, 'reject_membership_application') &&
      !hasRole(input.actor, 'founder')
    ) {
      throw new MembershipApplicationServiceError({
        status: 403,
        code: 'FORBIDDEN',
        message: 'Not authorized to reject membership applications.',
      })
    }

    const application = await this.repository.findById(input.applicationId)

    if (!application) {
      return null
    }

    assertCanTransition(application, 'rejected')

    const updated = await this.repository.reject({
      application,
      reviewerUserId: input.actor.userId,
      reason: input.reason,
    })

    console.info(
      createAuditEvent({
        correlationId: input.correlationId,
        actorUserId: input.actor.userId,
        actionType: 'membership_application.reject',
        targetType: 'membership_application',
        targetId: updated.id,
        result: 'success',
        reason: input.reason,
      }),
    )

    return updated
  }
}

import type { AppBindings } from '../types'

function errorResponse(error: unknown) {
  if (error instanceof MembershipApplicationServiceError) {
    return {
      status: error.status,
      body: {
        code: error.code,
        message: error.message,
      },
    }
  }

  if (error instanceof Error && error.name === 'ZodError') {
    return {
      status: 400,
      body: {
        code: 'VALIDATION_ERROR',
        message: 'Invalid request payload.',
      },
    }
  }

  return {
    status: 500,
    body: {
      code: 'INTERNAL_SERVER_ERROR',
      message: 'An unexpected error occurred.',
    },
  }
}

export function createMembershipApplicationRoutes() {
  const app = new Hono<AppBindings>()

  app.get('/membership-applications', async (c) => {
    try {
      const actor = c.get('actor')
      const items = await membershipApplicationService.listForActor({ actor })
      return c.json(items)
    } catch (error) {
      const handled = errorResponse(error)
      return c.json(handled.body, handled.status)
    }
  })

  app.post('/membership-applications', async (c) => {
    try {
      const actor = c.get('actor')
      const requestId = c.get('requestId')
      const json = await c.req.json().catch(() => ({}))
      const data = createMembershipApplicationDraftSchema.parse(json)

      const created = await membershipApplicationService.createDraft({
        actor,
        data,
        correlationId: requestId,
      })

      return c.json(created, 201)
    } catch (error) {
      const handled = errorResponse(error)
      return c.json(handled.body, handled.status)
    }
  })

  app.get('/membership-applications/:applicationId', async (c) => {
    try {
      const actor = c.get('actor')
      const applicationId = c.req.param('applicationId')

      const item = await membershipApplicationService.readById({
        actor,
        applicationId,
      })

      if (!item) {
        return c.json(
          {
            code: 'NOT_FOUND',
            message: 'Membership application not found.',
          },
          404,
        )
      }

      return c.json(item)
    } catch (error) {
      const handled = errorResponse(error)
      return c.json(handled.body, handled.status)
    }
  })

  app.post('/membership-applications/:applicationId/submit', async (c) => {
    try {
      const actor = c.get('actor')
      const requestId = c.get('requestId')
      const applicationId = c.req.param('applicationId')

      const updated = await membershipApplicationService.submit({
        actor,
        applicationId,
        correlationId: requestId,
      })

      if (!updated) {
        return c.json(
          {
            code: 'NOT_FOUND',
            message: 'Membership application not found.',
          },
          404,
        )
      }

      return c.json(updated)
    } catch (error) {
      const handled = errorResponse(error)
      return c.json(handled.body, handled.status)
    }
  })

async function resetDatabase() {
  await prisma.auditEvent.deleteMany()
  await prisma.membershipStatusEvent.deleteMany()
  await prisma.membershipRecord.deleteMany()
  await prisma.membershipDecision.deleteMany()
  await prisma.membershipApplicationReview.deleteMany()
  await prisma.membershipApplication.deleteMany()
  await prisma.userRoleAssignment.deleteMany()
  await prisma.applicationRole.deleteMany()
  await prisma.user.deleteMany()
}

async function seedUser(input: {
  id: string
  email: string
  displayName: string
}) {
  return prisma.user.create({
    data: {
      id: input.id,
      email: input.email,
      displayName: input.displayName,
    },
  })
}

function authHeaders(input: {
  userId: string
  email: string
  roles: string[]
}): HeadersInit {
  return {
    'content-type': 'application/json',
    'x-user-id': input.userId,
    'x-user-email': input.email,
    'x-user-roles': input.roles.join(','),
  }
}

test('membership application draft and submit flow works for applicant', {
  skip: !hasDatabase,
}, async () => {
