import { z } from 'zod'

const nodeEnvSchema = z.enum(['development', 'test', 'production']).default('development')

export const sharedEnvSchema = z.object({
  NODE_ENV: nodeEnvSchema,
  LOG_LEVEL: z.string().default('info'),
  DATABASE_URL: z.string().min(1, 'DATABASE_URL is required'),
})

export const govApiEnvSchema = sharedEnvSchema.extend({
  GOV_API_HOST: z.string().default('0.0.0.0'),
  GOV_API_PORT: z.coerce.number().int().positive().default(3002),
  GOV_API_BASE_URL: z.string().url(),
  KEYCLOAK_BASE_URL: z.string().url(),
  KEYCLOAK_REALM: z.string().min(1),
  KEYCLOAK_CLIENT_ID: z.string().min(1),
  KEYCLOAK_GOV_API_CLIENT_ID: z.string().min(1),
  KEYCLOAK_GOV_API_CLIENT_SECRET: z.string().min(1),
  KEYCLOAK_JWKS_URL: z.string().url(),
  KEYCLOAK_ISSUER: z.string().url(),
})

export const webEnvSchema = sharedEnvSchema.extend({
  WEB_PORT: z.coerce.number().int().positive().default(3000),
  WEB_BASE_URL: z.string().url(),
  VITE_APP_NAME: z.string().default('Ardtire Society'),
  VITE_GOV_API_BASE_URL: z.string().url(),
})

export const cmsEnvSchema = sharedEnvSchema.extend({
  CMS_PORT: z.coerce.number().int().positive().default(3001),
  CMS_BASE_URL: z.string().url(),
  PAYLOAD_SECRET: z.string().min(1),
  PAYLOAD_PUBLIC_SERVER_URL: z.string().url(),
})

export type SharedEnv = z.infer<typeof sharedEnvSchema>
export type GovApiEnv = z.infer<typeof govApiEnvSchema>
export type WebEnv = z.infer<typeof webEnvSchema>

export function permissionsForRoles(roles: readonly AppRole[]): Permission[] {
  const set = new Set<Permission>()
  for (const role of roles) {
    for (const permission of rolePermissionMap[role] ?? []) {
      set.add(permission)
    }
  }
  return [...set]
}

packages/authz/src/actor.ts

import type { AppRole } from './roles'
import { permissionsForRoles, type Permission } from './permissions'

export interface ActorContext {
  isAuthenticated: boolean
  userId: string | null
  email: string | null
  roles: AppRole[]
  permissions: Permission[]
}

export function buildActorContext(input: {
  userId?: string | null
  email?: string | null
  roles?: readonly AppRole[]
}): ActorContext {
  const roles = [...(input.roles ?? [])]
  return {
    isAuthenticated: Boolean(input.userId),
    userId: input.userId ?? null,
    email: input.email ?? null,
    roles,
    permissions: permissionsForRoles(roles),
  }
}

export function hasRole(actor: ActorContext, role: AppRole): boolean {
  return actor.roles.includes(role)
}

export function hasPermission(actor: ActorContext, permission: Permission): boolean {
  return actor.permissions.includes(permission)
}

import type { MiddlewareHandler } from 'hono'
import { randomUUID } from 'node:crypto'
import type { AppBindings } from '../types'

export const requestIdMiddleware: MiddlewareHandler<AppBindings> = async (c, next) => {
  const incoming = c.req.header('x-request-id')
  const requestId = incoming && incoming.trim() ? incoming : randomUUID()

  c.set('requestId', requestId)
  c.header('x-request-id', requestId)

  await next()
}

apps/gov-api/src/http/middleware/auth-context.ts

import type { MiddlewareHandler } from 'hono'
import { buildActorContext, type AppRole, appRoles } from '@ardtire/authz'
import type { AppBindings } from '../types'

const validRoles = new Set<string>(appRoles)

function parseRoles(raw: string | undefined): AppRole[] {
  if (!raw) return []
  return raw
    .split(',')
    .map((value) => value.trim())
    .filter((value): value is AppRole => validRoles.has(value))
}

export const authContextMiddleware: MiddlewareHandler<AppBindings> = async (c, next) => {
  const userId = c.req.header('x-user-id') ?? null
  const email = c.req.header('x-user-email') ?? null
  const roles = parseRoles(c.req.header('x-user-roles'))

  c.set(
    'actor',
    buildActorContext({
      userId,
      email,
      roles,
    }),
  )

  await next()
}

import { Hono } from 'hono'
import type { AppBindings } from '../types'

export function createAuthContextRoutes() {
  const app = new Hono<AppBindings>()

  app.get('/auth/context', (c) => {
    const actor = c.get('actor')

    return c.json({
      isAuthenticated: actor.isAuthenticated,
      userId: actor.userId,
      email: actor.email,
      roles: actor.roles,
      permissions: actor.permissions,
    })
  })

  return app
}

apps/gov-api/src/app.ts

import { Hono } from 'hono'
import { authContextMiddleware } from './http/middleware/auth-context'
import { requestIdMiddleware } from './http/middleware/request-id'
import { createAuthContextRoutes } from './http/routes/auth-context'
import { createHealthRoutes } from './http/routes/health'
import type { AppBindings } from './http/types'

export function createApp() {
  const app = new Hono<AppBindings>()

  app.use('*', requestIdMiddleware)
  app.use('*', authContextMiddleware)

  app.onError((error, c) => {
    console.error('Unhandled request error', {
      requestId: c.get('requestId'),
      message: error.message,
      stack: error.stack,
    })

    return c.json(
      {
        code: 'INTERNAL_SERVER_ERROR',
        message: 'An unexpected error occurred.',
      },
      500,
    )
  })

  app.route('/', createHealthRoutes())
  app.route('/', createAuthContextRoutes())

  return app
}
