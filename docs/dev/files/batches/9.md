import { describe, expect, it } from "vitest";
import { IdempotencyService } from "./idempotency-service";
import { InMemoryIdempotencyRepository } from "../../infrastructure/persistence/in-memory/in-memory-idempotency-repository";
import { ConflictError } from "../../http/errors";

describe("IdempotencyService", () => {
  it("returns null when no idempotency key is provided", async () => {
    const service = new IdempotencyService(new InMemoryIdempotencyRepository());

    const result = await service.getExistingResponse({
      actorPersonId: "person-1",
      operationName: "proposal.createDraft",
      requestBody: { title: "A" },
    });

    expect(result).toBeNull();
  });

  it("stores and replays a response for the same key and payload", async () => {
    const service = new IdempotencyService(new InMemoryIdempotencyRepository());

    await service.storeResponse({
      actorPersonId: "person-1",
      operationName: "proposal.createDraft",
      idempotencyKey: "idem-1",
      requestBody: { title: "A" },
      response: {
        id: "proposal-1",
        title: "A",
      },
    });

    const replay = await service.getExistingResponse<{
      id: string;
      title: string;
    }>({
      actorPersonId: "person-1",
      operationName: "proposal.createDraft",
      idempotencyKey: "idem-1",
      requestBody: { title: "A" },
    });

    expect(replay).toEqual({
      id: "proposal-1",
      title: "A",
    });
  });

  it("rejects reuse of the same key with a different payload", async () => {
    const service = new IdempotencyService(new InMemoryIdempotencyRepository());

    await service.storeResponse({
      actorPersonId: "person-1",
      operationName: "proposal.createDraft",
      idempotencyKey: "idem-1",
      requestBody: { title: "A" },
      response: { id: "proposal-1" },
    });

    await expect(
      service.getExistingResponse({
        actorPersonId: "person-1",
        operationName: "proposal.createDraft",
        idempotencyKey: "idem-1",
        requestBody: { title: "B" },
      }),
    ).rejects.toBeInstanceOf(ConflictError);
  });

  it("treats the same key under different operations as distinct", async () => {
    const service = new IdempotencyService(new InMemoryIdempotencyRepository());

    await service.storeResponse({
      actorPersonId: "person-1",
      operationName: "proposal.createDraft",
      idempotencyKey: "idem-1",
      requestBody: { title: "A" },
      response: { id: "proposal-1" },
    });

    const result = await service.getExistingResponse({
      actorPersonId: "person-1",
      operationName: "ballot.create",
      idempotencyKey: "idem-1",
      requestBody: { title: "A" },
    });

    expect(result).toBeNull();
  });
});

import { describe, expect, it } from "vitest";
import {
  assertConcurrencyMatch,
  assertConcurrencyTokenPresent,
} from "./concurrency";
import { ConflictError } from "../../http/errors";

describe("concurrency helpers", () => {
  it("accepts matching concurrency tokens", () => {
    expect(() =>
      assertConcurrencyMatch(
        "2026-03-18T12:00:00.000Z",
        "2026-03-18T12:00:00.000Z",
      ),
    ).not.toThrow();
  });

  it("rejects mismatched concurrency tokens", () => {
    expect(() =>
      assertConcurrencyMatch(
        "2026-03-18T12:00:00.000Z",
        "2026-03-18T12:01:00.000Z",
      ),
    ).toThrow(ConflictError);
  });

  it("requires a concurrency token", () => {
    expect(() => assertConcurrencyTokenPresent(undefined)).toThrow(ConflictError);
    expect(() => assertConcurrencyTokenPresent("")).toThrow(ConflictError);
    expect(() => assertConcurrencyTokenPresent("2026-03-18T12:00:00.000Z")).not.toThrow();
  });
});

import { describe, expect, it, vi } from "vitest";
import type { Context } from "hono";
import { runIdempotentCommand } from "./idempotency";
import type { HonoEnv } from "../types/hono";
import { IdempotencyService } from "../application/shared/idempotency-service";
import { InMemoryIdempotencyRepository } from "../infrastructure/persistence/in-memory/in-memory-idempotency-repository";

function createMockContext(input?: {
  actorPersonId?: string;
  idempotencyKey?: string;
}) {
  const idempotencyService = new IdempotencyService(
    new InMemoryIdempotencyRepository(),
  );

  const store = new Map<string, unknown>();
  store.set("requestContext", {
    requestId: "req-1",
    actor: input?.actorPersonId
      ? {
          personId: input.actorPersonId,
          isAuthenticated: true,
          roles: [],
          authorityGrants: [],
        }
      : null,
  });
  store.set("appContext", {
    handlers: {} as never,
    services: {
      transactionRunner: {
        runInTransaction: async <T>(work: () => Promise<T>) => work(),
      },
      idempotencyService,
    },
  });

  const c = {
    get(key: string) {
      return store.get(key);
    },
    req: {
      header(name: string) {
        if (name.toLowerCase() === "idempotency-key") {
          return input?.idempotencyKey;
        }
        return undefined;
      },
    },
  } as unknown as Context<HonoEnv>;

  return { c, idempotencyService };
}

describe("runIdempotentCommand", () => {
  it("executes normally when no idempotency key is present", async () => {
    const { c } = createMockContext({ actorPersonId: "person-1" });
    const execute = vi.fn(async () => ({ ok: true }));

    const result = await runIdempotentCommand({
      c,
      operationName: "proposal.createDraft",
      requestBody: { title: "A" },
      execute,
    });

    expect(result).toEqual({ ok: true });
    expect(execute).toHaveBeenCalledTimes(1);
  });

  it("replays the stored response on repeated calls", async () => {
    const { c } = createMockContext({
      actorPersonId: "person-1",
      idempotencyKey: "idem-1",
    });

    const execute1 = vi.fn(async () => ({ id: "proposal-1" }));

    const first = await runIdempotentCommand({
      c,
      operationName: "proposal.createDraft",
      requestBody: { title: "A" },
      execute: execute1,
    });

    const execute2 = vi.fn(async () => ({ id: "proposal-2" }));

    const second = await runIdempotentCommand({
      c,
      operationName: "proposal.createDraft",
      requestBody: { title: "A" },
      execute: execute2,
    });

    expect(first).toEqual({ id: "proposal-1" });
    expect(second).toEqual({ id: "proposal-1" });
    expect(execute1).toHaveBeenCalledTimes(1);
    expect(execute2).toHaveBeenCalledTimes(0);
  });
});
