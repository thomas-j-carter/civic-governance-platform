export class InMemoryRuleRepository implements RuleRepository {
  private readonly rules = new Map<string, GovernanceRule>();

  async createRule(rule: GovernanceRule): Promise<void> {
    this.rules.set(rule.id, structuredClone(rule));
  }

  async getRuleById(ruleId: string): Promise<GovernanceRule | null> {
    const rule = this.rules.get(ruleId);
    return rule ? structuredClone(rule) : null;
  }

  async listRules(): Promise<GovernanceRule[]> {
    return [...this.rules.values()].map((rule) => structuredClone(rule));
  }
}

export class InMemoryRuleVersionRepository implements RuleVersionRepository {
  private readonly versions = new Map<string, RuleVersion>();

  async createRuleVersion(version: RuleVersion): Promise<void> {
    this.versions.set(version.id, structuredClone(version));
  }

  async updateRuleVersion(version: RuleVersion): Promise<void> {
    this.versions.set(version.id, structuredClone(version));
  }

  async getRuleVersionById(ruleVersionId: string): Promise<RuleVersion | null> {
    const version = this.versions.get(ruleVersionId);
    return version ? structuredClone(version) : null;
  }

  async listRuleVersions(governanceRuleId: string): Promise<RuleVersion[]> {
    return [...this.versions.values()]
      .filter((version) => version.governanceRuleId === governanceRuleId)
      .sort((a, b) => a.effectiveFrom.localeCompare(b.effectiveFrom))
      .map((version) => structuredClone(version));
  }
}

export class InMemoryRuleVersionScopeRepository implements RuleVersionScopeRepository {
  private readonly scopes = new Map<string, RuleVersionScope>();

  async replaceScopes(ruleVersionId: string, scopes: RuleVersionScope[]): Promise<void> {
    for (const [key, value] of [...this.scopes.entries()]) {
      if (value.ruleVersionId === ruleVersionId) {
        this.scopes.delete(key);
      }
    }

    for (const scope of scopes) {
      this.scopes.set(scope.id, structuredClone(scope));
    }
  }

  async listScopes(ruleVersionId: string): Promise<RuleVersionScope[]> {
    return [...this.scopes.values()]
      .filter((scope) => scope.ruleVersionId === ruleVersionId)
      .map((scope) => structuredClone(scope));
  }
}

export class InMemoryRuleResolutionRepository implements RuleResolutionRepository {
  constructor(
    private readonly ruleRepository: InMemoryRuleRepository,
    private readonly versionRepository: InMemoryRuleVersionRepository,
    private readonly scopeRepository: InMemoryRuleVersionScopeRepository,
  ) {}

  async resolveApplicableRuleVersion(input: ResolveRuleVersionInput) {
    const rules = await this.ruleRepository.listRules();
    const matchingRules = rules.filter((rule) => rule.ruleType === input.ruleType);

    const at = input.at;

    const candidates: Array<{
      rule: GovernanceRule;
      version: RuleVersion;
      scopes: RuleVersionScope[];
      score: number;
    }> = [];

    for (const rule of matchingRules) {
      const versions = await this.versionRepository.listRuleVersions(rule.id);

      for (const version of versions) {
        if (version.status !== "ACTIVE") {
          continue;
        }

        if (version.effectiveFrom > at) {
          continue;
        }

        if (version.effectiveTo && version.effectiveTo < at) {
          continue;
        }

        const scopes = await this.scopeRepository.listScopes(version.id);

        let score = 0;

        if (scopes.length === 0) {
          score = 1;
        } else {
          const matched = scopes.some((scope) => {
            if (!input.scope?.scopeType) {
              return scope.scopeType === "GLOBAL";
            }

            if (scope.scopeType !== input.scope.scopeType) {
              return false;
            }

            if (!scope.scopeId) {
              return true;
            }

            return scope.scopeId === input.scope.scopeId;
          });

          if (!matched) {
            continue;
          }

          score = scopes.some((scope) => scope.scopeId) ? 3 : 2;
        }

        candidates.push({ rule, version, scopes, score });
      }
    }

    candidates.sort((a, b) => {
      if (b.score !== a.score) {
        return b.score - a.score;
      }
      return b.version.effectiveFrom.localeCompare(a.version.effectiveFrom);
    });

    const selected = candidates[0];
    return selected
      ? {
          rule: selected.rule,
          version: selected.version,
          scopes: selected.scopes,
        }
      : null;
  }
}

export class PrismaRuleResolutionRepository implements RuleResolutionRepository {
  constructor(private readonly prisma: PrismaClient) {}

  async resolveApplicableRuleVersion(input: ResolveRuleVersionInput) {
    const records = await this.prisma.ruleVersion.findMany({
      where: {
        status: "ACTIVE",
        effectiveFrom: { lte: new Date(input.at) },
        OR: [
          { effectiveTo: null },
          { effectiveTo: { gte: new Date(input.at) } },
        ],
        governanceRule: {
          ruleType: input.ruleType,
        },
      },
      include: {
        governanceRule: true,
        scopes: true,
      },
      orderBy: {
        effectiveFrom: "desc",
      },
    });

    const scored = records
      .map((record) => {
        const scopes = record.scopes.map(mapRuleVersionScopeRecordToDomain);

        let score = 0;

        if (scopes.length === 0) {
          score = 1;
        } else {
          const matched = scopes.some((scope) => {
            if (!input.scope?.scopeType) {
              return scope.scopeType === "GLOBAL";
            }

            if (scope.scopeType !== input.scope.scopeType) {
              return false;
            }

            if (!scope.scopeId) {
              return true;
            }

            return scope.scopeId === input.scope.scopeId;
          });

          if (!matched) {
            return null;
          }

          score = scopes.some((scope) => scope.scopeId) ? 3 : 2;
        }

        return {
          rule: mapGovernanceRuleRecordToDomain(record.governanceRule),
          version: mapRuleVersionRecordToDomain(record),
          scopes,
          score,
        };
      })
      .filter((value): value is NonNullable<typeof value> => Boolean(value))
      .sort((a, b) => {
        if (b.score !== a.score) {
          return b.score - a.score;
        }
        return b.version.effectiveFrom.localeCompare(a.version.effectiveFrom);
      });

    const selected = scored[0];
    return selected
      ? {
          rule: selected.rule,
          version: selected.version,
          scopes: selected.scopes,
        }
      : null;
  }
}

import { NotFoundError } from "../../http/errors";
import type { RuleType, RuleScopeType } from "../../domain/rules/rule-types";
import type { RuleResolutionRepository } from "../../domain/rules/rule-repository";

export interface ResolvedRuleVersion {
  ruleId: string;
  ruleCode: string;
  ruleType: RuleType;
  ruleVersionId: string;
  versionLabel: string;
  parameters?: Record<string, unknown>;
}

export class RuleResolutionService {
  constructor(private readonly repository: RuleResolutionRepository) {}

  async resolveOrThrow(input: {
    ruleType: RuleType;
    at: string;
    scope?: {
      scopeType?: RuleScopeType;
      scopeId?: string;
    };
  }): Promise<ResolvedRuleVersion> {
    const resolved = await this.repository.resolveApplicableRuleVersion(input);

    if (!resolved) {
      throw new NotFoundError(
        `No active rule version found for rule type ${input.ruleType}.`,
        {
          code: "rule_version_not_found",
          type: "https://gov-api.ardtiresociety.org/problems/rule-version-not-found",
        },
      );
    }

    return {
      ruleId: resolved.rule.id,
      ruleCode: resolved.rule.code,
      ruleType: resolved.rule.ruleType,
      ruleVersionId: resolved.version.id,
      versionLabel: resolved.version.versionLabel,
      parameters: resolved.version.parametersJson
        ? (JSON.parse(resolved.version.parametersJson) as Record<string, unknown>)
        : undefined,
    };
  }
}

import { NotFoundError } from "../../http/errors";
import type {
  RuleRepository,
  RuleVersionRepository,
  RuleVersionScopeRepository,
} from "../../domain/rules/rule-repository";
import type { ListRuleVersionsQuery, ListRuleVersionsQueryHandler, RuleVersionDto } from "./list-rule-versions";
import { toRuleVersionDto } from "./rule-mappers";

export class ListRuleVersionsHandler implements ListRuleVersionsQueryHandler {
  constructor(
    private readonly ruleRepository: RuleRepository,
    private readonly versionRepository: RuleVersionRepository,
    private readonly scopeRepository: RuleVersionScopeRepository,
  ) {}

  async execute(query: ListRuleVersionsQuery): Promise<RuleVersionDto[]> {
    const rule = await this.ruleRepository.getRuleById(query.ruleId);

    if (!rule) {
      throw new NotFoundError(`Rule ${query.ruleId} was not found.`);
    }

    const versions = await this.versionRepository.listRuleVersions(query.ruleId);

    const output: RuleVersionDto[] = [];
    for (const version of versions) {
      const scopes = await this.scopeRepository.listScopes(version.id);
      output.push(toRuleVersionDto(version, scopes));
    }

    return output;
  }
}

import { NotFoundError } from "../../http/errors";
import { assertCertificationCanBeCreated } from "../../domain/certifications/certification-rules";
import type { BallotRepository } from "../../domain/ballots/ballot-repository";
import type { CertificationRepository } from "../../domain/certifications/certification-repository";
import type { AuditWriter } from "../audit/audit-writer";
import type { RuleResolutionService } from "../rules/rule-resolution-service";
import type {
  CreateCertificationCommand,
  CreateCertificationCommandHandler,
  CertificationDetailDto,
} from "./create-certification";
import { toCertificationDetailDto } from "./certification-mappers";

export class CreateCertificationHandler implements CreateCertificationCommandHandler {
  constructor(
    private readonly ballotRepository: BallotRepository,
    private readonly certificationRepository: CertificationRepository,
    private readonly auditWriter: AuditWriter,
    private readonly ruleResolutionService: RuleResolutionService,
  ) {}

  async execute(command: CreateCertificationCommand): Promise<CertificationDetailDto> {
    const ballot = await this.ballotRepository.getBallotById(command.input.ballotId);
    if (!ballot) {
      throw new NotFoundError(`Ballot ${command.input.ballotId} was not found.`);
    }

    const existing = await this.certificationRepository.findCertificationByBallotId(
      command.input.ballotId,
    );
    assertCertificationCanBeCreated(existing);

    const at = new Date().toISOString();

    const quorumRule = await this.ruleResolutionService.resolveOrThrow({
      ruleType: "QUORUM",
      at,
      scope: {
        scopeType: "GLOBAL",
      },
    });

    const thresholdRule = await this.ruleResolutionService.resolveOrThrow({
      ruleType: "THRESHOLD",
      at,
      scope: {
        scopeType: "GLOBAL",
      },
    });

    const certificationRule = await this.ruleResolutionService.resolveOrThrow({
      ruleType: "CERTIFICATION",
      at,
      scope: {
        scopeType: "GLOBAL",
      },
    });

    const record = {
      id: crypto.randomUUID(),
      ballotId: command.input.ballotId,
      status: "PENDING" as const,
      notes: command.input.notes,
      quorumRuleVersionId: quorumRule.ruleVersionId,
      thresholdRuleVersionId: thresholdRule.ruleVersionId,
      certificationRuleVersionId: certificationRule.ruleVersionId,
      createdAt: at,
      updatedAt: at,
    };

    await this.certificationRepository.createCertification(record);

    await this.auditWriter.write({
      actionType: "CREATE",
      entityType: "CertificationRecord",
      entityId: record.id,
      actorPersonId: command.actor.personId,
      ruleVersionIds: [
        quorumRule.ruleVersionId,
        thresholdRule.ruleVersionId,
        certificationRule.ruleVersionId,
      ],
      summary: "Certification record created.",
      metadata: {
        ballotId: record.ballotId,
        quorumRuleCode: quorumRule.ruleCode,
        thresholdRuleCode: thresholdRule.ruleCode,
        certificationRuleCode: certificationRule.ruleCode,
      },
    });

    return toCertificationDetailDto(record);
  }
}

  app.get("/", requireAuth, requireAuthority("rule.read"), async (c) => {
    const appContext = c.get("appContext");
    const result = await appContext.handlers.rules.list.execute();
    return c.json(jsonData(result), 200);
  });

  app.get("/:ruleId/versions", requireAuth, requireAuthority("rule.read"), async (c) => {
    const appContext = c.get("appContext");
    const ruleId = c.req.param("ruleId");
    const result = await appContext.handlers.rules.listVersions.execute({ ruleId });
    return c.json(jsonData(result), 200);
  });

  return app;
}
