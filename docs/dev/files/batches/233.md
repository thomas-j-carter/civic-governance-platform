6. apps/gov-api/src/application/records/promote-proposal-to-record-handler.ts

import { NotFoundError } from "../../http/errors";
import type { ProposalRepository } from "../../domain/proposals/proposal-repository";
import type { OfficialRecordRepository } from "../../domain/records/record-repository";
import type { AuditWriter } from "../audit/audit-writer";
import type {
  PromoteProposalToRecordCommand,
  PromoteProposalToRecordCommandHandler,
} from "./promote-proposal-to-record";
import type { OfficialRecordDto } from "./create-record";
import { toOfficialRecordDto } from "./record-promotion-mappers";

export class PromoteProposalToRecordHandler
  implements PromoteProposalToRecordCommandHandler
{
  constructor(
    private readonly proposalRepository: ProposalRepository,
    private readonly officialRecordRepository: OfficialRecordRepository,
    private readonly auditWriter: AuditWriter,
  ) {}

  async execute(
    command: PromoteProposalToRecordCommand,
  ): Promise<OfficialRecordDto> {
    const proposal = await this.proposalRepository.getProposalById(
      command.input.proposalId,
    );

    if (!proposal) {
      throw new NotFoundError(`Proposal ${command.input.proposalId} was not found.`);
    }

    const now = new Date().toISOString();

    const record = {
      id: crypto.randomUUID(),
      recordType: command.input.recordType,
      title: command.input.title,
      summary: command.input.summary,
      sourceEntityType: "proposal",
      sourceEntityId: proposal.id,
      status: "DRAFT" as const,
      createdAt: now,
      updatedAt: now,
    };

    await this.officialRecordRepository.createRecord(record);

    await this.auditWriter.write({
      actionType: "CREATE",
      entityType: "OfficialRecord",
      entityId: record.id,
      actorPersonId: command.actor.personId,
      summary: "Proposal promoted to official record draft.",
      metadata: {
        proposalId: proposal.id,
        recordType: record.recordType,
      },
    });

    return toOfficialRecordDto(record);
  }
}

7. apps/gov-api/src/application/records/officialize-record-handler.ts

import { ConflictError, NotFoundError } from "../../http/errors";
import type { OfficialRecordRepository } from "../../domain/records/record-repository";
import type { AuditWriter } from "../audit/audit-writer";
import type {
  OfficializeRecordCommand,
  OfficializeRecordCommandHandler,
} from "./officialize-record";
import type { OfficialRecordDto } from "./create-record";
import { toOfficialRecordDto } from "./record-promotion-mappers";

export class OfficializeRecordHandler implements OfficializeRecordCommandHandler {
  constructor(
    private readonly officialRecordRepository: OfficialRecordRepository,
    private readonly auditWriter: AuditWriter,
  ) {}

  async execute(command: OfficializeRecordCommand): Promise<OfficialRecordDto> {
    const record = await this.officialRecordRepository.getRecordById(command.recordId);

    if (!record) {
      throw new NotFoundError(`Official record ${command.recordId} was not found.`);
    }

    if (record.status === "OFFICIAL") {
      throw new ConflictError("Record is already officialized.", {
        code: "record_already_officialized",
      });
    }

    record.status = "OFFICIAL";
    record.officializedAt = command.input.officializedAt ?? new Date().toISOString();
    record.updatedAt = new Date().toISOString();

    await this.officialRecordRepository.updateRecord(record);

    await this.auditWriter.write({
      actionType: "APPROVE",
      entityType: "OfficialRecord",
      entityId: record.id,
      actorPersonId: command.actor.personId,
      summary: "Official record officialized.",
      metadata: {
        note: command.input.note,
        officializedAt: record.officializedAt,
      },
    });

    return toOfficialRecordDto(record);
  }
}

8. apps/gov-api/src/application/gazette/add-record-to-gazette-issue-handler.ts

import { ConflictError, NotFoundError } from "../../http/errors";
import type { GazetteEntryRepository, GazetteIssueRepository } from "../../domain/gazette/gazette-repository";
import type { OfficialRecordRepository } from "../../domain/records/record-repository";
import type { AuditWriter } from "../audit/audit-writer";
import type {
  AddRecordToGazetteIssueCommand,
  AddRecordToGazetteIssueCommandHandler,
  GazetteEntryDto,
} from "./add-record-to-gazette-issue";
import { toGazetteEntryDto } from "../records/record-promotion-mappers";

export class AddRecordToGazetteIssueHandler
  implements AddRecordToGazetteIssueCommandHandler
{
  constructor(
    private readonly gazetteIssueRepository: GazetteIssueRepository,
    private readonly gazetteEntryRepository: GazetteEntryRepository,
    private readonly officialRecordRepository: OfficialRecordRepository,
    private readonly auditWriter: AuditWriter,
  ) {}

  async execute(
    command: AddRecordToGazetteIssueCommand,
  ): Promise<GazetteEntryDto> {
    const issue = await this.gazetteIssueRepository.getIssueById(
      command.input.gazetteIssueId,
    );

    if (!issue) {
      throw new NotFoundError(
        `Gazette issue ${command.input.gazetteIssueId} was not found.`,
      );
    }

    const record = await this.officialRecordRepository.getRecordById(
      command.input.officialRecordId,
    );

    if (!record) {
      throw new NotFoundError(
        `Official record ${command.input.officialRecordId} was not found.`,
      );
    }

    if (issue.publicationState === "PUBLISHED") {
      throw new ConflictError("Cannot add entries to a published gazette issue.", {
        code: "gazette_issue_already_published",
      });
    }

    const existingEntries = await this.gazetteEntryRepository.listEntriesByIssueId(issue.id);

    const publicationOrder =
      command.input.publicationOrder ??
      (existingEntries.length === 0
        ? 1
        : Math.max(...existingEntries.map((entry) => entry.publicationOrder)) + 1);

    const now = new Date().toISOString();

    const entry = {
      id: crypto.randomUUID(),
      gazetteIssueId: issue.id,
      officialRecordId: record.id,
      titleSnapshot: record.title,
      summarySnapshot: record.summary,
      publicationOrder,
      createdAt: now,
    };

    await this.gazetteEntryRepository.createEntry(entry);

    await this.auditWriter.write({
      actionType: "PUBLISH",
      entityType: "GazetteEntry",
      entityId: entry.id,
      actorPersonId: command.actor.personId,
      summary: "Official record added to gazette issue.",
      metadata: {
        gazetteIssueId: issue.id,
        officialRecordId: record.id,
        publicationOrder,
      },
    });

    return toGazetteEntryDto(entry);
  }
}

import { Hono } from "hono";
import type { HonoEnv } from "../types/hono";
import { jsonData } from "../http/json";
import { requireAuth } from "../middleware/require-auth";
import { requireAuthority } from "../middleware/require-authority";
import { ValidationError } from "../http/errors";
import { runIdempotentCommand } from "../http/idempotency";

export function createRecordPromotionRoutes(): Hono<HonoEnv> {
  const app = new Hono<HonoEnv>();

  app.post(
    "/promotions/proposal-to-record",
    requireAuth,
    requireAuthority("record.promote"),
    async (c) => {
      const appContext = c.get("appContext");
      const actor = c.get("requestContext").actor!;
      const body = await c.req.json<{
        proposalId?: string;
        recordType?: string;
        title?: string;
        summary?: string;
      }>();

      const errors: Record<string, string[]> = {};
      if (!body.proposalId?.trim()) errors.proposalId = ["proposalId is required."];
      if (!body.recordType?.trim()) errors.recordType = ["recordType is required."];
      if (!body.title?.trim()) errors.title = ["title is required."];

      if (Object.keys(errors).length > 0) {
        throw new ValidationError("Promote proposal to record request was invalid.", errors);
      }

      const requestPayload = {
        proposalId: body.proposalId!,
        recordType: body.recordType!,
        title: body.title!,
        summary: body.summary,
      };

      const result = await runIdempotentCommand({
        c,
        operationName: "record.promoteProposalToRecord",
        requestBody: requestPayload,
        execute: async () =>
          appContext.handlers.recordPromotion.promoteProposalToRecord.execute({
            actor: { personId: actor.personId },
            input: requestPayload,
          }),
      });

      return c.json(jsonData(result), 201);
    },
  );

  app.post(
    "/:recordId/actions/officialize",
    requireAuth,
    requireAuthority("record.officialize"),
    async (c) => {
      const appContext = c.get("appContext");
      const actor = c.get("requestContext").actor!;
      const recordId = c.req.param("recordId");
      const body = await c.req.json<{ officializedAt?: string; note?: string }>().catch(() => ({}));

      const result = await appContext.handlers.recordPromotion.officializeRecord.execute({
        actor: { personId: actor.personId },
        recordId,
        input: {
          officializedAt: body.officializedAt,
          note: body.note,
        },
      });

      return c.json(jsonData(result), 200);
    },
  );

  return app;
}
