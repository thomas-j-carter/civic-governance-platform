import { ConflictError } from "../../http/errors";
import type { Ballot } from "./ballot";
import type { BallotEligibilitySnapshotEntry } from "./ballot-eligibility";

export function assertBallotCanBeOpened(ballot: Ballot): void {
  if (!["DRAFT", "SCHEDULED"].includes(ballot.state)) {
    throw new ConflictError(
      `Ballot cannot be opened from current state ${ballot.state}.`,
      {
        type: "https://gov-api.ardtiresociety.org/problems/invalid-state-transition",
        code: "invalid_state_transition",
      },
    );
  }
}

export function assertBallotCanBeClosed(ballot: Ballot): void {
  if (ballot.state !== "OPEN") {
    throw new ConflictError(
      `Ballot cannot be closed from current state ${ballot.state}.`,
      {
        type: "https://gov-api.ardtiresociety.org/problems/invalid-state-transition",
        code: "invalid_state_transition",
      },
    );
  }
}

export function assertBallotCanBeCancelled(ballot: Ballot): void {
  if (["CLOSED", "RESULT_COMPUTED", "CANCELLED"].includes(ballot.state)) {
    throw new ConflictError(
      `Ballot cannot be cancelled from current state ${ballot.state}.`,
      {
        type: "https://gov-api.ardtiresociety.org/problems/invalid-state-transition",
        code: "invalid_state_transition",
      },
    );
  }
}

export function assertBallotIsOpen(ballot: Ballot): void {
  if (ballot.state !== "OPEN") {
    throw new ConflictError("Ballot is not open for voting.", {
      type: "https://gov-api.ardtiresociety.org/problems/ballot-closed",
      code: "ballot_closed",
    });
  }
}

export function assertEligibleToVote(
  eligibility: BallotEligibilitySnapshotEntry | null,
): void {
  if (!eligibility || eligibility.eligibilityStatus !== "ELIGIBLE") {
    throw new ConflictError("Member is not eligible to vote on this ballot.", {
      type: "https://gov-api.ardtiresociety.org/problems/eligibility-failed",
      code: "eligibility_failed",
    });
  }
}

export function assertNoDuplicateVote(existingVote: unknown): void {
  if (existingVote) {
    throw new ConflictError("Member has already cast a vote for this ballot.", {
      type: "https://gov-api.ardtiresociety.org/problems/duplicate-vote",
      code: "duplicate_vote",
    });
  }
}

8. apps/gov-api/src/infrastructure/persistence/in-memory/in-memory-ballot-repository.ts

import type {
  BallotEligibilityRepository,
  BallotRepository,
  ListBallotsInput,
  ListBallotsResult,
  VoteRepository,
  VoteTallyRepository,
} from "../../../domain/ballots/ballot-repository";
import type { Ballot } from "../../../domain/ballots/ballot";
import type { BallotEligibilitySnapshotEntry } from "../../../domain/ballots/ballot-eligibility";
import type { Vote } from "../../../domain/ballots/vote";
import type { VoteTally } from "../../../domain/ballots/vote-tally";

export class InMemoryBallotRepository implements BallotRepository {
  private readonly ballots = new Map<string, Ballot>();

  async createBallot(input: Ballot): Promise<void> {
    this.ballots.set(input.id, structuredClone(input));
  }

  async updateBallot(input: Ballot): Promise<void> {
    this.ballots.set(input.id, structuredClone(input));
  }

  async getBallotById(ballotId: string): Promise<Ballot | null> {
    const ballot = this.ballots.get(ballotId);
    return ballot ? structuredClone(ballot) : null;
  }

  async listBallots(input: ListBallotsInput): Promise<ListBallotsResult> {
    let items = [...this.ballots.values()];

    if (input.state) {
      items = items.filter((item) => item.state === input.state);
    }

    items.sort((a, b) => b.createdAt.localeCompare(a.createdAt));

    const total = items.length;
    const totalPages = Math.max(1, Math.ceil(total / input.limit));
    const offset = (input.page - 1) * input.limit;
    const paged = items.slice(offset, offset + input.limit).map((item) => structuredClone(item));

    return {
      items: paged,
      page: input.page,
      limit: input.limit,
      total,
      totalPages,
    };
  }
}

export class InMemoryBallotEligibilityRepository implements BallotEligibilityRepository {
  private readonly entries = new Map<string, BallotEligibilitySnapshotEntry>();

  async replaceSnapshot(
    ballotId: string,
    entries: BallotEligibilitySnapshotEntry[],
  ): Promise<void> {
    for (const [key, value] of [...this.entries.entries()]) {
      if (value.ballotId === ballotId) {
        this.entries.delete(key);
      }
    }

    for (const entry of entries) {
      this.entries.set(entry.id, structuredClone(entry));
    }
  }

  async listSnapshot(ballotId: string): Promise<BallotEligibilitySnapshotEntry[]> {
    return [...this.entries.values()]
      .filter((entry) => entry.ballotId === ballotId)
      .sort((a, b) => a.memberId.localeCompare(b.memberId))
      .map((entry) => structuredClone(entry));
  }

  async getEligibilityForMember(
    ballotId: string,
    memberId: string,
  ): Promise<BallotEligibilitySnapshotEntry | null> {
    const entry = [...this.entries.values()].find(
      (item) => item.ballotId === ballotId && item.memberId === memberId,
    );

    return entry ? structuredClone(entry) : null;
  }
}

export class InMemoryVoteRepository implements VoteRepository {
  private readonly votes = new Map<string, Vote>();

  async createVote(input: Vote): Promise<void> {
    this.votes.set(input.id, structuredClone(input));
  }

  async listVotes(ballotId: string): Promise<Vote[]> {
    return [...this.votes.values()]
      .filter((vote) => vote.ballotId === ballotId)
      .sort((a, b) => a.castAt.localeCompare(b.castAt))
      .map((vote) => structuredClone(vote));
  }

  async getVoteByBallotAndMember(ballotId: string, memberId: string): Promise<Vote | null> {
    const vote = [...this.votes.values()].find(
      (item) => item.ballotId === ballotId && item.memberId === memberId,
    );

    return vote ? structuredClone(vote) : null;
  }
}

export class InMemoryVoteTallyRepository implements VoteTallyRepository {
  private readonly tallies = new Map<string, VoteTally>();

  async upsertTally(input: VoteTally): Promise<void> {
    this.tallies.set(input.ballotId, structuredClone(input));
  }

  async getTallyByBallotId(ballotId: string): Promise<VoteTally | null> {
    const tally = this.tallies.get(ballotId);
    return tally ? structuredClone(tally) : null;
  }
}

import { NotFoundError } from "../../http/errors";
import { assertBallotCanBeOpened } from "../../domain/ballots/ballot-rules";
import type {
  BallotEligibilityRepository,
  BallotRepository,
} from "../../domain/ballots/ballot-repository";
import type { OpenBallotCommand, OpenBallotCommandHandler } from "./open-ballot";
import type { BallotDetailDto } from "./create-ballot";
import { toBallotDetailDto } from "./ballot-mappers";

export class OpenBallotHandler implements OpenBallotCommandHandler {
  constructor(
    private readonly ballotRepository: BallotRepository,
    private readonly ballotEligibilityRepository: BallotEligibilityRepository,
  ) {}

  async execute(command: OpenBallotCommand): Promise<BallotDetailDto> {
    const ballot = await this.ballotRepository.getBallotById(command.ballotId);

        ballotId: ballot.id,
        memberId: "dev-member-id",
        eligibilityStatus: "ELIGIBLE",
        snapshotAt: now,
        createdAt: now,
      },
    ]);

    return toBallotDetailDto(ballot);
  }
}

24. apps/gov-api/src/application/ballots/close-ballot-handler.ts

import { NotFoundError } from "../../http/errors";
import { assertBallotCanBeClosed } from "../../domain/ballots/ballot-rules";
import type {
  BallotRepository,
  VoteRepository,
  VoteTallyRepository,
} from "../../domain/ballots/ballot-repository";
import type { CloseBallotCommand, CloseBallotCommandHandler } from "./close-ballot";
import type { BallotDetailDto } from "./create-ballot";
import { toBallotDetailDto } from "./ballot-mappers";

export class CloseBallotHandler implements CloseBallotCommandHandler {
  constructor(
    private readonly ballotRepository: BallotRepository,
    private readonly voteRepository: VoteRepository,
    private readonly voteTallyRepository: VoteTallyRepository,
  ) {}

  async execute(command: CloseBallotCommand): Promise<BallotDetailDto> {
    const ballot = await this.ballotRepository.getBallotById(command.ballotId);
    if (!ballot) {
      throw new NotFoundError(`Ballot ${command.ballotId} was not found.`);
    }

    assertBallotCanBeClosed(ballot);

    const now = new Date().toISOString();
    const votes = await this.voteRepository.listVotes(ballot.id);

    const yesCount = votes.filter((vote) => vote.choice === "YES").length;
    const noCount = votes.filter((vote) => vote.choice === "NO").length;
    const abstainCount = votes.filter((vote) => vote.choice === "ABSTAIN").length;
    const totalCount = votes.length;

    ballot.state = "CLOSED";
    ballot.closedAt = now;
    ballot.updatedAt = now;

    await this.ballotRepository.updateBallot(ballot);

    await this.voteTallyRepository.upsertTally({
      id: crypto.randomUUID(),
      ballotId: ballot.id,
      yesCount,
      noCount,
      abstainCount,
      totalCount,
      quorumMet: totalCount > 0,
      thresholdMet: yesCount > noCount,
      computedAt: now,
      createdAt: now,
    });

    return toBallotDetailDto(ballot);
  }
}

28. apps/gov-api/src/application/ballots/cast-vote-handler.ts

import { NotFoundError, ConflictError } from "../../http/errors";
import {
  assertBallotIsOpen,
  assertEligibleToVote,
  assertNoDuplicateVote,
} from "../../domain/ballots/ballot-rules";
import type {
  BallotEligibilityRepository,
  BallotRepository,
  VoteRepository,
} from "../../domain/ballots/ballot-repository";
import type { CastVoteCommand, CastVoteCommandHandler } from "./cast-vote";
import type { VoteDto } from "./list-votes";
import { toVoteDto } from "./ballot-mappers";

export class CastVoteHandler implements CastVoteCommandHandler {
  constructor(
    private readonly ballotRepository: BallotRepository,
    private readonly ballotEligibilityRepository: BallotEligibilityRepository,
    private readonly voteRepository: VoteRepository,
  ) {}

  async execute(command: CastVoteCommand): Promise<VoteDto> {
    const ballot = await this.ballotRepository.getBallotById(command.ballotId);
    if (!ballot) {
      throw new NotFoundError(`Ballot ${command.ballotId} was not found.`);
    }

    if (!command.actor.memberId) {
      throw new ConflictError("Authenticated actor must be a member to cast a vote.", {
        code: "member_context_required",
      });
    }

    assertBallotIsOpen(ballot);

    const eligibility = await this.ballotEligibilityRepository.getEligibilityForMember(
      ballot.id,
      command.actor.memberId,
    );
    assertEligibleToVote(eligibility);

    const existingVote = await this.voteRepository.getVoteByBallotAndMember(
      ballot.id,
      command.actor.memberId,
    );
    assertNoDuplicateVote(existingVote);

    const now = new Date().toISOString();

    const vote = {
      id: crypto.randomUUID(),
      ballotId: ballot.id,
      memberId: command.actor.memberId,
      personId: command.actor.personId,
      choice: command.input.choice,
      castAt: now,
      recordedAt: now,
      createdAt: now,
    };

    await this.voteRepository.createVote(vote);

    return toVoteDto(vote);
  }
}

export function createBallotRoutes(): Hono<HonoEnv> {
  const app = new Hono<HonoEnv>();

  app.get("/", requireAuth, async (c) => {

    });

    return c.json(
      jsonList(result.items, {
        page: result.page,
        limit: result.limit,
        total: result.total,
        totalPages: result.totalPages,
      }),
      200,
    );
  });

  app.post("/", requireAuth, requireAuthority("ballot.create"), async (c) => {
    const appContext = c.get("appContext");
    const requestContext = c.get("requestContext");
    const actor = requestContext.actor!;

    const body = await c.req.json<{
      proposalId?: string;
      title?: string;
      description?: string;
      scheduledOpenAt?: string;
      scheduledCloseAt?: string;
    }>();

    const errors: Record<string, string[]> = {};

    if (!body.title?.trim()) {
      errors.title = ["Title is required."];
    }

    if (Object.keys(errors).length > 0) {
      throw new ValidationError("Create ballot request was invalid.", errors);
    }

    const result = await appContext.handlers.ballots.create.execute({
      actor: {
        personId: actor.personId,
        memberId: actor.memberId,
      },
      input: {
        proposalId: body.proposalId,
        title: body.title!,
        description: body.description,
        scheduledOpenAt: body.scheduledOpenAt,
        scheduledCloseAt: body.scheduledCloseAt,
      },
    });

    return c.json(jsonData(result), 201);
  });

  app.get("/:ballotId", requireAuth, async (c) => {
    const appContext = c.get("appContext");
    const ballotId = c.req.param("ballotId");

    const result = await appContext.handlers.ballots.getDetail.execute({ ballotId });

    if (!result) {
      throw new NotFoundError(`Ballot ${ballotId} was not found.`);
    }

    return c.json(jsonData(result), 200);
  });

  app.post(
    "/:ballotId/actions/open",
    requireAuth,
    requireAuthority("ballot.open"),
    async (c) => {
      const appContext = c.get("appContext");
      const requestContext = c.get("requestContext");
      const actor = requestContext.actor!;
      const ballotId = c.req.param("ballotId");

      const result = await appContext.handlers.ballots.open.execute({
        actor: {
          personId: actor.personId,
          memberId: actor.memberId,

        ballotId,

    },
  );

      const requestContext = c.get("requestContext");
      const actor = requestContext.actor!;
      const ballotId = c.req.param("ballotId");

      const appContext = c.get("appContext");
      const requestContext = c.get("requestContext");
      const actor = requestContext.actor!;
      const ballotId = c.req.param("ballotId");

      const result = await appContext.handlers.ballots.cancel.execute({
        actor: {

        ballotId,
      });

      return c.json(jsonData(result), 200);
    },
  );

  app.get("/:ballotId/eligibility", requireAuth, async (c) => {
    const appContext = c.get("appContext");
    const ballotId = c.req.param("ballotId");

    const result = await appContext.handlers.ballots.listEligibility.execute({
      ballotId,
    });

    return c.json(jsonData(result), 200);
  });

  app.get("/:ballotId/votes", requireAuth, async (c) => {
    const appContext = c.get("appContext");
    const ballotId = c.req.param("ballotId");

    const result = await appContext.handlers.ballots.listVotes.execute({
      ballotId,
    });

    return c.json(jsonData(result), 200);
  });

  app.post("/:ballotId/votes", requireAuth, requireAuthority("vote.cast"), async (c) => {
    const appContext = c.get("appContext");
    const requestContext = c.get("requestContext");
    const actor = requestContext.actor!;
    const ballotId = c.req.param("ballotId");
