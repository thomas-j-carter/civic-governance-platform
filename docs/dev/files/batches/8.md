export function assertConcurrencyMatch(
  currentUpdatedAt: string,
  expectedUpdatedAt: string,

  if (currentUpdatedAt !== expectedUpdatedAt) {

      type: "https://gov-api.ardtiresociety.org/problems/concurrency-conflict",
      code: "concurrency_conflict",
    });
  }

apps/gov-api/src/infrastructure/persistence/in-memory/in-memory-idempotency-repository.ts

import type { IdempotencyRecord } from "../../../domain/idempotency/idempotency-record";
import type { IdempotencyRepository } from "../../../domain/idempotency/idempotency-repository";

export class InMemoryIdempotencyRepository implements IdempotencyRepository {
  private readonly records = new Map<string, IdempotencyRecord>();

  private buildCompositeKey(
    actorPersonId: string,
    operationName: string,
    idempotencyKey: string,
  ): string {
    return `${actorPersonId}:${operationName}:${idempotencyKey}`;
  }

  async getByKey(
    actorPersonId: string,
    operationName: string,
    idempotencyKey: string,
  ): Promise<IdempotencyRecord | null> {
    const key = this.buildCompositeKey(actorPersonId, operationName, idempotencyKey);
    const record = this.records.get(key);
    return record ? structuredClone(record) : null;
  }

  async create(record: IdempotencyRecord): Promise<void> {
    const key = this.buildCompositeKey(
      record.actorPersonId,
      record.operationName,
      record.idempotencyKey,
    );
    this.records.set(key, structuredClone(record));
  }
}

import type { PrismaClient } from "@prisma/client";
import type { IdempotencyRecord } from "../../../domain/idempotency/idempotency-record";
import type { IdempotencyRepository } from "../../../domain/idempotency/idempotency-repository";

export class PrismaIdempotencyRepository implements IdempotencyRepository {
  constructor(private readonly prisma: PrismaClient) {}

  async getByKey(
    actorPersonId: string,
    operationName: string,
    idempotencyKey: string,
  ): Promise<IdempotencyRecord | null> {
    const record = await this.prisma.idempotencyRecord.findFirst({
      where: {
        actorPersonId,
        operationName,
        idempotencyKey,
      },
    });

    if (!record) {
      return null;
    }

    return {
      id: record.id,
      idempotencyKey: record.idempotencyKey,
      actorPersonId: record.actorPersonId,
      operationName: record.operationName,
      requestHash: record.requestHash,
      responseJson: record.responseJson,
      createdAt: record.createdAt.toISOString(),
    };
  }

  async create(record: IdempotencyRecord): Promise<void> {
    await this.prisma.idempotencyRecord.create({
      data: {
        id: record.id,
        idempotencyKey: record.idempotencyKey,
        actorPersonId: record.actorPersonId,
        operationName: record.operationName,
        requestHash: record.requestHash,
        responseJson: record.responseJson,
        createdAt: new Date(record.createdAt),
      },
    });
  }
}

4. Idempotency service

apps/gov-api/src/application/shared/idempotency-service.ts

import { createHash } from "node:crypto";
import { ConflictError } from "../../http/errors";
import type { IdempotencyRepository } from "../../domain/idempotency/idempotency-repository";

export class IdempotencyService {
  constructor(private readonly repository: IdempotencyRepository) {}

  async getExistingResponse<T>(input: {
    actorPersonId: string;
    operationName: string;
    idempotencyKey?: string;
    requestBody: unknown;
  }): Promise<T | null> {
    if (!input.idempotencyKey) {
      return null;
    }

    const existing = await this.repository.getByKey(
      input.actorPersonId,
      input.operationName,
      input.idempotencyKey,
    );

    if (!existing) {
      return null;
    }

    const requestHash = hashRequestBody(input.requestBody);
    if (existing.requestHash !== requestHash) {
      throw new ConflictError(
        "Idempotency key has already been used with a different request payload.",
        {
          type: "https://gov-api.ardtiresociety.org/problems/idempotency-key-reused",
          code: "idempotency_key_reused",
        },
      );
    }

    return JSON.parse(existing.responseJson) as T;
  }

  async storeResponse<T>(input: {
    actorPersonId: string;
    operationName: string;
    idempotencyKey?: string;
    requestBody: unknown;
    response: T;
  }): Promise<void> {
    if (!input.idempotencyKey) {
      return;
    }

    await this.repository.create({
      id: crypto.randomUUID(),
      idempotencyKey: input.idempotencyKey,
      actorPersonId: input.actorPersonId,
      operationName: input.operationName,
      requestHash: hashRequestBody(input.requestBody),
      responseJson: JSON.stringify(input.response),
      createdAt: new Date().toISOString(),
    });
  }
}

function hashRequestBody(value: unknown): string {
  const normalized = JSON.stringify(value ?? null);
  return createHash("sha256").update(normalized).digest("hex");
}

6. Harden create proposal draft

We should make this atomic because it does three writes:

1. create proposal


2. create proposal version


3. update proposal currentVersionId



apps/gov-api/src/application/proposals/create-proposal-draft-handler.ts (hardened)

import type { ProposalRepository, ProposalVersionRepository } from "../../domain/proposals/proposal-repository";
import type { TransactionRunner } from "../shared/transaction";
import type { CreateProposalDraftCommand, CreateProposalDraftCommandHandler, ProposalDetailDto } from "./create-proposal-draft";
import { toProposalDetailDto } from "./proposal-mappers";

export class CreateProposalDraftHandler implements CreateProposalDraftCommandHandler {
  constructor(
    private readonly proposalRepository: ProposalRepository,
    private readonly proposalVersionRepository: ProposalVersionRepository,
    private readonly transactionRunner: TransactionRunner,
  ) {}

  async execute(command: CreateProposalDraftCommand): Promise<ProposalDetailDto> {
    return this.transactionRunner.runInTransaction(async () => {
      const proposalId = crypto.randomUUID();
      const versionId = crypto.randomUUID();
      const now = new Date().toISOString();

      const proposal = {
        id: proposalId,
        title: command.input.title,
        summary: command.input.summary,
        proposalType: (command.input.proposalType ?? "GENERAL") as ProposalDetailDto["proposalType"],
        currentStage: "DRAFT" as const,
        proposerPersonId: command.actor.personId,
        proposerMemberId: command.actor.memberId,
        currentVersionId: undefined,
        createdAt: now,
        updatedAt: now,
      };

      await this.proposalRepository.createProposal(proposal);

      const version = {
        id: versionId,
        proposalId,
        versionNumber: 1,
        titleSnapshot: command.input.title,
        bodyMarkdown: command.input.bodyMarkdown,
        createdByPersonId: command.actor.personId,
        createdAt: now,
      };

      await this.proposalVersionRepository.createProposalVersion(version);

      proposal.currentVersionId = versionId;
      proposal.updatedAt = new Date().toISOString();

      await this.proposalRepository.updateProposal(proposal);

      return toProposalDetailDto(proposal);
    });
  }
}

7. Harden open ballot

This should be atomic because it does:

1. transition ballot state


2. generate eligibility snapshot



apps/gov-api/src/application/ballots/open-ballot-handler.ts (hardened)

import { NotFoundError } from "../../http/errors";
import { assertBallotCanBeOpened } from "../../domain/ballots/ballot-rules";
import type {
  BallotEligibilityRepository,
  BallotRepository,
} from "../../domain/ballots/ballot-repository";
import type { TransactionRunner } from "../shared/transaction";
import type { OpenBallotCommand, OpenBallotCommandHandler } from "./open-ballot";
import type { BallotDetailDto } from "./create-ballot";
import { toBallotDetailDto } from "./ballot-mappers";

export class OpenBallotHandler implements OpenBallotCommandHandler {
  constructor(
    private readonly ballotRepository: BallotRepository,
    private readonly ballotEligibilityRepository: BallotEligibilityRepository,
    private readonly transactionRunner: TransactionRunner,
  ) {}

  async execute(command: OpenBallotCommand): Promise<BallotDetailDto> {
    return this.transactionRunner.runInTransaction(async () => {
      const ballot = await this.ballotRepository.getBallotById(command.ballotId);
      if (!ballot) {
        throw new NotFoundError(`Ballot ${command.ballotId} was not found.`);
      }

      assertBallotCanBeOpened(ballot);

      const now = new Date().toISOString();

      ballot.state = "OPEN";
      ballot.openedAt = now;
      ballot.updatedAt = now;

      await this.ballotRepository.updateBallot(ballot);

      await this.ballotEligibilityRepository.replaceSnapshot(ballot.id, [
        {
          id: crypto.randomUUID(),
          ballotId: ballot.id,
          memberId: "dev-member-id",
          eligibilityStatus: "ELIGIBLE",
          snapshotAt: now,
          createdAt: now,
        },
      ]);

      return toBallotDetailDto(ballot);
    });
  }
}

8. Harden close ballot

This should be atomic because it does:

1. read votes


2. compute tally


3. close ballot


4. persist tally



apps/gov-api/src/application/ballots/close-ballot-handler.ts (hardened)

import { NotFoundError } from "../../http/errors";
import { assertBallotCanBeClosed } from "../../domain/ballots/ballot-rules";
import type {
  BallotRepository,
  VoteRepository,
  VoteTallyRepository,
} from "../../domain/ballots/ballot-repository";
import type { TransactionRunner } from "../shared/transaction";
import type { CloseBallotCommand, CloseBallotCommandHandler } from "./close-ballot";
import type { BallotDetailDto } from "./create-ballot";
import { toBallotDetailDto } from "./ballot-mappers";

export class CloseBallotHandler implements CloseBallotCommandHandler {
  constructor(
    private readonly ballotRepository: BallotRepository,
    private readonly voteRepository: VoteRepository,
    private readonly voteTallyRepository: VoteTallyRepository,
    private readonly transactionRunner: TransactionRunner,
  ) {}

  async execute(command: CloseBallotCommand): Promise<BallotDetailDto> {
    return this.transactionRunner.runInTransaction(async () => {
      const ballot = await this.ballotRepository.getBallotById(command.ballotId);
      if (!ballot) {
        throw new NotFoundError(`Ballot ${command.ballotId} was not found.`);
      }

      assertBallotCanBeClosed(ballot);

      const now = new Date().toISOString();
      const votes = await this.voteRepository.listVotes(ballot.id);

      const yesCount = votes.filter((vote) => vote.choice === "YES").length;
      const noCount = votes.filter((vote) => vote.choice === "NO").length;
      const abstainCount = votes.filter((vote) => vote.choice === "ABSTAIN").length;
      const totalCount = votes.length;

      ballot.state = "CLOSED";
      ballot.closedAt = now;
      ballot.updatedAt = now;

      await this.ballotRepository.updateBallot(ballot);

      await this.voteTallyRepository.upsertTally({
        id: crypto.randomUUID(),
        ballotId: ballot.id,
        yesCount,
        noCount,
        abstainCount,
        totalCount,
        quorumMet: totalCount > 0,
        thresholdMet: yesCount > noCount,
        computedAt: now,
        createdAt: now,
      });

      return toBallotDetailDto(ballot);
    });
  }
}

import { NotFoundError } from "../../http/errors";
import { assertCertificationCanBeCertified } from "../../domain/certifications/certification-rules";
import type {
  CertificationRepository,
} from "../../domain/certifications/certification-repository";
import type { VoteTallyRepository } from "../../domain/ballots/ballot-repository";
import type { TransactionRunner } from "../shared/transaction";
import type {
  CertifyResultCommand,
  CertifyResultCommandHandler,
} from "./certify-result";
import type { CertificationDetailDto } from "./create-certification";
import { toCertificationDetailDto } from "./certification-mappers";

export class CertifyResultHandler implements CertifyResultCommandHandler {
  constructor(
    private readonly certificationRepository: CertificationRepository,
    private readonly voteTallyRepository: VoteTallyRepository,
    private readonly transactionRunner: TransactionRunner,
  ) {}

  async execute(command: CertifyResultCommand): Promise<CertificationDetailDto> {
    return this.transactionRunner.runInTransaction(async () => {
      const record = await this.certificationRepository.getCertificationById(command.certificationId);
      if (!record) {
        throw new NotFoundError(`Certification ${command.certificationId} was not found.`);
      }

      assertCertificationCanBeCertified(record);

      const tally = await this.voteTallyRepository.getTallyByBallotId(record.ballotId);
      if (!tally) {
        throw new NotFoundError(`No tally exists for ballot ${record.ballotId}.`);
      }

      record.status = "CERTIFIED";
      record.certifiedByPersonId = command.actor.personId;
      record.certifiedAt = new Date().toISOString();
      record.updatedAt = new Date().toISOString();

      await this.certificationRepository.updateCertification(record);

      return toCertificationDetailDto(record);
    });
  }
}

10. Idempotency helper for routes

A good first pass is route-level reuse for command handlers.

apps/gov-api/src/http/idempotency.ts

import type { Context } from "hono";
import type { HonoEnv } from "../types/hono";
import { UnauthorizedError } from "./errors";

export async function runIdempotentCommand<T>(input: {
  c: Context<HonoEnv>;
  operationName: string;
  requestBody: unknown;
  execute: () => Promise<T>;
}): Promise<T> {
  const requestContext = input.c.get("requestContext");
  const actor = requestContext.actor;

  if (!actor?.personId) {
    throw new UnauthorizedError();
  }

  const appContext = input.c.get("appContext");
  const idempotencyKey = input.c.req.header("idempotency-key") ?? undefined;

  const existing = await appContext.services.idempotencyService.getExistingResponse<T>({
    actorPersonId: actor.personId,
    operationName: input.operationName,
    idempotencyKey,
    requestBody: input.requestBody,
  });

  if (existing) {
    return existing;
  }

  const result = await input.execute();

  await appContext.services.idempotencyService.storeResponse({
    actorPersonId: actor.personId,
    operationName: input.operationName,
    idempotencyKey,
    requestBody: input.requestBody,
    response: result,
  });

  return result;
}
