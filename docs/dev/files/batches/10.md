import { z } from "zod";

const envSchema = z.object({
  NODE_ENV: z.enum(["development", "test", "production"]).default("development"),
  PORT: z.coerce.number().int().positive().default(3002),
  GOV_API_PERSISTENCE: z.enum(["memory", "prisma"]).default("memory"),
  DATABASE_URL: z.string().optional(),
  LOG_LEVEL: z.enum(["debug", "info", "warn", "error"]).default("info"),
  SERVICE_NAME: z.string().default("gov-api"),
  SERVICE_VERSION: z.string().default("0.1.0"),
});

export type AppEnv = z.infer<typeof envSchema>;

export function readEnv(source: NodeJS.ProcessEnv = process.env): AppEnv {
  const parsed = envSchema.safeParse(source);

  if (!parsed.success) {
    const issues = parsed.error.issues
      .map((issue) => `${issue.path.join(".")}: ${issue.message}`)
      .join("; ");
    throw new Error(`Invalid environment configuration: ${issues}`);
  }

  const env = parsed.data;

  if (env.GOV_API_PERSISTENCE === "prisma" && !env.DATABASE_URL) {
    throw new Error(
      "DATABASE_URL is required when GOV_API_PERSISTENCE=prisma.",
    );
  }

  return env;
}

export type LogLevel = "debug" | "info" | "warn" | "error";

export interface Logger {
  debug(message: string, meta?: Record<string, unknown>): void;
  info(message: string, meta?: Record<string, unknown>): void;
  warn(message: string, meta?: Record<string, unknown>): void;
  error(message: string, meta?: Record<string, unknown>): void;
}

const levelPriority: Record<LogLevel, number> = {
  debug: 10,
  info: 20,
  warn: 30,
  error: 40,
};

export function createLogger(level: LogLevel): Logger {
  function shouldLog(target: LogLevel): boolean {
    return levelPriority[target] >= levelPriority[level];
  }

  function write(
    target: LogLevel,
    message: string,
    meta?: Record<string, unknown>,
  ): void {
    if (!shouldLog(target)) {
      return;
    }

    const payload = {
      ts: new Date().toISOString(),
      level: target,
      message,
      ...(meta ? { meta } : {}),
    };

    const line = JSON.stringify(payload);

    if (target === "error") {
      console.error(line);
      return;
    }

    if (target === "warn") {
      console.warn(line);
      return;
    }

    console.log(line);
  }

  return {
    debug: (message, meta) => write("debug", message, meta),
    info: (message, meta) => write("info", message, meta),
    warn: (message, meta) => write("warn", message, meta),
    error: (message, meta) => write("error", message, meta),
  };
}

import type { MiddlewareHandler } from "hono";
import type { HonoEnv } from "../types/hono";
import type { Logger } from "./logger";

export function createRequestLoggingMiddleware(
  logger: Logger,
): MiddlewareHandler<HonoEnv> {
  return async (c, next) => {
    const startedAt = Date.now();
    const method = c.req.method;
    const path = c.req.path;

    await next();

    const durationMs = Date.now() - startedAt;
    const requestContext = c.get("requestContext");

    logger.info("request_completed", {
      requestId: requestContext.requestId,
      method,
      path,
      status: c.res.status,
      durationMs,
      actorPersonId: requestContext.actor?.personId,
    });
  };
}

export interface ReadinessCheckResult {
  ok: boolean;
  checks: Record<string, { ok: boolean; detail?: string }>;
}

export interface ReadinessDependency {
  name: string;
  check(): Promise<{ ok: boolean; detail?: string }>;
}

export async function runReadinessChecks(
  dependencies: ReadinessDependency[],
): Promise<ReadinessCheckResult> {
  const entries = await Promise.all(
    dependencies.map(async (dependency) => {
      const result = await dependency.check();
      return [dependency.name, result] as const;
    }),
  );

  const checks = Object.fromEntries(entries);
  const ok = Object.values(checks).every((check) => check.ok);

  return { ok, checks };
}

export function createPrismaReadinessDependency(
  prisma: PrismaClient,

    async check() {
      try {
        await prisma.$queryRaw`SELECT 1`;
        return { ok: true };
      } catch (error) {
        return {
          ok: false,
          detail: error instanceof Error ? error.message : String(error),
        };
      }
    },
  };

import { Hono } from "hono";
import type { AppContext } from "./context/app-context";
import type { HonoEnv } from "./types/hono";
import { requestContextMiddleware } from "./middleware/request-context";
import { createAppContextMiddleware } from "./middleware/app-context";
import { errorHandler } from "./middleware/error-handler";
import { createApiRoutes } from "./routes";
import type { Logger } from "./observability/logger";
import { createRequestLoggingMiddleware } from "./observability/request-logging";
import type { ReadinessDependency } from "./health/readiness";
import { runReadinessChecks } from "./health/readiness";

export interface CreateAppOptions {
  logger: Logger;
  readinessDependencies?: ReadinessDependency[];
  serviceName?: string;
  serviceVersion?: string;
}

export function createApp(
  appContext: AppContext,
  options: CreateAppOptions,
): Hono<HonoEnv> {
  const app = new Hono<HonoEnv>();

  app.use("*", requestContextMiddleware);
  app.use("*", createAppContextMiddleware(appContext));
  app.use("*", errorHandler);
  app.use("*", createRequestLoggingMiddleware(options.logger));

  app.get("/health", (c) => {
    return c.json(
      {
        ok: true,
        service: options.serviceName ?? "gov-api",
        version: options.serviceVersion ?? "0.1.0",
      },
      200,
    );
  });

  app.get("/ready", async (c) => {
    const result = await runReadinessChecks(options.readinessDependencies ?? []);

    return c.json(result, result.ok ? 200 : 503);
  });

  app.route("/api/v1", createApiRoutes());

  return app;
}

import { serve } from "@hono/node-server";
import { readEnv } from "./config/env";
import { createLogger } from "./observability/logger";
import { createApp } from "./app";
import { buildAppContext } from "./server/build-app-context";
import { createPrismaReadinessDependency } from "./infrastructure/prisma/prisma-readiness";

async function main(): Promise<void> {
  const env = readEnv();
  const logger = createLogger(env.LOG_LEVEL);
  const { appContext, prisma } = buildAppContext(env);

  const readinessDependencies = prisma
    ? [createPrismaReadinessDependency(prisma)]
    : [];

  const app = createApp(appContext, {
    logger,
    readinessDependencies,
    serviceName: env.SERVICE_NAME,
    serviceVersion: env.SERVICE_VERSION,
  });

  const server = serve({
    fetch: app.fetch,
    port: env.PORT,
  });

  logger.info("service_started", {
    service: env.SERVICE_NAME,
    version: env.SERVICE_VERSION,
    port: env.PORT,
    persistence: env.GOV_API_PERSISTENCE,
  });

  const shutdown = async (signal: string) => {
    logger.info("service_stopping", { signal });

    server.close();

    if (prisma) {
      await prisma.$disconnect();
    }

    logger.info("service_stopped", { signal });
    process.exit(0);
  };

  process.on("SIGINT", () => {
    void shutdown("SIGINT");
  });

  process.on("SIGTERM", () => {
    void shutdown("SIGTERM");
  });
}

void main();
