2. apps/gov-api/src/infrastructure/persistence/in-memory/in-memory-office-holder-admin-repository.ts

import type { OfficeHolder } from "../../../domain/governance/office-holder";
import type { OfficeHolderAdministrationRepository } from "../../../domain/governance/office-holder-admin-repository";
import type { InMemoryOfficeHolderRepository } from "./in-memory-authority-repository";

export class InMemoryOfficeHolderAdministrationRepository
  implements OfficeHolderAdministrationRepository
{
  constructor(
    private readonly repository: InMemoryOfficeHolderRepository & {
      items?: Map<string, OfficeHolder>;
    },
  ) {}

  async listOfficeHolders(): Promise<OfficeHolder[]> {
    const items = this.repository.items ? [...this.repository.items.values()] : [];

    return items
      .sort((a, b) => b.startedAt.localeCompare(a.startedAt))
      .map((item) => structuredClone(item));
  }

  async getOfficeHolderById(officeHolderId: string): Promise<OfficeHolder | null> {
    const item = this.repository.items?.get(officeHolderId);
    return item ? structuredClone(item) : null;
  }

  async updateOfficeHolder(officeHolder: OfficeHolder): Promise<void> {
    if ("items" in this.repository && this.repository.items) {
      this.repository.items.set(officeHolder.id, structuredClone(officeHolder));
      return;
    }

    throw new Error("In-memory office holder repository does not support updateOfficeHolder.");
  }
}

import type { PrismaClient } from "@prisma/client";
import type { OfficeHolder } from "../../../domain/governance/office-holder";
import type { OfficeHolderAdministrationRepository } from "../../../domain/governance/office-holder-admin-repository";
import { mapOfficeHolderRecordToDomain } from "../mappers/authority-mappers";

export class PrismaOfficeHolderAdministrationRepository
  implements OfficeHolderAdministrationRepository
{
  constructor(private readonly prisma: PrismaClient) {}

  async listOfficeHolders(): Promise<OfficeHolder[]> {
    const records = await this.prisma.officeHolder.findMany({
      orderBy: { startedAt: "desc" },
    });

    return records.map(mapOfficeHolderRecordToDomain);
  }

  async getOfficeHolderById(officeHolderId: string): Promise<OfficeHolder | null> {
    const record = await this.prisma.officeHolder.findUnique({
      where: { id: officeHolderId },
    });

    return record ? mapOfficeHolderRecordToDomain(record) : null;
  }

  async updateOfficeHolder(officeHolder: OfficeHolder): Promise<void> {
    await this.prisma.officeHolder.update({
      where: { id: officeHolder.id },
      data: {
        officeId: officeHolder.officeId,
        personId: officeHolder.personId,
        appointedAt: officeHolder.appointedAt ? new Date(officeHolder.appointedAt) : null,
        startedAt: new Date(officeHolder.startedAt),
        endedAt: officeHolder.endedAt ? new Date(officeHolder.endedAt) : null,
      },
    });
  }
}

11. apps/gov-api/src/application/governance/appoint-office-holder-handler.ts

import { NotFoundError } from "../../http/errors";
import type {
  OfficeHolderRepository,
  OfficeRepository,
} from "../../domain/authority/authority-repository";
import type { AuditWriter } from "../audit/audit-writer";
import type {
  AppointOfficeHolderCommand,
  AppointOfficeHolderCommandHandler,
} from "./appoint-office-holder";
import type { OfficeHolderDto } from "./office-holder-dto";
import { toOfficeHolderDto } from "./office-holder-mappers";

export class AppointOfficeHolderHandler
  implements AppointOfficeHolderCommandHandler
{
  constructor(
    private readonly officeRepository: OfficeRepository,
    private readonly officeHolderRepository: OfficeHolderRepository,
    private readonly auditWriter: AuditWriter,
  ) {}

  async execute(command: AppointOfficeHolderCommand): Promise<OfficeHolderDto> {
    const office = await this.officeRepository.getOfficeById(command.input.officeId);

    if (!office) {
      throw new NotFoundError(`Office ${command.input.officeId} was not found.`);
    }

    const now = new Date().toISOString();

    const holder = {
      id: crypto.randomUUID(),
      officeId: command.input.officeId,
      personId: command.input.personId,
      appointedAt: command.input.appointedAt ?? now,
      startedAt: command.input.startedAt ?? now,
      createdAt: now,
    };

    await this.officeHolderRepository.createOfficeHolder(holder);

    await this.auditWriter.write({
      actionType: "ASSIGN",
      entityType: "OfficeHolder",
      entityId: holder.id,
      actorPersonId: command.actor.personId,
      summary: "Office holder appointed.",
      metadata: {
        officeId: holder.officeId,
        personId: holder.personId,
      },
    });

    return toOfficeHolderDto(holder);
  }
}

14. apps/gov-api/src/application/governance/end-office-holder-handler.ts

import { ConflictError, NotFoundError } from "../../http/errors";
import type { OfficeHolderAdministrationRepository } from "../../domain/governance/office-holder-admin-repository";
import type { AuditWriter } from "../audit/audit-writer";
import type {
  EndOfficeHolderCommand,
  EndOfficeHolderCommandHandler,
} from "./end-office-holder";
import type { OfficeHolderDto } from "./office-holder-dto";
import { toOfficeHolderDto } from "./office-holder-mappers";

export class EndOfficeHolderHandler implements EndOfficeHolderCommandHandler {
  constructor(
    private readonly repository: OfficeHolderAdministrationRepository,
    private readonly auditWriter: AuditWriter,
  ) {}

  async execute(command: EndOfficeHolderCommand): Promise<OfficeHolderDto> {
    const holder = await this.repository.getOfficeHolderById(command.officeHolderId);

    if (!holder) {
      throw new NotFoundError(`Office holder ${command.officeHolderId} was not found.`);
    }

    if (holder.endedAt) {
      throw new ConflictError("Office holder term is already ended.", {
        code: "office_holder_already_ended",
      });
    }

    holder.endedAt = command.input.endedAt ?? new Date().toISOString();

    await this.repository.updateOfficeHolder(holder);

    await this.auditWriter.write({
      actionType: "ARCHIVE",
      entityType: "OfficeHolder",
      entityId: holder.id,
      actorPersonId: command.actor.personId,
      summary: "Office holder term ended.",
      metadata: {
        endedAt: holder.endedAt,
        reason: command.input.reason,
      },
    });

    return toOfficeHolderDto(holder);
  }
}

15. apps/gov-api/src/routes/office-holders.ts

import { Hono } from "hono";
import type { HonoEnv } from "../types/hono";
import { jsonData } from "../http/json";
import { requireAuth } from "../middleware/require-auth";
import { requireAuthority } from "../middleware/require-authority";
import { NotFoundError, ValidationError } from "../http/errors";
import { runIdempotentCommand } from "../http/idempotency";

export function createOfficeHolderRoutes(): Hono<HonoEnv> {
  const app = new Hono<HonoEnv>();

  app.get("/", requireAuth, requireAuthority("office_holder.read"), async (c) => {
    const appContext = c.get("appContext");
    const result = await appContext.handlers.officeHolders.list.execute();
    return c.json(jsonData(result), 200);
  });

  app.get(
    "/:officeHolderId",
    requireAuth,
    requireAuthority("office_holder.read"),
    async (c) => {
      const appContext = c.get("appContext");
      const officeHolderId = c.req.param("officeHolderId");

      const result = await appContext.handlers.officeHolders.get.execute({
        officeHolderId,
      });

      if (!result) {
        throw new NotFoundError(`Office holder ${officeHolderId} was not found.`);
      }

      return c.json(jsonData(result), 200);
    },
  );

  app.post(
    "/",
    requireAuth,
    requireAuthority("office_holder.appoint"),
    async (c) => {
      const appContext = c.get("appContext");
      const actor = c.get("requestContext").actor!;
      const body = await c.req.json<{
        officeId?: string;
        personId?: string;
        appointedAt?: string;
        startedAt?: string;
      }>();

      const errors: Record<string, string[]> = {};
      if (!body.officeId?.trim()) {
        errors.officeId = ["officeId is required."];
      }
      if (!body.personId?.trim()) {
        errors.personId = ["personId is required."];
      }

      if (Object.keys(errors).length > 0) {
        throw new ValidationError("Appoint office holder request was invalid.", errors);
      }

      const requestPayload = {
        officeId: body.officeId!,
        personId: body.personId!,
        appointedAt: body.appointedAt,
        startedAt: body.startedAt,
      };

      const result = await runIdempotentCommand({
        c,
        operationName: "officeHolder.appoint",
        requestBody: requestPayload,
        execute: async () =>
          appContext.handlers.officeHolders.appoint.execute({
            actor: { personId: actor.personId },
            input: requestPayload,
          }),
      });

      return c.json(jsonData(result), 201);
    },
  );

  app.post(
    "/:officeHolderId/actions/end",
    requireAuth,
    requireAuthority("office_holder.end"),
    async (c) => {
      const appContext = c.get("appContext");
      const actor = c.get("requestContext").actor!;
      const officeHolderId = c.req.param("officeHolderId");
      const body = await c.req.json<{ endedAt?: string; reason?: string }>().catch(() => ({}));

      const result = await appContext.handlers.officeHolders.end.execute({
        actor: { personId: actor.personId },
        officeHolderId,
        input: {
          endedAt: body.endedAt,
          reason: body.reason,
        },
      });

      return c.json(jsonData(result), 200);
    },
  );

  return app;
}
