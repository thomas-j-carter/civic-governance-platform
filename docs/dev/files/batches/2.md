import { ConflictError } from "../../http/errors";
import type { Proposal } from "./proposal";
import type { ProposalVersion } from "./proposal-version";

export function assertProposalIsDraft(proposal: Proposal): void {
  if (proposal.currentStage !== "DRAFT") {
    throw new ConflictError(
      `Proposal cannot be modified from current stage ${proposal.currentStage}.`,
      {
        type: "https://gov-api.ardtiresociety.org/problems/invalid-state-transition",
        code: "invalid_state_transition",
      },
    );
  }
}

export function assertProposalCanBeSubmitted(proposal: Proposal): void {
  if (proposal.currentStage !== "DRAFT") {
    throw new ConflictError(
      `Proposal cannot be submitted from current stage ${proposal.currentStage}.`,
      {
        type: "https://gov-api.ardtiresociety.org/problems/invalid-state-transition",
        code: "invalid_state_transition",
      },
    );
  }

  if (!proposal.currentVersionId) {
    throw new ConflictError("Proposal must have a current version before submission.", {
      code: "missing_current_version",
    });
  }
}

export function assertProposalCanBeAssignedCommittee(proposal: Proposal): void {
  if (!["SUBMITTED", "ELIGIBILITY_REVIEW"].includes(proposal.currentStage)) {
    throw new ConflictError(
      `Proposal cannot be assigned to committee from current stage ${proposal.currentStage}.`,
      {
        type: "https://gov-api.ardtiresociety.org/problems/invalid-state-transition",
        code: "invalid_state_transition",
      },
    );
  }
}

export function assertVersionBelongsToProposal(
  proposal: Proposal,
  version: ProposalVersion,
): void {
  if (version.proposalId !== proposal.id) {
    throw new ConflictError("Proposal version does not belong to the specified proposal.", {
      type: "https://gov-api.ardtiresociety.org/problems/version-proposal-mismatch",
      code: "version_proposal_mismatch",
    });
  }
}

6. apps/gov-api/src/infrastructure/persistence/in-memory/in-memory-proposal-repository.ts

import type {
  ListProposalsInput,
  ListProposalsResult,
  ProposalRepository,
  ProposalVersionRepository,
} from "../../../domain/proposals/proposal-repository";
import type { Proposal } from "../../../domain/proposals/proposal";
import type { ProposalVersion } from "../../../domain/proposals/proposal-version";

export class InMemoryProposalRepository implements ProposalRepository {
  private readonly proposals = new Map<string, Proposal>();

  async createProposal(input: Proposal): Promise<void> {
    this.proposals.set(input.id, structuredClone(input));
  }

  async updateProposal(input: Proposal): Promise<void> {
    this.proposals.set(input.id, structuredClone(input));
  }

  async getProposalById(proposalId: string): Promise<Proposal | null> {
    const proposal = this.proposals.get(proposalId);
    return proposal ? structuredClone(proposal) : null;
  }

  async listProposals(input: ListProposalsInput): Promise<ListProposalsResult> {
    let items = [...this.proposals.values()];

    if (input.stage) {
      items = items.filter((item) => item.currentStage === input.stage);
    }

    if (input.proposalType) {
      items = items.filter((item) => item.proposalType === input.proposalType);
    }

    items.sort((a, b) => b.createdAt.localeCompare(a.createdAt));

    const total = items.length;
    const totalPages = Math.max(1, Math.ceil(total / input.limit));
    const offset = (input.page - 1) * input.limit;
    const paged = items.slice(offset, offset + input.limit).map((item) => structuredClone(item));

    return {
      items: paged,
      page: input.page,
      limit: input.limit,
      total,
      totalPages,
    };
  }
}

export class InMemoryProposalVersionRepository implements ProposalVersionRepository {
  private readonly versions = new Map<string, ProposalVersion>();

  async createProposalVersion(input: ProposalVersion): Promise<void> {
    this.versions.set(input.id, structuredClone(input));
  }

  async listProposalVersions(proposalId: string): Promise<ProposalVersion[]> {
    return [...this.versions.values()]
      .filter((item) => item.proposalId === proposalId)
      .sort((a, b) => a.versionNumber - b.versionNumber)
      .map((item) => structuredClone(item));
  }

  async getProposalVersionById(proposalVersionId: string): Promise<ProposalVersion | null> {
    const version = this.versions.get(proposalVersionId);
    return version ? structuredClone(version) : null;
  }

  async getNextVersionNumber(proposalId: string): Promise<number> {
    const versions = [...this.versions.values()].filter((item) => item.proposalId === proposalId);
    if (versions.length === 0) {
      return 1;
    }

    return Math.max(...versions.map((item) => item.versionNumber)) + 1;
  }
}

8. apps/gov-api/src/application/proposals/create-proposal-draft-handler.ts

import type { ProposalRepository, ProposalVersionRepository } from "../../domain/proposals/proposal-repository";
import type { CreateProposalDraftCommand, CreateProposalDraftCommandHandler, ProposalDetailDto } from "./create-proposal-draft";
import { toProposalDetailDto } from "./proposal-mappers";

export class CreateProposalDraftHandler implements CreateProposalDraftCommandHandler {
  constructor(
    private readonly proposalRepository: ProposalRepository,
    private readonly proposalVersionRepository: ProposalVersionRepository,
  ) {}

  async execute(command: CreateProposalDraftCommand): Promise<ProposalDetailDto> {
    const proposalId = crypto.randomUUID();
    const versionId = crypto.randomUUID();
    const now = new Date().toISOString();

    const proposal = {
      id: proposalId,
      title: command.input.title,
      summary: command.input.summary,
      proposalType: (command.input.proposalType ?? "GENERAL") as ProposalDetailDto["proposalType"],
      currentStage: "DRAFT" as const,
      proposerPersonId: command.actor.personId,
      proposerMemberId: command.actor.memberId,
      currentVersionId: undefined,
      createdAt: now,
      updatedAt: now,
    };

    await this.proposalRepository.createProposal(proposal);

    const version = {
      id: versionId,
      proposalId,
      versionNumber: 1,
      titleSnapshot: command.input.title,
      bodyMarkdown: command.input.bodyMarkdown,
      createdByPersonId: command.actor.personId,
      createdAt: now,
    };

    await this.proposalVersionRepository.createProposalVersion(version);

    proposal.currentVersionId = versionId;
    proposal.updatedAt = new Date().toISOString();

    await this.proposalRepository.updateProposal(proposal);

    return toProposalDetailDto(proposal);
  }
}

12. apps/gov-api/src/application/proposals/create-proposal-version-handler.ts

import { NotFoundError } from "../../http/errors";
import { assertProposalIsDraft } from "../../domain/proposals/proposal-rules";
import type { ProposalRepository, ProposalVersionRepository } from "../../domain/proposals/proposal-repository";
import type { CreateProposalVersionCommand, CreateProposalVersionCommandHandler } from "./create-proposal-version";
import type { ProposalVersionDto } from "./list-proposal-versions";
import { toProposalVersionDto } from "./proposal-mappers";

export class CreateProposalVersionHandler implements CreateProposalVersionCommandHandler {
  constructor(
    private readonly proposalRepository: ProposalRepository,
    private readonly proposalVersionRepository: ProposalVersionRepository,
  ) {}

  async execute(command: CreateProposalVersionCommand): Promise<ProposalVersionDto> {
    const proposal = await this.proposalRepository.getProposalById(command.proposalId);
    if (!proposal) {
      throw new NotFoundError(`Proposal ${command.proposalId} was not found.`);
    }

    assertProposalIsDraft(proposal);

    const versionNumber = await this.proposalVersionRepository.getNextVersionNumber(command.proposalId);

    const version = {
      id: crypto.randomUUID(),
      proposalId: command.proposalId,
      versionNumber,
      titleSnapshot: command.input.titleSnapshot,
      bodyMarkdown: command.input.bodyMarkdown,
      changeSummary: command.input.changeSummary,
      createdByPersonId: command.actor.personId,
      createdAt: new Date().toISOString(),
    };

    await this.proposalVersionRepository.createProposalVersion(version);

    return toProposalVersionDto(version);
  }
}

13. apps/gov-api/src/application/proposals/set-current-proposal-version-handler.ts

import { NotFoundError } from "../../http/errors";
import {
  assertProposalIsDraft,
  assertVersionBelongsToProposal,
} from "../../domain/proposals/proposal-rules";
import type { ProposalRepository, ProposalVersionRepository } from "../../domain/proposals/proposal-repository";
import type { SetCurrentProposalVersionCommand, SetCurrentProposalVersionCommandHandler } from "./set-current-proposal-version";
import type { ProposalDetailDto } from "./create-proposal-draft";
import { toProposalDetailDto } from "./proposal-mappers";

export class SetCurrentProposalVersionHandler implements SetCurrentProposalVersionCommandHandler {
  constructor(
    private readonly proposalRepository: ProposalRepository,
    private readonly proposalVersionRepository: ProposalVersionRepository,
  ) {}

  async execute(command: SetCurrentProposalVersionCommand): Promise<ProposalDetailDto> {
    const proposal = await this.proposalRepository.getProposalById(command.proposalId);
    if (!proposal) {
      throw new NotFoundError(`Proposal ${command.proposalId} was not found.`);
    }

    assertProposalIsDraft(proposal);

    const version = await this.proposalVersionRepository.getProposalVersionById(
      command.input.proposalVersionId,
    );

    if (!version) {
      throw new NotFoundError(
        `Proposal version ${command.input.proposalVersionId} was not found.`,
      );
    }

    assertVersionBelongsToProposal(proposal, version);

    proposal.currentVersionId = version.id;
    proposal.title = version.titleSnapshot;
    proposal.updatedAt = new Date().toISOString();

    await this.proposalRepository.updateProposal(proposal);

    return toProposalDetailDto(proposal);
  }
}

          return {
            personId: query.actor.personId,
            userAccountId: query.actor.userAccountId,
            displayName: "Development User",
            primaryEmail: query.actor.email,
            member: query.actor.memberId
              ? {

                }
              : undefined,
          };
        },

              id: `role-${index + 1}`,

            offices: [],
            authorityGrants: query.actor.authorityGrants,
          };
        },
      },

      createDraft: new CreateProposalDraftHandler(
        proposalRepository,
        proposalVersionRepository,
      ),
      getDetail: new GetProposalDetailHandler(proposalRepository),
      list: new ListProposalsHandler(proposalRepository),
      listVersions: new ListProposalVersionsHandler(proposalVersionRepository),
      createVersion: new CreateProposalVersionHandler(
        proposalRepository,

      ),
      setCurrentVersion: new SetCurrentProposalVersionHandler(
        proposalRepository,
        proposalVersionRepository,
      ),
      submit: new SubmitProposalHandler(proposalRepository),
      assignCommittee: new AssignCommitteeHandler(proposalRepository),
    },
  },
