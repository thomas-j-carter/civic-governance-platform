import type { PrismaClient } from "@prisma/client";
import type {
  ListProposalsInput,
  ListProposalsResult,
  ProposalRepository,
  ProposalVersionRepository,
} from "../../../domain/proposals/proposal-repository";
import type { Proposal } from "../../../domain/proposals/proposal";
import type { ProposalVersion } from "../../../domain/proposals/proposal-version";
import {
  mapProposalRecordToDomain,
  mapProposalVersionRecordToDomain,
} from "../mappers/proposal-mappers";

export class PrismaProposalRepository implements ProposalRepository {
  constructor(private readonly prisma: PrismaClient) {}

  async createProposal(input: Proposal): Promise<void> {
    await this.prisma.proposal.create({
      data: {
        id: input.id,
        proposalNumber: input.proposalNumber ?? null,
        title: input.title,
        summary: input.summary ?? null,
        proposalType: input.proposalType,
        currentStage: input.currentStage,
        proposerPersonId: input.proposerPersonId ?? null,
        proposerMemberId: input.proposerMemberId ?? null,
        currentVersionId: input.currentVersionId ?? null,
        submittedAt: input.submittedAt ? new Date(input.submittedAt) : null,
        withdrawnAt: input.withdrawnAt ? new Date(input.withdrawnAt) : null,
        rejectedAt: input.rejectedAt ? new Date(input.rejectedAt) : null,
        createdAt: new Date(input.createdAt),
        updatedAt: new Date(input.updatedAt),
      },
    });
  }

  async updateProposal(input: Proposal): Promise<void> {
    await this.prisma.proposal.update({
      where: { id: input.id },
      data: {
        proposalNumber: input.proposalNumber ?? null,
        title: input.title,
        summary: input.summary ?? null,
        proposalType: input.proposalType,
        currentStage: input.currentStage,
        proposerPersonId: input.proposerPersonId ?? null,
        proposerMemberId: input.proposerMemberId ?? null,
        currentVersionId: input.currentVersionId ?? null,
        submittedAt: input.submittedAt ? new Date(input.submittedAt) : null,
        withdrawnAt: input.withdrawnAt ? new Date(input.withdrawnAt) : null,
        rejectedAt: input.rejectedAt ? new Date(input.rejectedAt) : null,
        updatedAt: new Date(input.updatedAt),
      },
    });
  }

  async getProposalById(proposalId: string): Promise<Proposal | null> {
    const record = await this.prisma.proposal.findUnique({
      where: { id: proposalId },
    });

    return record ? mapProposalRecordToDomain(record) : null;
  }

  async listProposals(input: ListProposalsInput): Promise<ListProposalsResult> {
    const where = {
      ...(input.stage ? { currentStage: input.stage } : {}),
      ...(input.proposalType ? { proposalType: input.proposalType } : {}),
    };

    const [total, items] = await this.prisma.$transaction([
      this.prisma.proposal.count({ where }),
      this.prisma.proposal.findMany({
        where,
        orderBy: { createdAt: "desc" },
        skip: (input.page - 1) * input.limit,
        take: input.limit,
      }),
    ]);

    return {
      items: items.map(mapProposalRecordToDomain),
      page: input.page,
      limit: input.limit,
      total,
      totalPages: Math.max(1, Math.ceil(total / input.limit)),
    };
  }
}

export class PrismaProposalVersionRepository implements ProposalVersionRepository {
  constructor(private readonly prisma: PrismaClient) {}

  async createProposalVersion(input: ProposalVersion): Promise<void> {
    await this.prisma.proposalVersion.create({
      data: {
        id: input.id,
        proposalId: input.proposalId,
        versionNumber: input.versionNumber,
        titleSnapshot: input.titleSnapshot,
        bodyMarkdown: input.bodyMarkdown,
        changeSummary: input.changeSummary ?? null,
        createdByPersonId: input.createdByPersonId,
        createdAt: new Date(input.createdAt),
      },
    });
  }

  async listProposalVersions(proposalId: string): Promise<ProposalVersion[]> {
    const records = await this.prisma.proposalVersion.findMany({
      where: { proposalId },
      orderBy: { versionNumber: "asc" },
    });

    return records.map(mapProposalVersionRecordToDomain);
  }

  async getProposalVersionById(proposalVersionId: string): Promise<ProposalVersion | null> {
    const record = await this.prisma.proposalVersion.findUnique({
      where: { id: proposalVersionId },
    });

    return record ? mapProposalVersionRecordToDomain(record) : null;
  }

  async getNextVersionNumber(proposalId: string): Promise<number> {
    const aggregate = await this.prisma.proposalVersion.aggregate({
      where: { proposalId },
      _max: { versionNumber: true },
    });

    return (aggregate._max.versionNumber ?? 0) + 1;
  }
