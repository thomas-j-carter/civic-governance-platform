3. Proposal Repository Integration Test

prisma-proposal-repository.integration.test.ts

import { describe, expect, it } from "vitest";
import { getPrismaClient } from "../prisma-client";
import {
  PrismaProposalRepository,
  PrismaProposalVersionRepository,
} from "./prisma-proposal-repository";

describe("PrismaProposalRepository (integration)", () => {
  const prisma = getPrismaClient();
  const proposalRepo = new PrismaProposalRepository(prisma);
  const versionRepo = new PrismaProposalVersionRepository(prisma);

  it("creates proposal and version and retrieves them", async () => {
    const proposalId = "proposal-1";
    const versionId = "version-1";

    await proposalRepo.createProposal({
      id: proposalId,
      title: "Test Proposal",
      proposalType: "GENERAL",
      currentStage: "DRAFT",
      createdAt: new Date().toISOString(),
      updatedAt: new Date().toISOString(),
    });

    await versionRepo.createProposalVersion({
      id: versionId,
      proposalId,
      versionNumber: 1,
      titleSnapshot: "Test Proposal",
      bodyMarkdown: "# Hello",
      createdByPersonId: "person-1",
      createdAt: new Date().toISOString(),
    });

    const proposal = await proposalRepo.getProposalById(proposalId);

    expect(proposal).not.toBeNull();
    expect(proposal?.title).toBe("Test Proposal");

    const versions = await versionRepo.listProposalVersions(proposalId);

    expect(versions).toHaveLength(1);
    expect(versions[0].versionNumber).toBe(1);
  });

  it("computes next version number correctly", async () => {
    const proposalId = "proposal-2";

    await proposalRepo.createProposal({
      id: proposalId,
      title: "Version Test",
      proposalType: "GENERAL",
      currentStage: "DRAFT",
      createdAt: new Date().toISOString(),
      updatedAt: new Date().toISOString(),
    });

    const v1 = await versionRepo.getNextVersionNumber(proposalId);
    expect(v1).toBe(1);

    await versionRepo.createProposalVersion({
      id: "v1",
      proposalId,
      versionNumber: 1,
      titleSnapshot: "v1",
      bodyMarkdown: "# v1",
      createdByPersonId: "person-1",
      createdAt: new Date().toISOString(),
    });

    const v2 = await versionRepo.getNextVersionNumber(proposalId);
    expect(v2).toBe(2);
  });
});

4. Ballot + Vote + Tally Integration

prisma-ballot-repository.integration.test.ts

import { describe, expect, it } from "vitest";
import { getPrismaClient } from "../prisma-client";
import {
  PrismaBallotRepository,
  PrismaBallotEligibilityRepository,
  PrismaVoteRepository,
  PrismaVoteTallyRepository,
} from "./prisma-ballot-repository";

describe("Ballot + Vote + Tally (integration)", () => {
  const prisma = getPrismaClient();

  const ballotRepo = new PrismaBallotRepository(prisma);
  const eligibilityRepo = new PrismaBallotEligibilityRepository(prisma);
  const voteRepo = new PrismaVoteRepository(prisma);
  const tallyRepo = new PrismaVoteTallyRepository(prisma);

  it("creates ballot, eligibility snapshot, votes, and tally", async () => {
    const ballotId = "ballot-1";

    await ballotRepo.createBallot({
      id: ballotId,
      title: "Final Vote",
      state: "DRAFT",
      createdByPersonId: "person-1",
      createdAt: new Date().toISOString(),
      updatedAt: new Date().toISOString(),
    });

    const ballot = await ballotRepo.getBallotById(ballotId);
    expect(ballot).not.toBeNull();

    await eligibilityRepo.replaceSnapshot(ballotId, [
      {
        id: "elig-1",
        ballotId,
        memberId: "member-1",
        eligibilityStatus: "ELIGIBLE",
        snapshotAt: new Date().toISOString(),
        createdAt: new Date().toISOString(),
      },
    ]);

    const eligibility = await eligibilityRepo.listSnapshot(ballotId);
    expect(eligibility).toHaveLength(1);

    await voteRepo.createVote({
      id: "vote-1",
      ballotId,
      memberId: "member-1",
      choice: "YES",
      castAt: new Date().toISOString(),
      recordedAt: new Date().toISOString(),
      createdAt: new Date().toISOString(),
    });

    const votes = await voteRepo.listVotes(ballotId);
    expect(votes).toHaveLength(1);

    await tallyRepo.upsertTally({
      id: "tally-1",
      ballotId,
      yesCount: 1,
      noCount: 0,
      abstainCount: 0,
      totalCount: 1,
      quorumMet: true,
      thresholdMet: true,
      computedAt: new Date().toISOString(),
      createdAt: new Date().toISOString(),
    });

    const tally = await tallyRepo.getTallyByBallotId(ballotId);
    expect(tally?.yesCount).toBe(1);
  });
});

5. Certification + Ratification Integration

prisma-certification-repository.integration.test.ts

import { describe, expect, it } from "vitest";
import { getPrismaClient } from "../prisma-client";
import {
  PrismaCertificationRepository,
  PrismaRatificationRepository,
} from "./prisma-certification-repository";

describe("Certification + Ratification (integration)", () => {
  const prisma = getPrismaClient();

  const certRepo = new PrismaCertificationRepository(prisma);
  const ratRepo = new PrismaRatificationRepository(prisma);

  it("creates and updates certification", async () => {
    const certId = "cert-1";

    await certRepo.createCertification({
      id: certId,
      ballotId: "ballot-1",
      status: "PENDING",
      quorumRuleVersionId: "rule-q",
      thresholdRuleVersionId: "rule-t",
      createdAt: new Date().toISOString(),
      updatedAt: new Date().toISOString(),
    });

    let cert = await certRepo.getCertificationById(certId);
    expect(cert?.status).toBe("PENDING");

    await certRepo.updateCertification({
      ...cert!,
      status: "CERTIFIED",
      certifiedByPersonId: "person-1",
      certifiedAt: new Date().toISOString(),
      updatedAt: new Date().toISOString(),
    });

    cert = await certRepo.getCertificationById(certId);
    expect(cert?.status).toBe("CERTIFIED");
  });

  it("creates ratification", async () => {
    await ratRepo.createRatification({
      id: "rat-1",
      proposalId: "proposal-1",
      status: "RATIFIED",
      ratifiedByPersonId: "person-1",
      ratifiedAt: new Date().toISOString(),
      createdAt: new Date().toISOString(),
      updatedAt: new Date().toISOString(),
    });

    const rat = await ratRepo.getRatificationById("rat-1");

    expect(rat?.status).toBe("RATIFIED");
  });
});

import { describe, expect, it } from "vitest";
import { getPrismaClient } from "../prisma-client";

  PrismaOfficialRecordRepository,

  const issueRepo = new PrismaGazetteIssueRepository(prisma);
