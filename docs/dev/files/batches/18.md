3. apps/gov-api/src/infrastructure/persistence/in-memory/in-memory-person-repository.ts

import type { Person } from "../../../domain/person/person";
import type { PersonRepository } from "../../../domain/person/person-repository";

export class InMemoryPersonRepository implements PersonRepository {
  private readonly items = new Map<string, Person>();

  async createPerson(person: Person): Promise<void> {
    this.items.set(person.id, structuredClone(person));
  }

  async getPersonById(personId: string): Promise<Person | null> {
    const person = this.items.get(personId);
    return person ? structuredClone(person) : null;
  }

  async findPersonByPrimaryEmail(email: string): Promise<Person | null> {
    const lowered = email.toLowerCase();
    const person = [...this.items.values()].find(
      (item) => item.primaryEmail?.toLowerCase() === lowered,
    );
    return person ? structuredClone(person) : null;
  }

  async listPersons(): Promise<Person[]> {
    return [...this.items.values()]
      .sort((a, b) => b.createdAt.localeCompare(a.createdAt))
      .map((item) => structuredClone(item));
  }
}

import type { PrismaClient } from "@prisma/client";
import type { Person } from "../../../domain/person/person";
import type { PersonRepository } from "../../../domain/person/person-repository";
import { mapPersonRecordToDomain } from "../mappers/person-mappers";

export class PrismaPersonRepository implements PersonRepository {
  constructor(private readonly prisma: PrismaClient) {}

  async createPerson(person: Person): Promise<void> {
    await this.prisma.person.create({
      data: {
        id: person.id,
        displayName: person.displayName,
        primaryEmail: person.primaryEmail ?? null,
        givenName: person.givenName ?? null,
        familyName: person.familyName ?? null,
        isActive: person.isActive,
        createdAt: new Date(person.createdAt),
        updatedAt: new Date(person.updatedAt),
      },
    });
  }

  async getPersonById(personId: string): Promise<Person | null> {
    const record = await this.prisma.person.findUnique({
      where: { id: personId },
    });

    return record ? mapPersonRecordToDomain(record) : null;
  }

  async findPersonByPrimaryEmail(email: string): Promise<Person | null> {
    const record = await this.prisma.person.findFirst({
      where: {
        primaryEmail: { equals: email, mode: "insensitive" },
      },
    });

    return record ? mapPersonRecordToDomain(record) : null;
  }

  async listPersons(): Promise<Person[]> {
    const records = await this.prisma.person.findMany({
      orderBy: { createdAt: "desc" },
    });

    return records.map(mapPersonRecordToDomain);
  }
}

10. apps/gov-api/src/application/person/create-person-handler.ts

import { ConflictError } from "../../http/errors";
import type { PersonRepository } from "../../domain/person/person-repository";
import type { AuditWriter } from "../audit/audit-writer";
import type {
  CreatePersonCommand,
  CreatePersonCommandHandler,
  PersonDto,
} from "./create-person";
import { toPersonDto } from "./person-mappers";

export class CreatePersonHandler implements CreatePersonCommandHandler {
  constructor(
    private readonly repository: PersonRepository,
    private readonly auditWriter: AuditWriter,
  ) {}

  async execute(command: CreatePersonCommand): Promise<PersonDto> {
    if (command.input.primaryEmail) {
      const existing = await this.repository.findPersonByPrimaryEmail(
        command.input.primaryEmail,
      );

      if (existing) {
        throw new ConflictError("A person with that primary email already exists.", {
          code: "person_email_exists",
          type: "https://gov-api.ardtiresociety.org/problems/person-email-exists",
        });
      }
    }

    const now = new Date().toISOString();

    const person = {
      id: crypto.randomUUID(),
      displayName: command.input.displayName,
      primaryEmail: command.input.primaryEmail,
      givenName: command.input.givenName,
      familyName: command.input.familyName,
      isActive: true,
      createdAt: now,
      updatedAt: now,
    };

    await this.repository.createPerson(person);

    await this.auditWriter.write({
      actionType: "CREATE",
      entityType: "Person",
      entityId: person.id,
      actorPersonId: command.actor.personId,
      summary: "Person created.",
      metadata: {
        primaryEmail: person.primaryEmail,
      },
    });

    return toPersonDto(person);
  }
}

14. apps/gov-api/src/application/onboarding/create-person-onboarding-handler.ts

import { ConflictError } from "../../http/errors";
import type { MemberRepository } from "../../domain/authority/authority-repository";
import type { PersonRepository } from "../../domain/person/person-repository";
import type { PersonIdentityLinkRepository } from "../../domain/auth/auth-repository";
import type { TransactionRunner } from "../shared/transaction";
import type { AuditWriter } from "../audit/audit-writer";
import type {
  CreatePersonOnboardingCommand,
  CreatePersonOnboardingCommandHandler,
  PersonOnboardingResultDto,
} from "./create-person-onboarding";
import { toPersonDto } from "../person/person-mappers";

export class CreatePersonOnboardingHandler
  implements CreatePersonOnboardingCommandHandler
{
  constructor(
    private readonly personRepository: PersonRepository,
    private readonly memberRepository: MemberRepository,
    private readonly identityLinkRepository: PersonIdentityLinkRepository,
    private readonly transactionRunner: TransactionRunner,
    private readonly auditWriter: AuditWriter,
  ) {}

  async execute(
    command: CreatePersonOnboardingCommand,
  ): Promise<PersonOnboardingResultDto> {
    return this.transactionRunner.runInTransaction(async () => {
      if (command.input.primaryEmail) {
        const existing = await this.personRepository.findPersonByPrimaryEmail(
          command.input.primaryEmail,
        );
        if (existing) {
          throw new ConflictError("A person with that primary email already exists.", {
            code: "person_email_exists",
          });
        }
      }

      if (command.input.createIdentityLink) {
        if (!command.input.issuer?.trim() || !command.input.subject?.trim()) {
          throw new ConflictError(
            "issuer and subject are required when createIdentityLink is true.",
            {
              code: "identity_link_input_required",
            },
          );
        }

        const existingLink = await this.identityLinkRepository.findByIssuerAndSubject(
          command.input.issuer,
          command.input.subject,
        );

        if (existingLink) {
          throw new ConflictError("Identity link already exists for issuer and subject.", {
            code: "person_identity_link_exists",
          });
        }
      }

      const now = new Date().toISOString();

      const person = {
        id: crypto.randomUUID(),
        displayName: command.input.displayName,
        primaryEmail: command.input.primaryEmail,
        givenName: command.input.givenName,
        familyName: command.input.familyName,
        isActive: true,
        createdAt: now,
        updatedAt: now,
      };

      await this.personRepository.createPerson(person);

      let member:
        | {
            memberId: string;
            memberNumber?: string;
            membershipStatus: string;
          }
        | undefined;

      if (command.input.createMember) {
        const memberRecord = {
          id: crypto.randomUUID(),
          personId: person.id,
          memberNumber: command.input.memberNumber,
          status: command.input.membershipStatus ?? "ACTIVE",
          admittedAt: now,
          createdAt: now,
          updatedAt: now,
        };

        await this.memberRepository.createMember(memberRecord);

        member = {
          memberId: memberRecord.id,
          memberNumber: memberRecord.memberNumber,
          membershipStatus: memberRecord.status,
        };
      }

      let identityLink:
        | {
            id: string;
            issuer: string;
            subject: string;
            email?: string;
          }
        | undefined;

      if (command.input.createIdentityLink) {
        const link = {
          id: crypto.randomUUID(),
          personId: person.id,
          issuer: command.input.issuer!,
          subject: command.input.subject!,
          email: command.input.primaryEmail,
          createdAt: now,
          updatedAt: now,
        };

        await this.identityLinkRepository.createLink(link);

        identityLink = {
          id: link.id,
          issuer: link.issuer,
          subject: link.subject,
          email: link.email,
        };
      }

      await this.auditWriter.write({
        actionType: "CREATE",
        entityType: "Person",
        entityId: person.id,
        actorPersonId: command.actor.personId,
        summary: "Person onboarding completed.",
        metadata: {
          createdMember: Boolean(member),
          createdIdentityLink: Boolean(identityLink),
        },
      });

      return {
        person: toPersonDto(person),
        member,
        identityLink,
      };
    });
  }
}

15. apps/gov-api/src/routes/persons.ts

import { Hono } from "hono";
import type { HonoEnv } from "../types/hono";
import { jsonData } from "../http/json";
import { requireAuth } from "../middleware/require-auth";
import { requireAuthority } from "../middleware/require-authority";
import { NotFoundError, ValidationError } from "../http/errors";
import { runIdempotentCommand } from "../http/idempotency";

export function createPersonRoutes(): Hono<HonoEnv> {
  const app = new Hono<HonoEnv>();

  app.get("/", requireAuth, requireAuthority("person.read"), async (c) => {
    const appContext = c.get("appContext");
    const result = await appContext.handlers.persons.list.execute();
    return c.json(jsonData(result), 200);
  });

  app.get("/:personId", requireAuth, requireAuthority("person.read"), async (c) => {
    const appContext = c.get("appContext");
    const personId = c.req.param("personId");

    const result = await appContext.handlers.persons.get.execute({ personId });

    if (!result) {
      throw new NotFoundError(`Person ${personId} was not found.`);
    }

    return c.json(jsonData(result), 200);
  });

  app.post("/", requireAuth, requireAuthority("person.create"), async (c) => {
    const appContext = c.get("appContext");
    const actor = c.get("requestContext").actor!;
    const body = await c.req.json<{
      displayName?: string;
      primaryEmail?: string;
      givenName?: string;
      familyName?: string;
    }>();

    if (!body.displayName?.trim()) {
      throw new ValidationError("Create person request was invalid.", {
        displayName: ["displayName is required."],
      });
    }

    const requestPayload = {
      displayName: body.displayName!,
      primaryEmail: body.primaryEmail,
      givenName: body.givenName,
      familyName: body.familyName,
    };

    const result = await runIdempotentCommand({
      c,
      operationName: "person.create",
      requestBody: requestPayload,
      execute: async () =>
        appContext.handlers.persons.create.execute({
          actor: { personId: actor.personId },
          input: requestPayload,
        }),
    });

    return c.json(jsonData(result), 201);
  });

  app.post(
    "/onboarding",
    requireAuth,
    requireAuthority("person.onboard"),
    async (c) => {
      const appContext = c.get("appContext");
      const actor = c.get("requestContext").actor!;
      const body = await c.req.json<{
        displayName?: string;
        primaryEmail?: string;
        givenName?: string;
        familyName?: string;
        createMember?: boolean;
        memberNumber?: string;
        membershipStatus?: string;
        createIdentityLink?: boolean;
        issuer?: string;
        subject?: string;
      }>();

      if (!body.displayName?.trim()) {
        throw new ValidationError("Person onboarding request was invalid.", {
          displayName: ["displayName is required."],
        });
      }

      const requestPayload = {
        displayName: body.displayName!,
        primaryEmail: body.primaryEmail,
        givenName: body.givenName,
        familyName: body.familyName,
        createMember: body.createMember,
        memberNumber: body.memberNumber,
        membershipStatus: body.membershipStatus,
        createIdentityLink: body.createIdentityLink,
        issuer: body.issuer,
        subject: body.subject,
      };

      const result = await runIdempotentCommand({
        c,
        operationName: "person.onboard",
        requestBody: requestPayload,
        execute: async () =>
          appContext.handlers.onboarding.createPerson.execute({
            actor: { personId: actor.personId },
            input: requestPayload,
          }),
      });

      return c.json(jsonData(result), 201);
    },
  );

  return app;
}
