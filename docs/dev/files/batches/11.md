import { PrismaClient } from "@prisma/client";

const prisma = new PrismaClient();

async function main(): Promise<void> {

  const proposalVersionId = crypto.randomUUID();
  const ballotId = crypto.randomUUID();
  const certificationId = crypto.randomUUID();
  const recordId = crypto.randomUUID();
  const gazetteIssueId = crypto.randomUUID();
  const gazetteEntryId = crypto.randomUUID();

  await prisma.proposal.create({
    data: {
      id: proposalId,
      title: "Seeded Founding Resolution",
      summary: "Initial seeded proposal for local development.",
      proposalType: "GENERAL",
      currentStage: "DRAFT",
      proposerPersonId: "seed-person-1",
      proposerMemberId: "seed-member-1",
      currentVersionId: proposalVersionId,
    },
  });

  await prisma.proposalVersion.create({
    data: {
      id: proposalVersionId,
      proposalId,
      versionNumber: 1,
      titleSnapshot: "Seeded Founding Resolution",
      bodyMarkdown: "# Seeded Founding Resolution\n\nThis is a seeded proposal.",
      createdByPersonId: "seed-person-1",
    },
  });

  await prisma.ballot.create({
    data: {
      id: ballotId,
      proposalId,
      title: "Seeded Final Vote",
      description: "Seed ballot for local testing.",
      state: "DRAFT",
      createdByPersonId: "seed-person-1",
    },
  });

  await prisma.certificationRecord.create({
    data: {
      id: certificationId,
      ballotId,
      status: "PENDING",
      quorumRuleVersionId: "seed-rule-quorum-v1",
      thresholdRuleVersionId: "seed-rule-threshold-v1",
      certificationRuleVersionId: "seed-rule-certification-v1",
      notes: "Seed certification record.",
    },
  });

  await prisma.officialRecord.create({
    data: {
      id: recordId,
      recordType: "ACT",
      title: "Seeded Official Record",
      summary: "Seed official record for local development.",
      sourceEntityType: "proposal",
      sourceEntityId: proposalId,
      status: "DRAFT",
    },
  });

  await prisma.gazetteIssue.create({
    data: {
      id: gazetteIssueId,
      issueNumber: "0001",
      title: "Seed Gazette Issue",
      publicationState: "DRAFT",
      createdByPersonId: "seed-person-1",
    },
  });

  await prisma.gazetteEntry.create({
    data: {
      id: gazetteEntryId,

      officialRecordId: recordId,
      titleSnapshot: "Seeded Official Record",
      summarySnapshot: "Seed gazette entry.",
      publicationOrder: 1,
    },
  });

  console.log("Seed complete.");
}

main()
  .catch((error) => {
    console.error("Seed failed:", error);
    process.exit(1);
  })
  .finally(async () => {
    await prisma.$disconnect();
  });
