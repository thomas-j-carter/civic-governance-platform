    );

    const roles: ResolvedAuthorityContext["roles"] = [];
    const authorityGrantSet = new Set<string>();

    for (const assignment of assignments) {
      const role = await this.roleRepository.getRoleById(assignment.roleId);
      if (!role) continue;

      roles.push({
        id: role.id,
        code: role.code,
        name: role.name,
        description: role.description,
      });

      for (const grant of roleCodeToAuthorityGrants[role.code] ?? []) {
        authorityGrantSet.add(grant);
      }
    }

    for (const delegation of delegations) {
      for (const grant of delegation.authorityGrantCodes) {
        authorityGrantSet.add(grant);
      }
    }

    for (const officeHolder of officeHolders) {
      const office = await this.officeRepository.getOfficeById(officeHolder.officeId);
      if (!office) continue;

      if (office.code === "chief-certifier") {
        authorityGrantSet.add("result.certify");
      }

      if (office.code === "gazette-keeper") {
        authorityGrantSet.add("gazette.issue.publish");
        authorityGrantSet.add("gazette.entry.create");
      }
    }

    return {
      roles,
      offices: officeHolders.map((item) => ({
        id: item.id,
        officeId: item.officeId,
        personId: item.personId,
        startedAt: item.startedAt,
        endedAt: item.endedAt,
      })),
      authorityGrants: [...authorityGrantSet].sort(),
      member: member
        ? {
            memberId: member.id,
            memberNumber: member.memberNumber,
            membershipStatus: member.status,
          }
        : undefined,
    };
  }
}
