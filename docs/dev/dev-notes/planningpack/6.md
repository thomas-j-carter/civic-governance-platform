import { createGovClient } from '@ardtire/gov-client'
import { devActors, type DevActorPreset } from './dev-actors'

function getGovApiBaseUrl(): string {
  const value = import.meta.env.VITE_GOV_API_BASE_URL
  if (!value) {
    throw new Error('VITE_GOV_API_BASE_URL is required.')
  }
  return value
}

export function createWebGovClient(preset: DevActorPreset) {
  const actor = devActors[preset]

  return createGovClient({
    baseUrl: getGovApiBaseUrl(),
    getHeaders: () => ({
      'x-user-id': actor.userId,
      'x-user-email': actor.email,
      'x-user-roles': actor.roles.join(','),
    }),
  })
}

import type { MiddlewareHandler } from 'hono'
import { prisma } from '../../lib/prisma'
import type { AppBindings } from '../types'

function deriveDisplayName(email: string | null, userId: string): string {
  if (!email) return userId
  return email.split('@')[0] || userId
}

export const devActorProvisioningMiddleware: MiddlewareHandler<AppBindings> = async (
  c,
  next,
) => {
  if (process.env.NODE_ENV === 'production') {
    await next()
    return
  }

  const actor = c.get('actor')

  if (actor.userId) {
    await prisma.user.upsert({
      where: {
        id: actor.userId,
      },
      create: {
        id: actor.userId,
        email: actor.email ?? `${actor.userId}@local.invalid`,
        displayName: deriveDisplayName(actor.email, actor.userId),
      },
      update: {
        email: actor.email ?? undefined,
        displayName: deriveDisplayName(actor.email, actor.userId),
      },
    })
  }

  await next()
}

import { Hono } from 'hono'
import { authContextMiddleware } from './http/middleware/auth-context'
import { devActorProvisioningMiddleware } from './http/middleware/dev-actor-provisioning'
import { requestIdMiddleware } from './http/middleware/request-id'
import { createAuthContextRoutes } from './http/routes/auth-context'
import { createHealthRoutes } from './http/routes/health'
import { createMembershipApplicationRoutes } from './http/routes/membership-applications'
import type { AppBindings } from './http/types'

export function createApp() {
  const app = new Hono<AppBindings>()

  app.use('*', requestIdMiddleware)
  app.use('*', authContextMiddleware)
  app.use('*', devActorProvisioningMiddleware)

  app.onError((error, c) => {
    console.error('Unhandled request error', {
      requestId: c.get('requestId'),
      message: error.message,
      stack: error.stack,
    })

    return c.json(
      {
        code: 'INTERNAL_SERVER_ERROR',
        message: 'An unexpected error occurred.',
      },
      500,
    )
  })

  app.route('/', createHealthRoutes())
  app.route('/', createAuthContextRoutes())
  app.route('/', createMembershipApplicationRoutes())

  return app
}
