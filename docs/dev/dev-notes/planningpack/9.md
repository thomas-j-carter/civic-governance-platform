I’m also making the remaining bootstrap auth assumptions explicit: dev header auth becomes opt-in via env, while authenticated-user auto-provisioning becomes a named switch instead of hidden behavior. TanStack Start’s server-side auth/session pattern is a good fit for this because the refresh logic can stay in server utilities instead of leaking into client components.

export async function exchangeAuthorizationCodeForToken(
  code: string,
): Promise<KeycloakTokenSet> {
  const discovery = await getDiscoveryDocument()

  const body = new URLSearchParams()
  body.set('grant_type', 'authorization_code')
  body.set('code', code)
  body.set('client_id', getClientId())
  body.set('client_secret', getClientSecret())
  body.set('redirect_uri', getKeycloakCallbackUrl())

  const response = await fetch(discovery.token_endpoint, {
    method: 'POST',
    headers: {
      'content-type': 'application/x-www-form-urlencoded',
    },
    body,
  })

  if (!response.ok) {
    const text = await response.text()
    throw new Error(`Failed to exchange Keycloak authorization code: ${text}`)
  }

  const tokenResponse = (await response.json()) as TokenResponse
  return normalizeTokenResponse(tokenResponse)
}

    method: 'POST',
    headers: {

function normalizeTokenResponse(tokenResponse: TokenResponse): KeycloakTokenSet {
  const claims = decodeJwt(tokenResponse.access_token)

  return {
    accessToken: tokenResponse.access_token,
    refreshToken: tokenResponse.refresh_token ?? null,
    idToken: tokenResponse.id_token ?? null,
    accessTokenExpiresAt: Date.now() + tokenResponse.expires_in * 1000,
    refreshTokenExpiresAt:
      typeof tokenResponse.refresh_expires_in === 'number'
        ? Date.now() + tokenResponse.refresh_expires_in * 1000
        : null,
    user: mapClaimsToAuthenticatedUser(claims),
  }
}

function mapClaimsToAuthenticatedUser(claims: Record<string, unknown>): AuthenticatedUser {
  const validRoles = new Set<string>(appRoles)

  const realmRoles = Array.isArray((claims.realm_access as { roles?: unknown[] } | undefined)?.roles)
    ? ((claims.realm_access as { roles?: unknown[] }).roles ?? [])
    : []

  const resourceRolesObject =
    (claims.resource_access as Record<string, { roles?: unknown[] }> | undefined) ?? {}
  const resourceRoles = Object.values(resourceRolesObject).flatMap((entry) =>
    Array.isArray(entry.roles) ? entry.roles : [],
  )

  const mergedRoles = [...realmRoles, ...resourceRoles].filter(
    (value): value is string => typeof value === 'string' && validRoles.has(value),
  )

  return {
    sub: typeof claims.sub === 'string' ? claims.sub : '',
    email: typeof claims.email === 'string' ? claims.email : null,
    name: typeof claims.name === 'string' ? claims.name : null,
    preferredUsername:
      typeof claims.preferred_username === 'string' ? claims.preferred_username : null,
    roles: Array.from(new Set(mergedRoles)) as AppRole[],
  }
}

5) Add apps/web/src/lib/auth/ensure-session.ts

import {
  refreshKeycloakAccessToken,
} from './keycloak'
import type { AuthenticatedUser } from './session'
import { useArdtireAuthSession } from './session'

const REFRESH_WINDOW_MS = 60_000

export interface FreshAuthSessionResult {
  session: Awaited<ReturnType<typeof useArdtireAuthSession>>
  auth:
    | {
        accessToken: string
        idToken: string | null
        user: AuthenticatedUser
      }
    | null
}

export async function getFreshAuthenticatedSession(): Promise<FreshAuthSessionResult> {
  const session = await useArdtireAuthSession()
  const data = session.data

  if (!data.user || !data.accessToken || !data.accessTokenExpiresAt) {
    return { session, auth: null }
  }

  if (Date.now() + REFRESH_WINDOW_MS < data.accessTokenExpiresAt) {
    return {
      session,
      auth: {
        accessToken: data.accessToken,
        idToken: data.idToken ?? null,
        user: data.user,
      },
    }
  }

  if (!data.refreshToken) {
    await session.clear()
    return { session, auth: null }
  }

  if (
    typeof data.refreshTokenExpiresAt === 'number' &&
    Date.now() >= data.refreshTokenExpiresAt
  ) {
    await session.clear()
    return { session, auth: null }
  }

  try {
    const refreshed = await refreshKeycloakAccessToken(data.refreshToken)

    await session.update({
      ...session.data,
      accessToken: refreshed.accessToken,
      refreshToken: refreshed.refreshToken ?? data.refreshToken,
      idToken: refreshed.idToken ?? data.idToken,
      accessTokenExpiresAt: refreshed.accessTokenExpiresAt,
      refreshTokenExpiresAt:
        refreshed.refreshTokenExpiresAt ?? data.refreshTokenExpiresAt ?? null,
      user: refreshed.user,
    })

    return {
      session,
      auth: {
        accessToken: refreshed.accessToken,
        idToken: refreshed.idToken ?? data.idToken ?? null,
        user: refreshed.user,
      },
    }
  } catch (error) {
    console.error('Failed to refresh app session access token', error)
    await session.clear()
    return { session, auth: null }
  }
}

const validRoles = new Set<string>(appRoles)

function parseRoles(raw: string | undefined): AppRole[] {
  if (!raw) return []
  return raw
    .split(',')
    .map((value) => value.trim())
    .filter((value): value is AppRole => validRoles.has(value))
}

function parseBearerToken(headerValue: string | undefined): string | null {
  if (!headerValue) return null
  const match = headerValue.match(/^Bearer\s+(.+)$/i)
  return match?.[1] ?? null
}

function allowDevAuthHeaders(): boolean {
  return (
    process.env.NODE_ENV !== 'production' &&
    (process.env.ALLOW_DEV_AUTH_HEADERS ?? 'true').toLowerCase() === 'true'
  )
}

export const authContextMiddleware: MiddlewareHandler<AppBindings> = async (c, next) => {
  const bearerToken = parseBearerToken(c.req.header('authorization'))

  if (bearerToken) {
    try {
      const payload = await verifyKeycloakAccessToken(bearerToken)
      const userId = typeof payload.sub === 'string' ? payload.sub : null
      const email = typeof payload.email === 'string' ? payload.email : null
      const roles = extractAppRolesFromJwtPayload(payload)

      c.set(
        'actor',
        buildActorContext({
          userId,
          email,
          roles,
        }),
      )

      await next()
      return
    } catch (error) {
      console.error('Bearer token verification failed', error)

      return c.json(
        {
          code: 'UNAUTHORIZED',
          message: 'Invalid bearer token.',
        },
        401,
      )
    }
  }

  if (allowDevAuthHeaders()) {
    const userId = c.req.header('x-user-id') ?? null
    const email = c.req.header('x-user-email') ?? null
    const roles = parseRoles(c.req.header('x-user-roles'))

    c.set(
      'actor',
      buildActorContext({
        userId,
        email,
        roles,
      }),
    )

    await next()
    return
  }

  c.set(
    'actor',
    buildActorContext({
      userId: null,
      email: null,
      roles: [],
    }),
  )

  await next()
}
