4. apps/gov-api/src/infrastructure/persistence/in-memory/in-memory-audit-repository.ts

import type {
  AuditRepository,
  ListAuditEventsInput,
  ListAuditEventsResult,
} from "../../../domain/audit/audit-repository";
import type { AuditEvent } from "../../../domain/audit/audit-event";

export class InMemoryAuditRepository implements AuditRepository {
  private readonly events = new Map<string, AuditEvent>();

  async createAuditEvent(event: AuditEvent): Promise<void> {
    this.events.set(event.id, structuredClone(event));
  }

  async listAuditEvents(input: ListAuditEventsInput): Promise<ListAuditEventsResult> {
    let items = [...this.events.values()];

    if (input.actionType) {
      items = items.filter((item) => item.actionType === input.actionType);
    }

    if (input.entityType) {
      items = items.filter((item) => item.entityType === input.entityType);
    }

    if (input.entityId) {
      items = items.filter((item) => item.entityId === input.entityId);
    }

    items.sort((a, b) => b.occurredAt.localeCompare(a.occurredAt));

    const total = items.length;
    const totalPages = Math.max(1, Math.ceil(total / input.limit));
    const offset = (input.page - 1) * input.limit;
    const paged = items.slice(offset, offset + input.limit).map((item) => structuredClone(item));

    return {
      items: paged,
      page: input.page,
      limit: input.limit,
      total,
      totalPages,
    };
  }
}

6. apps/gov-api/src/infrastructure/prisma/repositories/prisma-audit-repository.ts

import type { PrismaClient } from "@prisma/client";
import type {
  AuditRepository,
  ListAuditEventsInput,
  ListAuditEventsResult,
} from "../../../domain/audit/audit-repository";
import type { AuditEvent } from "../../../domain/audit/audit-event";
import { mapAuditEventRecordToDomain } from "../mappers/audit-mappers";

export class PrismaAuditRepository implements AuditRepository {
  constructor(private readonly prisma: PrismaClient) {}

  async createAuditEvent(event: AuditEvent): Promise<void> {
    await this.prisma.auditEvent.create({
      data: {
        id: event.id,
        actionType: event.actionType,
        entityType: event.entityType,
        entityId: event.entityId,
        actorPersonId: event.actorPersonId ?? null,
        authoritySourceType: event.authoritySourceType ?? null,
        authoritySourceId: event.authoritySourceId ?? null,
        ruleVersionIds: event.ruleVersionIds ?? [],
        summary: event.summary ?? null,
        metadataJson: event.metadataJson ?? null,
        occurredAt: new Date(event.occurredAt),
        createdAt: new Date(event.createdAt),
      },
    });
  }

  async listAuditEvents(input: ListAuditEventsInput): Promise<ListAuditEventsResult> {
    const where = {
      ...(input.actionType ? { actionType: input.actionType } : {}),
      ...(input.entityType ? { entityType: input.entityType } : {}),
      ...(input.entityId ? { entityId: input.entityId } : {}),
    };

    const [total, items] = await this.prisma.$transaction([
      this.prisma.auditEvent.count({ where }),
      this.prisma.auditEvent.findMany({
        where,
        orderBy: { occurredAt: "desc" },
        skip: (input.page - 1) * input.limit,
        take: input.limit,
      }),
    ]);

    return {
      items: items.map(mapAuditEventRecordToDomain),
      page: input.page,
      limit: input.limit,
      total,
      totalPages: Math.max(1, Math.ceil(total / input.limit)),
    };
  }
}

import type { ProposalRepository, ProposalVersionRepository } from "../../domain/proposals/proposal-repository";
import type { TransactionRunner } from "../shared/transaction";
import type { AuditWriter } from "../audit/audit-writer";
import type { CreateProposalDraftCommand, CreateProposalDraftCommandHandler, ProposalDetailDto } from "./create-proposal-draft";
import { toProposalDetailDto } from "./proposal-mappers";

export class CreateProposalDraftHandler implements CreateProposalDraftCommandHandler {
  constructor(
    private readonly proposalRepository: ProposalRepository,
    private readonly proposalVersionRepository: ProposalVersionRepository,
    private readonly transactionRunner: TransactionRunner,
    private readonly auditWriter: AuditWriter,
  ) {}

  async execute(command: CreateProposalDraftCommand): Promise<ProposalDetailDto> {
    return this.transactionRunner.runInTransaction(async () => {
      const proposalId = crypto.randomUUID();
      const versionId = crypto.randomUUID();
      const now = new Date().toISOString();

      const proposal = {
        id: proposalId,
        title: command.input.title,
        summary: command.input.summary,
        proposalType: (command.input.proposalType ?? "GENERAL") as ProposalDetailDto["proposalType"],
        currentStage: "DRAFT" as const,
        proposerPersonId: command.actor.personId,
        proposerMemberId: command.actor.memberId,
        currentVersionId: undefined,
        createdAt: now,
        updatedAt: now,
      };

      await this.proposalRepository.createProposal(proposal);

      const version = {
        id: versionId,
        proposalId,
        versionNumber: 1,
        titleSnapshot: command.input.title,
        bodyMarkdown: command.input.bodyMarkdown,
        createdByPersonId: command.actor.personId,
        createdAt: now,
      };

      await this.proposalVersionRepository.createProposalVersion(version);

      proposal.currentVersionId = versionId;
      proposal.updatedAt = new Date().toISOString();

      await this.proposalRepository.updateProposal(proposal);

      await this.auditWriter.write({
        actionType: "CREATE",
        entityType: "Proposal",
        entityId: proposal.id,
        actorPersonId: command.actor.personId,
        summary: "Proposal draft created.",
        metadata: {
          proposalType: proposal.proposalType,
          currentVersionId: proposal.currentVersionId,
        },
      });

      return toProposalDetailDto(proposal);
    });
  }
}

22. apps/gov-api/src/routes/audit.ts

import { Hono } from "hono";
import type { HonoEnv } from "../types/hono";
import { jsonList } from "../http/json";
import { requireAuth } from "../middleware/require-auth";
import { requireAuthority } from "../middleware/require-authority";
import { parsePageInput } from "../application/shared/pagination";

export function createAuditRoutes(): Hono<HonoEnv> {
  const app = new Hono<HonoEnv>();

  app.get("/events", requireAuth, requireAuthority("audit.read"), async (c) => {
    const appContext = c.get("appContext");
    const page = parsePageInput({
      page: c.req.query("page"),
      limit: c.req.query("limit"),
    });

    const result = await appContext.handlers.audit.listEvents.execute({
      page,
      filters: {
        actionType: c.req.query("actionType") ?? undefined,
        entityType: c.req.query("entityType") ?? undefined,
        entityId: c.req.query("entityId") ?? undefined,
      },
    });

    return c.json(
      jsonList(result.items, {
        page: result.page,
        limit: result.limit,
        total: result.total,
        totalPages: result.totalPages,
      }),
      200,
    );
  });

  return app;
}
