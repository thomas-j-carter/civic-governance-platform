2. apps/gov-api/src/middleware/require-authority.test.ts

import { describe, expect, it } from "vitest";
import { Hono } from "hono";
import type { HonoEnv } from "../types/hono";
import { requestContextMiddleware } from "./request-context";
import { errorHandler } from "./error-handler";
import { requireAuthority } from "./require-authority";

describe("requireAuthority middleware", () => {
  function makeApp(authorityGrants: string[]) {
    const app = new Hono<HonoEnv>();

    app.use("*", requestContextMiddleware);
    app.use("*", errorHandler);

    app.use("*", async (c, next) => {
      const requestContext = c.get("requestContext");
      c.set("requestContext", {
        ...requestContext,
        actor: {
          personId: "person-1",
          isAuthenticated: true,
          roles: [],
          authorityGrants,
        },
      });
      await next();
    });

    app.get("/protected", requireAuthority("proposal.submit"), (c) => {
      return c.json({ ok: true }, 200);
    });

    return app;
  }

  it("allows access when required authority is present", async () => {
    const app = makeApp(["proposal.submit"]);
    const res = await app.request("http://localhost/protected");

    expect(res.status).toBe(200);
    await expect(res.json()).resolves.toEqual({ ok: true });
  });

  it("returns problem details when authority is missing", async () => {
    const app = makeApp(["proposal.create_draft"]);
    const res = await app.request("http://localhost/protected");

    expect(res.status).toBe(403);

    const body = await res.json();
    expect(body.title).toBe("Forbidden");
    expect(body.code).toBe("insufficient_authority");
  });
});
