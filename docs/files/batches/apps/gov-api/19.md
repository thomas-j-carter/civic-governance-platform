2. apps/gov-api/src/infrastructure/persistence/in-memory/in-memory-member-admin-repository.ts

import type { Member } from "../../../domain/membership/member";
import type { ListMembersResult, MemberAdministrationRepository } from "../../../domain/membership/member-repository";
import type { MemberRepository } from "../../../domain/authority/authority-repository";

export class InMemoryMemberAdministrationRepository
  implements MemberAdministrationRepository
{
  constructor(private readonly memberRepository: MemberRepository & {
    items?: Map<string, Member>;
  }) {}

  async listMembers(): Promise<ListMembersResult> {
    const items = this.memberRepository.items
      ? [...this.memberRepository.items.values()]
      : [];

    return {
      items: items
        .sort((a, b) => b.createdAt.localeCompare(a.createdAt))
        .map((item) => structuredClone(item)),
    };
  }

  async updateMember(member: Member): Promise<void> {
    if ("items" in this.memberRepository && this.memberRepository.items) {
      this.memberRepository.items.set(member.id, structuredClone(member));
      return;
    }

    throw new Error("In-memory member repository does not support updateMember.");
  }
}

4. apps/gov-api/src/infrastructure/prisma/repositories/prisma-member-admin-repository.ts

import type { PrismaClient } from "@prisma/client";
import type { Member } from "../../../domain/membership/member";
import type {
  ListMembersResult,
  MemberAdministrationRepository,
} from "../../../domain/membership/member-repository";
import { mapMemberRecordToDomain } from "../mappers/authority-mappers";

export class PrismaMemberAdministrationRepository
  implements MemberAdministrationRepository
{
  constructor(private readonly prisma: PrismaClient) {}

  async listMembers(): Promise<ListMembersResult> {
    const records = await this.prisma.member.findMany({
      orderBy: { createdAt: "desc" },
    });

    return {
      items: records.map(mapMemberRecordToDomain),
    };
  }

  async updateMember(member: Member): Promise<void> {
    await this.prisma.member.update({
      where: { id: member.id },
      data: {
        memberNumber: member.memberNumber ?? null,
        membershipClassId: member.membershipClassId ?? null,
        status: member.status,
        admittedAt: member.admittedAt ? new Date(member.admittedAt) : null,
        endedAt: member.endedAt ? new Date(member.endedAt) : null,
        updatedAt: new Date(member.updatedAt),
      },
    });
  }
}

14. apps/gov-api/src/application/members/admit-member-handler.ts

import { ConflictError, NotFoundError } from "../../http/errors";
import type { MemberRepository } from "../../domain/authority/authority-repository";
import type { MemberAdministrationRepository } from "../../domain/membership/member-repository";
import type { AuditWriter } from "../audit/audit-writer";
import type { AdmitMemberCommand, AdmitMemberCommandHandler } from "./admit-member";
import type { MemberDto } from "./member-dto";
import { toMemberDto } from "./member-mappers";

export class AdmitMemberHandler implements AdmitMemberCommandHandler {
  constructor(
    private readonly memberRepository: MemberRepository,
    private readonly memberAdminRepository: MemberAdministrationRepository,
    private readonly auditWriter: AuditWriter,
  ) {}

  async execute(command: AdmitMemberCommand): Promise<MemberDto> {
    const member = await this.memberRepository.getMemberById(command.memberId);

    if (!member) {
      throw new NotFoundError(`Member ${command.memberId} was not found.`);
    }

    if (member.status === "ACTIVE") {
      throw new ConflictError("Member is already active.", {
        code: "member_already_active",
      });
    }

    member.status = "ACTIVE";
    member.admittedAt = command.input.admittedAt ?? new Date().toISOString();
    member.endedAt = undefined;
    member.updatedAt = new Date().toISOString();

    await this.memberAdminRepository.updateMember(member);

    await this.auditWriter.write({
      actionType: "APPROVE",
      entityType: "Member",
      entityId: member.id,
      actorPersonId: command.actor.personId,
      summary: "Member admitted.",
    });

    return toMemberDto(member);
  }
}

import { Hono } from "hono";
import type { HonoEnv } from "../types/hono";
import { jsonData } from "../http/json";
import { requireAuth } from "../middleware/require-auth";
import { requireAuthority } from "../middleware/require-authority";
import { NotFoundError } from "../http/errors";

export function createMemberRoutes(): Hono<HonoEnv> {
  const app = new Hono<HonoEnv>();

  app.get("/", requireAuth, requireAuthority("member.read"), async (c) => {
    const appContext = c.get("appContext");
    const result = await appContext.handlers.members.list.execute();
    return c.json(jsonData(result), 200);
  });

  app.get("/:memberId", requireAuth, requireAuthority("member.read"), async (c) => {
    const appContext = c.get("appContext");
    const memberId = c.req.param("memberId");

    const result = await appContext.handlers.members.get.execute({ memberId });

    if (!result) {
      throw new NotFoundError(`Member ${memberId} was not found.`);
    }

    return c.json(jsonData(result), 200);
  });

  app.post(
    "/:memberId/actions/admit",
    requireAuth,
    requireAuthority("member.admit"),
    async (c) => {
      const appContext = c.get("appContext");
      const actor = c.get("requestContext").actor!;
      const memberId = c.req.param("memberId");
      const body = await c.req.json<{ admittedAt?: string }>().catch(() => ({}));

      const result = await appContext.handlers.members.admit.execute({
        actor: { personId: actor.personId },
        memberId,
        input: {
          admittedAt: body.admittedAt,
        },
      });

      return c.json(jsonData(result), 200);
    },
  );

  app.post(
    "/:memberId/actions/suspend",
    requireAuth,
    requireAuthority("member.suspend"),
    async (c) => {
      const appContext = c.get("appContext");
      const actor = c.get("requestContext").actor!;
      const memberId = c.req.param("memberId");
      const body = await c.req.json<{ reason?: string }>().catch(() => ({}));

      const result = await appContext.handlers.members.suspend.execute({
        actor: { personId: actor.personId },
        memberId,
        input: {
          reason: body.reason,
        },
      });

      return c.json(jsonData(result), 200);
    },
  );

  app.post(
    "/:memberId/actions/end",
    requireAuth,
    requireAuthority("member.end"),
    async (c) => {
      const appContext = c.get("appContext");
      const actor = c.get("requestContext").actor!;
      const memberId = c.req.param("memberId");
      const body = await c.req.json<{ endedAt?: string; reason?: string }>().catch(() => ({}));

      const result = await appContext.handlers.members.end.execute({
        actor: { personId: actor.personId },
        memberId,
        input: {
          endedAt: body.endedAt,
          reason: body.reason,
        },
      });

      return c.json(jsonData(result), 200);
    },
  );

  return app;
}
