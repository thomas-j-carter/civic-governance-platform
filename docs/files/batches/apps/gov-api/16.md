2. apps/gov-api/src/application/auth/jwks-token-validator.ts

import {
  createRemoteJWKSet,
  errors as joseErrors,
  jwtVerify,
  type JWTPayload,
  type JWTVerifyOptions,
} from "jose";
import type { AuthConfig } from "../../config/auth";
import type { AuthPrincipal } from "../../domain/auth/auth-principal";
import type { TokenValidator } from "./token-validator";
import { UnauthorizedError } from "../../http/errors";

export class JwksTokenValidator implements TokenValidator {
  private readonly jwks: ReturnType<typeof createRemoteJWKSet>;
  private readonly verifyOptions: JWTVerifyOptions;

  constructor(private readonly config: AuthConfig) {
    this.jwks = createRemoteJWKSet(new URL(config.AUTH_JWKS_URL!));
    this.verifyOptions = {
      issuer: config.AUTH_ISSUER,
      audience: config.AUTH_AUDIENCE,
    };
  }

  async validateBearerToken(token: string): Promise<AuthPrincipal> {
    try {
      const { payload } = await jwtVerify(token, this.jwks, this.verifyOptions);

      return this.toPrincipal(payload);
    } catch (error) {
      throw mapJoseErrorToUnauthorized(error);
    }
  }

  private toPrincipal(payload: JWTPayload): AuthPrincipal {
    if (!payload.sub || typeof payload.sub !== "string") {
      throw new UnauthorizedError("JWT subject claim was missing or invalid.", {
        code: "invalid_token_subject",
        type: "https://gov-api.ardtiresociety.org/problems/invalid-token-subject",
      });
    }

    const issuer =
      typeof payload.iss === "string" ? payload.iss : this.config.AUTH_ISSUER!;

    const audience = normalizeAudience(payload.aud);
    const email = typeof payload.email === "string" ? payload.email : undefined;
    const name = typeof payload.name === "string" ? payload.name : undefined;
    const preferredUsername =
      typeof payload.preferred_username === "string"
        ? payload.preferred_username
        : undefined;

    return {
      subject: payload.sub,
      issuer,
      audience,
      email,
      name,
      preferredUsername,
      claims: payload as Record<string, unknown>,
    };
  }
}

function normalizeAudience(
  aud: JWTPayload["aud"],
): string | undefined {
  if (typeof aud === "string") {
    return aud;
  }

  if (Array.isArray(aud)) {
    return aud.find((value): value is string => typeof value === "string");
  }

  return undefined;
}

function mapJoseErrorToUnauthorized(error: unknown): UnauthorizedError {
  if (error instanceof joseErrors.JWTExpired) {
    return new UnauthorizedError("JWT has expired.", {
      code: "token_expired",
      type: "https://gov-api.ardtiresociety.org/problems/token-expired",
    });
  }

  if (error instanceof joseErrors.JWTClaimValidationFailed) {
    return new UnauthorizedError("JWT claim validation failed.", {
      code: "token_claim_validation_failed",
      type: "https://gov-api.ardtiresociety.org/problems/token-claim-validation-failed",
      detail: error.message,
    });
  }

  if (
    error instanceof joseErrors.JWSSignatureVerificationFailed ||
    error instanceof joseErrors.JWSInvalid ||
    error instanceof joseErrors.JWTInvalid
  ) {
    return new UnauthorizedError("JWT signature or structure was invalid.", {
      code: "token_invalid",
      type: "https://gov-api.ardtiresociety.org/problems/token-invalid",
    });
  }

  if (
    error instanceof joseErrors.JWKSNoMatchingKey ||
    error instanceof joseErrors.JOSEError
  ) {
    return new UnauthorizedError("JWT could not be verified against JWKS.", {
      code: "token_verification_failed",
      type: "https://gov-api.ardtiresociety.org/problems/token-verification-failed",
      detail: error.message,
    });
  }

  return new UnauthorizedError("JWT validation failed.", {
    code: "token_validation_failed",
    type: "https://gov-api.ardtiresociety.org/problems/token-validation-failed",
  });
}

This implementation follows the documented jose approach: use createRemoteJWKSet(new URL(...)) and pass that to jwtVerify with issuer/audience options.

import { describe, expect, it } from "vitest";
import { SignJWT, generateKeyPair, exportJWK } from "jose";
import { JwksTokenValidator } from "./jwks-token-validator";
import { UnauthorizedError } from "../../http/errors";

describe("JwksTokenValidator", () => {
  it("verifies a JWT against a JWKS endpoint and returns a principal", async () => {
    const { privateKey, publicKey } = await generateKeyPair("RS256");
    const jwk = await exportJWK(publicKey);
    jwk.kid = "test-key-1";
    jwk.alg = "RS256";
    jwk.use = "sig";

    const issuer = "https://issuer.example.com/";
    const audience = "gov-api";
    const jwksUrl = "https://issuer.example.com/.well-known/jwks.json";

    const originalFetch = globalThis.fetch;

    globalThis.fetch = async (input) => {
      const url = typeof input === "string" ? input : input instanceof URL ? input.href : input.url;

      if (url === jwksUrl) {
        return new Response(
          JSON.stringify({
            keys: [jwk],
          }),
          {
            status: 200,
            headers: {
              "content-type": "application/json",
            },
          },
        );
      }

      return new Response("not found", { status: 404 });
    };

    try {
      const token = await new SignJWT({
        email: "dev@example.com",
        name: "Dev User",
        preferred_username: "dev",
      })
        .setProtectedHeader({ alg: "RS256", kid: "test-key-1" })
        .setIssuer(issuer)
        .setAudience(audience)
        .setSubject("person-1")
        .setIssuedAt()
        .setExpirationTime("10m")
        .sign(privateKey);

      const validator = new JwksTokenValidator({
        AUTH_MODE: "jwks",
        AUTH_ISSUER: issuer,
        AUTH_AUDIENCE: audience,
        AUTH_JWKS_URL: jwksUrl,
        DEV_AUTH_BEARER_TOKEN: "dev-token",
        DEV_AUTH_PERSON_ID: "seed-person-1",
        DEV_AUTH_EMAIL: "dev@example.com",
      });

      const principal = await validator.validateBearerToken(token);

      expect(principal.subject).toBe("person-1");
      expect(principal.issuer).toBe(issuer);
      expect(principal.audience).toBe(audience);
      expect(principal.email).toBe("dev@example.com");
      expect(principal.name).toBe("Dev User");
      expect(principal.preferredUsername).toBe("dev");
    } finally {
      globalThis.fetch = originalFetch;
    }
  });

  it("rejects a token with the wrong audience", async () => {
    const { privateKey, publicKey } = await generateKeyPair("RS256");
    const jwk = await exportJWK(publicKey);
    jwk.kid = "test-key-1";
    jwk.alg = "RS256";
    jwk.use = "sig";

    const issuer = "https://issuer.example.com/";
    const audience = "gov-api";
    const jwksUrl = "https://issuer.example.com/.well-known/jwks.json";

    const originalFetch = globalThis.fetch;

    globalThis.fetch = async () =>
      new Response(JSON.stringify({ keys: [jwk] }), {
        status: 200,
        headers: { "content-type": "application/json" },
      });

    try {
      const token = await new SignJWT({})
        .setProtectedHeader({ alg: "RS256", kid: "test-key-1" })
        .setIssuer(issuer)
        .setAudience("wrong-audience")
        .setSubject("person-1")
        .setIssuedAt()
        .setExpirationTime("10m")
        .sign(privateKey);

      const validator = new JwksTokenValidator({
        AUTH_MODE: "jwks",
        AUTH_ISSUER: issuer,
        AUTH_AUDIENCE: audience,
        AUTH_JWKS_URL: jwksUrl,
        DEV_AUTH_BEARER_TOKEN: "dev-token",
        DEV_AUTH_PERSON_ID: "seed-person-1",
        DEV_AUTH_EMAIL: "dev@example.com",
      });

      await expect(validator.validateBearerToken(token)).rejects.toBeInstanceOf(
        UnauthorizedError,
      );
    } finally {
      globalThis.fetch = originalFetch;
    }
  });
});

9. apps/gov-api/src/infrastructure/auth/auth-readiness.ts refinement

import type { ReadinessDependency } from "../../health/readiness";
import type { AuthConfig } from "../../config/auth";

export function createAuthReadinessDependency(
  config: AuthConfig,
): ReadinessDependency {
  return {
    name: "auth",
    async check() {
      if (config.AUTH_MODE === "dev") {
        return { ok: true, detail: "dev auth mode enabled" };
      }

      if (!config.AUTH_ISSUER || !config.AUTH_AUDIENCE || !config.AUTH_JWKS_URL) {
        return { ok: false, detail: "jwks auth mode is not fully configured" };
      }

      try {
        new URL(config.AUTH_JWKS_URL);
      } catch {
        return { ok: false, detail: "AUTH_JWKS_URL is not a valid URL" };
      }

      return { ok: true, detail: "jwks auth mode configured" };
    },
  };
}
