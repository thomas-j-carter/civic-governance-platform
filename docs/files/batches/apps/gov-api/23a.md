10. apps/gov-api/src/application/records/create-record-handler.ts

import type { OfficialRecordRepository } from "../../domain/records/record-repository";
import type { AuditWriter } from "../audit/audit-writer";
import type {
  CreateOfficialRecordCommand,
  CreateOfficialRecordCommandHandler,
  OfficialRecordDto,
} from "./create-record";
import { toOfficialRecordDto } from "./record-mappers";

export class CreateOfficialRecordHandler implements CreateOfficialRecordCommandHandler {
  constructor(
    private readonly officialRecordRepository: OfficialRecordRepository,
    private readonly auditWriter: AuditWriter,
  ) {}

  async execute(command: CreateOfficialRecordCommand): Promise<OfficialRecordDto> {
    const now = new Date().toISOString();

    const record = {
      id: crypto.randomUUID(),
      recordType: command.input.recordType,
      title: command.input.title,
      summary: command.input.summary,
      sourceEntityType: command.input.sourceEntityType,
      sourceEntityId: command.input.sourceEntityId,
      status: "DRAFT" as const,
      createdAt: now,
      updatedAt: now,
    };

    await this.officialRecordRepository.createRecord(record);

    await this.auditWriter.write({
      actionType: "CREATE",
      entityType: "OfficialRecord",
      entityId: record.id,
      actorPersonId: command.actor.personId,
      summary: "Official record created.",
      metadata: {
        recordType: record.recordType,
        sourceEntityType: record.sourceEntityType,
        sourceEntityId: record.sourceEntityId,
      },
    });

    return toOfficialRecordDto(record);
  }
}

14. apps/gov-api/src/application/records/promote-certification-to-record-handler.ts

import { ConflictError, NotFoundError } from "../../http/errors";
import type { CertificationRepository } from "../../domain/certifications/certification-repository";
import type { OfficialRecordRepository } from "../../domain/records/record-repository";
import type { AuditWriter } from "../audit/audit-writer";
import type {
  PromoteCertificationToRecordCommand,
  PromoteCertificationToRecordCommandHandler,
} from "./promote-certification-to-record";
import type { OfficialRecordDto } from "./create-record";
import { toOfficialRecordDto } from "./record-mappers";

export class PromoteCertificationToRecordHandler
  implements PromoteCertificationToRecordCommandHandler
{
  constructor(
    private readonly certificationRepository: CertificationRepository,
    private readonly officialRecordRepository: OfficialRecordRepository,
    private readonly auditWriter: AuditWriter,
  ) {}

  async execute(
    command: PromoteCertificationToRecordCommand,
  ): Promise<OfficialRecordDto> {
    const certification = await this.certificationRepository.getCertificationById(
      command.input.certificationId,
    );

    if (!certification) {
      throw new NotFoundError(
        `Certification ${command.input.certificationId} was not found.`,
      );
    }

    if (certification.status !== "CERTIFIED") {
      throw new ConflictError("Only certified results may be promoted to an official record.", {
        code: "certification_not_certified",
      });
    }

    const now = new Date().toISOString();

    const record = {
      id: crypto.randomUUID(),
      recordType: command.input.recordType,
      title: command.input.title,
      summary: command.input.summary,
      sourceEntityType: "certification",
      sourceEntityId: certification.id,
      status: "DRAFT" as const,
      createdAt: now,
      updatedAt: now,
    };

    await this.officialRecordRepository.createRecord(record);

    await this.auditWriter.write({
      actionType: "CREATE",
      entityType: "OfficialRecord",
      entityId: record.id,
      actorPersonId: command.actor.personId,
      summary: "Certified outcome promoted to official record draft.",
      metadata: {
        certificationId: certification.id,
        ballotId: certification.ballotId,
        recordType: record.recordType,
      },
    });

    return toOfficialRecordDto(record);
  }
}

15. apps/gov-api/src/routes/records.ts

import { Hono } from "hono";
import type { HonoEnv } from "../types/hono";
import { jsonData, jsonList } from "../http/json";
import { requireAuth } from "../middleware/require-auth";
import { requireAuthority } from "../middleware/require-authority";
import { parsePageInput } from "../application/shared/pagination";
import { NotFoundError, ValidationError } from "../http/errors";
import { runIdempotentCommand } from "../http/idempotency";

export function createRecordRoutes(): Hono<HonoEnv> {
  const app = new Hono<HonoEnv>();

  app.get("/", requireAuth, requireAuthority("record.read"), async (c) => {
    const appContext = c.get("appContext");

    const page = parsePageInput({
      page: c.req.query("page"),
      limit: c.req.query("limit"),
    });

    const result = await appContext.handlers.records.list.execute({
      page,
      filters: {
        recordType: c.req.query("recordType") ?? undefined,
        status: c.req.query("status") ?? undefined,
      },
    });

    return c.json(
      jsonList(result.items, {
        page: result.page,
        limit: result.limit,
        total: result.total,
        totalPages: result.totalPages,
      }),
      200,
    );
  });

  app.get("/:recordId", requireAuth, requireAuthority("record.read"), async (c) => {
    const appContext = c.get("appContext");
    const recordId = c.req.param("recordId");

    const result = await appContext.handlers.records.get.execute({
      recordId,
    });

    if (!result) {
      throw new NotFoundError(`Record ${recordId} was not found.`);
    }

    return c.json(jsonData(result), 200);
  });

  app.get(
    "/:recordId/versions",
    requireAuth,
    requireAuthority("record.read"),
    async (c) => {
      const appContext = c.get("appContext");
      const recordId = c.req.param("recordId");

      const result = await appContext.handlers.records.listVersions.execute({
        recordId,
      });

      return c.json(jsonData(result), 200);
    },
  );

  app.post("/", requireAuth, requireAuthority("record.create"), async (c) => {
    const appContext = c.get("appContext");
    const actor = c.get("requestContext").actor!;
    const body = await c.req.json<{
      recordType?: string;
      title?: string;
      summary?: string;
      sourceEntityType?: string;
      sourceEntityId?: string;
    }>();

    const errors: Record<string, string[]> = {};

    if (!body.recordType?.trim()) errors.recordType = ["recordType is required."];
    if (!body.title?.trim()) errors.title = ["title is required."];
    if (!body.sourceEntityType?.trim()) {
      errors.sourceEntityType = ["sourceEntityType is required."];
    }
    if (!body.sourceEntityId?.trim()) {
      errors.sourceEntityId = ["sourceEntityId is required."];
    }

    if (Object.keys(errors).length > 0) {
      throw new ValidationError("Create record request was invalid.", errors);
    }

    const requestPayload = {
      recordType: body.recordType!,
      title: body.title!,
      summary: body.summary,
      sourceEntityType: body.sourceEntityType!,
      sourceEntityId: body.sourceEntityId!,
    };

    const result = await runIdempotentCommand({
      c,
      operationName: "record.create",
      requestBody: requestPayload,
      execute: async () =>
        appContext.handlers.records.create.execute({
          actor: { personId: actor.personId },
          input: requestPayload,
        }),
    });

    return c.json(jsonData(result), 201);
  });

  app.post(
    "/promotions/certification-to-record",
    requireAuth,
    requireAuthority("record.promote"),
    async (c) => {
      const appContext = c.get("appContext");
      const actor = c.get("requestContext").actor!;
      const body = await c.req.json<{
        certificationId?: string;
        recordType?: string;
        title?: string;
        summary?: string;
      }>();

      const errors: Record<string, string[]> = {};

      if (!body.certificationId?.trim()) {
        errors.certificationId = ["certificationId is required."];
      }
      if (!body.recordType?.trim()) {
        errors.recordType = ["recordType is required."];
      }
      if (!body.title?.trim()) {
        errors.title = ["title is required."];
      }

      if (Object.keys(errors).length > 0) {
        throw new ValidationError(
          "Promote certification to record request was invalid.",
          errors,
        );
      }

      const requestPayload = {
        certificationId: body.certificationId!,
        recordType: body.recordType!,
        title: body.title!,
        summary: body.summary,
      };

      const result = await runIdempotentCommand({
        c,
        operationName: "record.promoteCertificationToRecord",
        requestBody: requestPayload,
        execute: async () =>
          appContext.handlers.recordPromotion.promoteCertificationToRecord.execute({
            actor: { personId: actor.personId },
            input: requestPayload,
          }),
      });

      return c.json(jsonData(result), 201);
    },
  );

  app.post(
    "/:recordId/actions/officialize",
    requireAuth,
    requireAuthority("record.officialize"),
    async (c) => {
      const appContext = c.get("appContext");
      const actor = c.get("requestContext").actor!;
      const recordId = c.req.param("recordId");
      const body = await c.req.json<{ officializedAt?: string; note?: string }>().catch(() => ({}));

      const result = await appContext.handlers.recordPromotion.officializeRecord.execute({
        actor: { personId: actor.personId },
        recordId,
        input: {
          officializedAt: body.officializedAt,
          note: body.note,
        },
      });

      return c.json(jsonData(result), 200);
    },
  );

  return app;
}
