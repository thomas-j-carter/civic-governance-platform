import type { PersonIdentityLink } from "../../../domain/auth/person-identity-link";
import type { PersonIdentityLinkRepository } from "../../../domain/auth/auth-repository";

export class InMemoryPersonIdentityLinkRepository
  implements PersonIdentityLinkRepository
{
  private readonly items = new Map<string, PersonIdentityLink>();

  async createLink(link: PersonIdentityLink): Promise<void> {
    this.items.set(link.id, structuredClone(link));
  }

  async findByIssuerAndSubject(
    issuer: string,
    subject: string,
  ): Promise<PersonIdentityLink | null> {
    const item = [...this.items.values()].find(
      (value) => value.issuer === issuer && value.subject === subject,
    );
    return item ? structuredClone(item) : null;
  }

  async findByEmail(email: string): Promise<PersonIdentityLink | null> {
    const lowered = email.toLowerCase();
    const item = [...this.items.values()].find(
      (value) => value.email?.toLowerCase() === lowered,
    );
    return item ? structuredClone(item) : null;
  }

  async findById(id: string): Promise<PersonIdentityLink | null> {
    const item = this.items.get(id);
    return item ? structuredClone(item) : null;
  }

  async listLinks(): Promise<PersonIdentityLink[]> {
    return [...this.items.values()]
      .sort((a, b) => b.createdAt.localeCompare(a.createdAt))
      .map((item) => structuredClone(item));
  }

  async listLinksByPersonId(personId: string): Promise<PersonIdentityLink[]> {
    return [...this.items.values()]
      .filter((item) => item.personId === personId)
      .sort((a, b) => b.createdAt.localeCompare(a.createdAt))
      .map((item) => structuredClone(item));
  }
}

import type { PrismaClient } from "@prisma/client";
import type { PersonIdentityLink } from "../../../domain/auth/person-identity-link";
import type { PersonIdentityLinkRepository } from "../../../domain/auth/auth-repository";
import { mapPersonIdentityLinkRecordToDomain } from "../mappers/auth-mappers";

export class PrismaPersonIdentityLinkRepository
  implements PersonIdentityLinkRepository
{
  constructor(private readonly prisma: PrismaClient) {}

  async createLink(link: PersonIdentityLink): Promise<void> {
    await this.prisma.personIdentityLink.create({
      data: {
        id: link.id,
        personId: link.personId,
        issuer: link.issuer,
        subject: link.subject,
        email: link.email ?? null,
        createdAt: new Date(link.createdAt),
        updatedAt: new Date(link.updatedAt),
      },
    });
  }

  async findByIssuerAndSubject(
    issuer: string,
    subject: string,
  ): Promise<PersonIdentityLink | null> {
    const record = await this.prisma.personIdentityLink.findFirst({
      where: { issuer, subject },
    });

    return record ? mapPersonIdentityLinkRecordToDomain(record) : null;
  }

  async findByEmail(email: string): Promise<PersonIdentityLink | null> {
    const record = await this.prisma.personIdentityLink.findFirst({
      where: { email: { equals: email, mode: "insensitive" } },
    });

    return record ? mapPersonIdentityLinkRecordToDomain(record) : null;
  }

  async findById(id: string): Promise<PersonIdentityLink | null> {
    const record = await this.prisma.personIdentityLink.findUnique({
      where: { id },
    });

    return record ? mapPersonIdentityLinkRecordToDomain(record) : null;
  }

  async listLinks(): Promise<PersonIdentityLink[]> {
    const records = await this.prisma.personIdentityLink.findMany({
      orderBy: { createdAt: "desc" },
    });

    return records.map(mapPersonIdentityLinkRecordToDomain);
  }

  async listLinksByPersonId(personId: string): Promise<PersonIdentityLink[]> {
    const records = await this.prisma.personIdentityLink.findMany({
      where: { personId },
      orderBy: { createdAt: "desc" },
    });

    return records.map(mapPersonIdentityLinkRecordToDomain);
  }
}

8. apps/gov-api/src/application/auth/create-person-identity-link-handler.ts

import { ConflictError } from "../../http/errors";
import type { PersonIdentityLinkRepository } from "../../domain/auth/auth-repository";
import type { AuditWriter } from "../audit/audit-writer";
import type {
  CreatePersonIdentityLinkCommand,
  CreatePersonIdentityLinkCommandHandler,
  PersonIdentityLinkDto,
} from "./create-person-identity-link";
import { toPersonIdentityLinkDto } from "./person-identity-link-mappers";

export class CreatePersonIdentityLinkHandler
  implements CreatePersonIdentityLinkCommandHandler
{
  constructor(
    private readonly repository: PersonIdentityLinkRepository,
    private readonly auditWriter: AuditWriter,
  ) {}

  async execute(
    command: CreatePersonIdentityLinkCommand,
  ): Promise<PersonIdentityLinkDto> {
    const existingByIssuerSubject = await this.repository.findByIssuerAndSubject(
      command.input.issuer,
      command.input.subject,
    );

    if (existingByIssuerSubject) {
      throw new ConflictError("A person identity link already exists for issuer and subject.", {
        code: "person_identity_link_exists",
        type: "https://gov-api.ardtiresociety.org/problems/person-identity-link-exists",
      });
    }

    const now = new Date().toISOString();

    const link = {
      id: crypto.randomUUID(),
      personId: command.input.personId,
      issuer: command.input.issuer,
      subject: command.input.subject,
      email: command.input.email,
      createdAt: now,
      updatedAt: now,
    };

    await this.repository.createLink(link);

    await this.auditWriter.write({
      actionType: "CREATE",
      entityType: "PersonIdentityLink",
      entityId: link.id,
      actorPersonId: command.actor.personId,
      summary: "Person identity link created.",
      metadata: {
        linkedPersonId: link.personId,
        issuer: link.issuer,
        subject: link.subject,
        email: link.email,
      },
    });

    return toPersonIdentityLinkDto(link);
  }
}

11. apps/gov-api/src/routes/auth-links.ts

import { Hono } from "hono";
import type { HonoEnv } from "../types/hono";
import { jsonData } from "../http/json";
import { requireAuth } from "../middleware/require-auth";
import { requireAuthority } from "../middleware/require-authority";
import { NotFoundError, ValidationError } from "../http/errors";
import { runIdempotentCommand } from "../http/idempotency";

export function createAuthLinkRoutes(): Hono<HonoEnv> {
  const app = new Hono<HonoEnv>();

  app.get(
    "/identity-links",
    requireAuth,
    requireAuthority("identity_link.read"),
    async (c) => {
      const appContext = c.get("appContext");
      const personId = c.req.query("personId") ?? undefined;

      const result = await appContext.handlers.authLinks.list.execute({
        personId,
      });

      return c.json(jsonData(result), 200);
    },
  );

  app.get(
    "/identity-links/:linkId",
    requireAuth,
    requireAuthority("identity_link.read"),
    async (c) => {
      const appContext = c.get("appContext");
      const linkId = c.req.param("linkId");

      const result = await appContext.handlers.authLinks.get.execute({
        linkId,
      });

      if (!result) {
        throw new NotFoundError(`Identity link ${linkId} was not found.`);
      }

      return c.json(jsonData(result), 200);
    },
  );

  app.post(
    "/identity-links",
    requireAuth,
    requireAuthority("identity_link.create"),
    async (c) => {
      const appContext = c.get("appContext");
      const actor = c.get("requestContext").actor!;
      const body = await c.req.json<{
        personId?: string;
        issuer?: string;
        subject?: string;
        email?: string;
      }>();

      const errors: Record<string, string[]> = {};

      if (!body.personId?.trim()) {
        errors.personId = ["personId is required."];
      }
      if (!body.issuer?.trim()) {
        errors.issuer = ["issuer is required."];
      }
      if (!body.subject?.trim()) {
        errors.subject = ["subject is required."];
      }

      if (Object.keys(errors).length > 0) {
        throw new ValidationError("Create identity link request was invalid.", errors);
      }

      const requestPayload = {
        personId: body.personId!,
        issuer: body.issuer!,
        subject: body.subject!,
        email: body.email,
      };

      const result = await runIdempotentCommand({
        c,
        operationName: "identityLink.create",
        requestBody: requestPayload,
        execute: async () =>
          appContext.handlers.authLinks.create.execute({
            actor: { personId: actor.personId },
            input: requestPayload,
          }),
      });

      return c.json(jsonData(result), 201);
    },
  );

  return app;
}
