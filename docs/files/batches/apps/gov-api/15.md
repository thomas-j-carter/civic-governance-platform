1. apps/gov-api/src/config/auth.ts

import { z } from "zod";

const authConfigSchema = z.object({
  AUTH_MODE: z.enum(["dev", "jwks"]).default("dev"),
  AUTH_ISSUER: z.string().optional(),
  AUTH_AUDIENCE: z.string().optional(),
  AUTH_JWKS_URL: z.string().optional(),
  DEV_AUTH_BEARER_TOKEN: z.string().default("dev-token"),
  DEV_AUTH_PERSON_ID: z.string().default("seed-person-1"),
  DEV_AUTH_EMAIL: z.string().email().default("dev@example.com"),
});

export type AuthConfig = z.infer<typeof authConfigSchema>;

export function readAuthConfig(
  source: NodeJS.ProcessEnv = process.env,
): AuthConfig {
  const parsed = authConfigSchema.safeParse(source);

  if (!parsed.success) {
    const issues = parsed.error.issues
      .map((issue) => `${issue.path.join(".")}: ${issue.message}`)
      .join("; ");
    throw new Error(`Invalid auth configuration: ${issues}`);
  }

  const config = parsed.data;

  if (config.AUTH_MODE === "jwks") {
    if (!config.AUTH_ISSUER) {
      throw new Error("AUTH_ISSUER is required when AUTH_MODE=jwks.");
    }
    if (!config.AUTH_AUDIENCE) {
      throw new Error("AUTH_AUDIENCE is required when AUTH_MODE=jwks.");
    }
    if (!config.AUTH_JWKS_URL) {
      throw new Error("AUTH_JWKS_URL is required when AUTH_MODE=jwks.");
    }
  }

  return config;
}

import type { PersonIdentityLink } from "../../../domain/auth/person-identity-link";
import type { PersonIdentityLinkRepository } from "../../../domain/auth/auth-repository";

export class InMemoryPersonIdentityLinkRepository
  implements PersonIdentityLinkRepository
{
  private readonly items = new Map<string, PersonIdentityLink>();

  async createLink(link: PersonIdentityLink): Promise<void> {
    this.items.set(link.id, structuredClone(link));
  }

  async findByIssuerAndSubject(
    issuer: string,
    subject: string,
  ): Promise<PersonIdentityLink | null> {
    const item = [...this.items.values()].find(
      (value) => value.issuer === issuer && value.subject === subject,
    );
    return item ? structuredClone(item) : null;
  }

  async findByEmail(email: string): Promise<PersonIdentityLink | null> {
    const lowered = email.toLowerCase();
    const item = [...this.items.values()].find(
      (value) => value.email?.toLowerCase() === lowered,
    );
    return item ? structuredClone(item) : null;
  }
}

import type { PrismaClient } from "@prisma/client";
import type { PersonIdentityLink } from "../../../domain/auth/person-identity-link";
import type { PersonIdentityLinkRepository } from "../../../domain/auth/auth-repository";
import { mapPersonIdentityLinkRecordToDomain } from "../mappers/auth-mappers";

export class PrismaPersonIdentityLinkRepository
  implements PersonIdentityLinkRepository
{
  constructor(private readonly prisma: PrismaClient) {}

  async createLink(link: PersonIdentityLink): Promise<void> {
    await this.prisma.personIdentityLink.create({
      data: {
        id: link.id,
        personId: link.personId,
        issuer: link.issuer,
        subject: link.subject,
        email: link.email ?? null,
        createdAt: new Date(link.createdAt),
        updatedAt: new Date(link.updatedAt),
      },
    });
  }

  async findByIssuerAndSubject(
    issuer: string,
    subject: string,
  ): Promise<PersonIdentityLink | null> {
    const record = await this.prisma.personIdentityLink.findFirst({
      where: { issuer, subject },
    });

    return record ? mapPersonIdentityLinkRecordToDomain(record) : null;
  }

  async findByEmail(email: string): Promise<PersonIdentityLink | null> {
    const record = await this.prisma.personIdentityLink.findFirst({
      where: { email: { equals: email, mode: "insensitive" } },
    });

    return record ? mapPersonIdentityLinkRecordToDomain(record) : null;
  }
}

11. apps/gov-api/src/application/auth/principal-to-person-service.ts

import type { PersonIdentityLinkRepository } from "../../domain/auth/auth-repository";
import type { AuthPrincipal } from "../../domain/auth/auth-principal";
import { UnauthorizedError } from "../../http/errors";

export class PrincipalToPersonService {
  constructor(private readonly repository: PersonIdentityLinkRepository) {}

  async resolvePersonId(principal: AuthPrincipal): Promise<string> {
    const byIssuerSubject = await this.repository.findByIssuerAndSubject(
      principal.issuer,
      principal.subject,
    );

    if (byIssuerSubject) {
      return byIssuerSubject.personId;
    }

    if (principal.email) {
      const byEmail = await this.repository.findByEmail(principal.email);
      if (byEmail) {
        return byEmail.personId;
      }
    }

    throw new UnauthorizedError("Authenticated principal is not linked to a person.", {
      code: "principal_not_linked",
      type: "https://gov-api.ardtiresociety.org/problems/principal-not-linked",
    });
  }
}

12. apps/gov-api/src/application/auth/authenticated-actor-service.ts

import type { AuthPrincipal } from "../../domain/auth/auth-principal";
import type { AuthenticatedActor } from "../../context/request-context";
import type { AuthorityResolutionService } from "../authority/authority-resolution-service";
import type { PrincipalToPersonService } from "./principal-to-person-service";

export class AuthenticatedActorService {
  constructor(
    private readonly principalToPersonService: PrincipalToPersonService,
    private readonly authorityResolutionService: AuthorityResolutionService,
  ) {}

  async resolveActor(principal: AuthPrincipal): Promise<AuthenticatedActor> {
    const personId = await this.principalToPersonService.resolvePersonId(principal);
    const resolved = await this.authorityResolutionService.resolveForPerson(
      personId,
      new Date().toISOString(),
    );

    return {
      personId,
      userAccountId: principal.subject,
      memberId: resolved.member?.memberId,
      email: principal.email,
      roles: resolved.roles.map((role) => role.code),
      authorityGrants: resolved.authorityGrants,
      isAuthenticated: true,
    };
  }
}

14. apps/gov-api/src/middleware/auth.ts (real boundary version)

import type { MiddlewareHandler } from "hono";
import type { HonoEnv } from "../types/hono";
import { UnauthorizedError } from "../http/errors";

export const authMiddleware: MiddlewareHandler<HonoEnv> = async (c, next) => {
  const authorization = c.req.header("authorization");

  if (!authorization?.startsWith("Bearer ")) {
    throw new UnauthorizedError();
  }

  const token = authorization.slice("Bearer ".length).trim();
  if (!token) {
    throw new UnauthorizedError("Bearer token was empty.");
  }

  const appContext = c.get("appContext");
  const requestContext = c.get("requestContext");

  const principal = await appContext.services.tokenValidator.validateBearerToken(token);
  const actor = await appContext.services.authenticatedActorService.resolveActor(principal);

  c.set("requestContext", {
    ...requestContext,
    actor,
  });

  await next();
};
