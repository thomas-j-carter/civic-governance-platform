export class InMemoryRoleAssignmentAdministrationRepository
  implements RoleAssignmentAdministrationRepository
{
  constructor(
    private readonly repository: InMemoryRoleAssignmentRepository & {
      items?: Map<string, RoleAssignment>;
    },
  ) {}

  async listRoleAssignments(): Promise<RoleAssignment[]> {
    const items = this.repository.items ? [...this.repository.items.values()] : [];

      .sort((a, b) => b.assignedAt.localeCompare(a.assignedAt))

14. apps/gov-api/src/application/authority/create-role-assignment-handler.ts

import { NotFoundError } from "../../http/errors";
import type {
  RoleAssignmentRepository,
  RoleRepository,
} from "../../domain/authority/authority-repository";
import type { AuditWriter } from "../audit/audit-writer";
import type {
  CreateRoleAssignmentCommand,
  CreateRoleAssignmentCommandHandler,
} from "./create-role-assignment";
import type { RoleAssignmentDto } from "./role-assignment-dto";
import { toRoleAssignmentDto } from "./authority-admin-mappers";

export class CreateRoleAssignmentHandler
  implements CreateRoleAssignmentCommandHandler
{
  constructor(
    private readonly roleRepository: RoleRepository,
    private readonly repository: RoleAssignmentRepository,
    private readonly auditWriter: AuditWriter,
  ) {}

  async execute(
    command: CreateRoleAssignmentCommand,
  ): Promise<RoleAssignmentDto> {
    const role = await this.roleRepository.getRoleById(command.input.roleId);

    if (!role) {
      throw new NotFoundError(`Role ${command.input.roleId} was not found.`);
    }

    const now = new Date().toISOString();

    const assignment = {
      id: crypto.randomUUID(),
      roleId: command.input.roleId,
      personId: command.input.personId,
      scopeType: command.input.scopeType,
      scopeId: command.input.scopeId,
      expiresAt: command.input.expiresAt,
      assignedAt: now,
      createdAt: now,
    };

    await this.repository.createRoleAssignment(assignment);

    await this.auditWriter.write({
      actionType: "ASSIGN",
      entityType: "RoleAssignment",
      entityId: assignment.id,
      actorPersonId: command.actor.personId,
      summary: "Role assigned.",
      metadata: {
        roleId: assignment.roleId,
        personId: assignment.personId,
        scopeType: assignment.scopeType,
        scopeId: assignment.scopeId,
      },
    });

    return toRoleAssignmentDto(assignment);
  }
}

17. apps/gov-api/src/application/authority/create-delegation-handler.ts

import { ConflictError } from "../../http/errors";
import type { DelegationRepository } from "../../domain/authority/authority-repository";
import type { AuditWriter } from "../audit/audit-writer";
import type {
  CreateDelegationCommand,
  CreateDelegationCommandHandler,
} from "./create-delegation";
import type { DelegationDto } from "./delegation-dto";
import { toDelegationDto } from "./authority-admin-mappers";

export class CreateDelegationHandler
  implements CreateDelegationCommandHandler
{
  constructor(
    private readonly repository: DelegationRepository,
    private readonly auditWriter: AuditWriter,
  ) {}

  async execute(command: CreateDelegationCommand): Promise<DelegationDto> {
    if (command.input.authorityGrantCodes.length === 0) {
      throw new ConflictError("At least one authority grant code is required.", {
        code: "delegation_grants_required",
      });
    }

    const now = new Date().toISOString();

    const delegation = {
      id: crypto.randomUUID(),
      delegatorPersonId: command.input.delegatorPersonId,
      delegatePersonId: command.input.delegatePersonId,
      scopeType: command.input.scopeType,
      scopeId: command.input.scopeId,
      authorityGrantCodes: command.input.authorityGrantCodes,
      reason: command.input.reason,
      effectiveAt: command.input.effectiveAt ?? now,
      expiresAt: command.input.expiresAt,
      createdAt: now,
    };

    await this.repository.createDelegation(delegation);

    await this.auditWriter.write({
      actionType: "DELEGATE",
      entityType: "Delegation",
      entityId: delegation.id,
      actorPersonId: command.actor.personId,
      summary: "Delegation created.",
      metadata: {
        delegatorPersonId: delegation.delegatorPersonId,
        delegatePersonId: delegation.delegatePersonId,
        authorityGrantCodes: delegation.authorityGrantCodes,
        scopeType: delegation.scopeType,
        scopeId: delegation.scopeId,
      },
    });

    return toDelegationDto(delegation);
  }
}

20. apps/gov-api/src/routes/authority-admin.ts

import { Hono } from "hono";
import type { HonoEnv } from "../types/hono";
import { jsonData } from "../http/json";
import { requireAuth } from "../middleware/require-auth";
import { requireAuthority } from "../middleware/require-authority";
import { NotFoundError, ValidationError } from "../http/errors";
import { runIdempotentCommand } from "../http/idempotency";

export function createAuthorityAdminRoutes(): Hono<HonoEnv> {
  const app = new Hono<HonoEnv>();

  app.get(
    "/role-assignments",
    requireAuth,
    requireAuthority("role_assignment.read"),
    async (c) => {
      const appContext = c.get("appContext");
      const result = await appContext.handlers.authorityAdmin.listRoleAssignments.execute();
      return c.json(jsonData(result), 200);
    },
  );

  app.get(
    "/role-assignments/:roleAssignmentId",
    requireAuth,
    requireAuthority("role_assignment.read"),
    async (c) => {
      const appContext = c.get("appContext");
      const roleAssignmentId = c.req.param("roleAssignmentId");

      const result = await appContext.handlers.authorityAdmin.getRoleAssignment.execute({
        roleAssignmentId,
      });

      if (!result) {
        throw new NotFoundError(`Role assignment ${roleAssignmentId} was not found.`);
      }

      return c.json(jsonData(result), 200);
    },
  );

  app.post(
    "/role-assignments",
    requireAuth,
    requireAuthority("role_assignment.create"),
    async (c) => {
      const appContext = c.get("appContext");
      const actor = c.get("requestContext").actor!;
      const body = await c.req.json<{
        roleId?: string;
        personId?: string;
        scopeType?: string;
        scopeId?: string;
        expiresAt?: string;
      }>();

      const errors: Record<string, string[]> = {};
      if (!body.roleId?.trim()) {
        errors.roleId = ["roleId is required."];
      }
      if (!body.personId?.trim()) {
        errors.personId = ["personId is required."];
      }

      if (Object.keys(errors).length > 0) {
        throw new ValidationError("Create role assignment request was invalid.", errors);
      }

      const requestPayload = {
        roleId: body.roleId!,
        personId: body.personId!,
        scopeType: body.scopeType,
        scopeId: body.scopeId,
        expiresAt: body.expiresAt,
      };

      const result = await runIdempotentCommand({
        c,
        operationName: "roleAssignment.create",
        requestBody: requestPayload,
        execute: async () =>
          appContext.handlers.authorityAdmin.createRoleAssignment.execute({
            actor: { personId: actor.personId },
            input: requestPayload,
          }),
      });

      return c.json(jsonData(result), 201);
    },
  );

  app.get(
    "/delegations",
    requireAuth,
    requireAuthority("delegation.read"),
    async (c) => {
      const appContext = c.get("appContext");
      const result = await appContext.handlers.authorityAdmin.listDelegations.execute();
      return c.json(jsonData(result), 200);
    },
  );

  app.get(
    "/delegations/:delegationId",
    requireAuth,
    requireAuthority("delegation.read"),
    async (c) => {
      const appContext = c.get("appContext");
      const delegationId = c.req.param("delegationId");

      const result = await appContext.handlers.authorityAdmin.getDelegation.execute({
        delegationId,
      });

      if (!result) {
        throw new NotFoundError(`Delegation ${delegationId} was not found.`);
      }

      return c.json(jsonData(result), 200);
    },
  );

  app.post(
    "/delegations",
    requireAuth,
    requireAuthority("delegation.create"),
    async (c) => {
      const appContext = c.get("appContext");
      const actor = c.get("requestContext").actor!;
      const body = await c.req.json<{
        delegatorPersonId?: string;
        delegatePersonId?: string;
        scopeType?: string;
        scopeId?: string;
        authorityGrantCodes?: string[];
        reason?: string;
        effectiveAt?: string;
        expiresAt?: string;
      }>();

      const errors: Record<string, string[]> = {};
      if (!body.delegatorPersonId?.trim()) {
        errors.delegatorPersonId = ["delegatorPersonId is required."];
      }
      if (!body.delegatePersonId?.trim()) {
        errors.delegatePersonId = ["delegatePersonId is required."];
      }
      if (!Array.isArray(body.authorityGrantCodes) || body.authorityGrantCodes.length === 0) {
        errors.authorityGrantCodes = ["authorityGrantCodes must contain at least one value."];
      }

      if (Object.keys(errors).length > 0) {
        throw new ValidationError("Create delegation request was invalid.", errors);
      }

      const requestPayload = {
        delegatorPersonId: body.delegatorPersonId!,
        delegatePersonId: body.delegatePersonId!,
        scopeType: body.scopeType,
        scopeId: body.scopeId,
        authorityGrantCodes: body.authorityGrantCodes!,
        reason: body.reason,
        effectiveAt: body.effectiveAt,
        expiresAt: body.expiresAt,
      };

      const result = await runIdempotentCommand({
        c,
        operationName: "delegation.create",
        requestBody: requestPayload,
        execute: async () =>
          appContext.handlers.authorityAdmin.createDelegation.execute({
            actor: { personId: actor.personId },
            input: requestPayload,
          }),
      });

      return c.json(jsonData(result), 201);
    },
  );

  return app;
}

What this batch accomplishes

This closes the next major institutional gap.

You now have first-class administration for the two core mutable authority mechanisms:

direct authority via role assignments

delegated authority via delegations


You now have:

create/list/get role assignment APIs

create/list/get delegation APIs

audit emission for both

idempotent create semantics

route surface for institutional authority administration

tests for handler and route behavior
