9. apps/gov-api/src/application/governance/create-governance-body-handler.ts

import { ConflictError } from "../../http/errors";
import type { GovernanceBodyRepository } from "../../domain/authority/authority-repository";
import type { AuditWriter } from "../audit/audit-writer";
import type {
  CreateGovernanceBodyCommand,
  CreateGovernanceBodyCommandHandler,
} from "./create-governance-body";
import type { GovernanceBodyAdminDto } from "./governance-admin-dto";
import { toGovernanceBodyAdminDto } from "./governance-admin-mappers";

export class CreateGovernanceBodyHandler
  implements CreateGovernanceBodyCommandHandler
{
  constructor(
    private readonly repository: GovernanceBodyRepository,
    private readonly auditWriter: AuditWriter,
  ) {}

  async execute(
    command: CreateGovernanceBodyCommand,
  ): Promise<GovernanceBodyAdminDto> {
    const existing = await this.repository.listBodies();
    if (existing.some((body) => body.code === command.input.code)) {
      throw new ConflictError("A governance body with that code already exists.", {
        code: "governance_body_code_exists",
      });
    }

    const now = new Date().toISOString();

    const body = {
      id: crypto.randomUUID(),
      code: command.input.code,
      name: command.input.name,
      description: command.input.description,
      bodyType: command.input.bodyType,
      isActive: true,
      createdAt: now,
      updatedAt: now,
    };

    await this.repository.createBody(body);

    await this.auditWriter.write({
      actionType: "CREATE",
      entityType: "GovernanceBody",
      entityId: body.id,
      actorPersonId: command.actor.personId,
      summary: "Governance body created.",
      metadata: {
        code: body.code,
        bodyType: body.bodyType,
      },
    });

    return toGovernanceBodyAdminDto(body);
  }
}

12. apps/gov-api/src/application/governance/create-office-handler.ts

import { ConflictError, NotFoundError } from "../../http/errors";
import type {
  GovernanceBodyRepository,
  OfficeRepository,
} from "../../domain/authority/authority-repository";
import type { AuditWriter } from "../audit/audit-writer";
import type {
  CreateOfficeCommand,
  CreateOfficeCommandHandler,
} from "./create-office";
import type { OfficeAdminDto } from "./governance-admin-dto";
import { toOfficeAdminDto } from "./governance-admin-mappers";

export class CreateOfficeHandler implements CreateOfficeCommandHandler {
  constructor(
    private readonly governanceBodyRepository: GovernanceBodyRepository,
    private readonly officeRepository: OfficeRepository,
    private readonly auditWriter: AuditWriter,
  ) {}

  async execute(command: CreateOfficeCommand): Promise<OfficeAdminDto> {
    const existing = await this.officeRepository.listOffices();
    if (existing.some((office) => office.code === command.input.code)) {
      throw new ConflictError("An office with that code already exists.", {
        code: "office_code_exists",
      });
    }

    if (command.input.governanceBodyId) {
      const body = await this.governanceBodyRepository.getBodyById(
        command.input.governanceBodyId,
      );

      if (!body) {
        throw new NotFoundError(
          `Governance body ${command.input.governanceBodyId} was not found.`,
        );
      }
    }

    const now = new Date().toISOString();

    const office = {
      id: crypto.randomUUID(),
      code: command.input.code,
      name: command.input.name,
      description: command.input.description,
      governanceBodyId: command.input.governanceBodyId,
      isActive: true,
      createdAt: now,
      updatedAt: now,
    };

    await this.officeRepository.createOffice(office);

    await this.auditWriter.write({
      actionType: "CREATE",
      entityType: "Office",
      entityId: office.id,
      actorPersonId: command.actor.personId,
      summary: "Office created.",
      metadata: {
        code: office.code,
        governanceBodyId: office.governanceBodyId,
      },
    });

    return toOfficeAdminDto(office);
  }
}

15. apps/gov-api/src/routes/governance-admin.ts

import { Hono } from "hono";
import type { HonoEnv } from "../types/hono";
import { jsonData } from "../http/json";
import { requireAuth } from "../middleware/require-auth";
import { requireAuthority } from "../middleware/require-authority";
import { NotFoundError, ValidationError } from "../http/errors";
import { runIdempotentCommand } from "../http/idempotency";

export function createGovernanceAdminRoutes(): Hono<HonoEnv> {
  const app = new Hono<HonoEnv>();

  app.get(
    "/bodies",
    requireAuth,
    requireAuthority("governance_body.read"),
    async (c) => {
      const appContext = c.get("appContext");
      const result = await appContext.handlers.governanceAdmin.listBodies.execute();
      return c.json(jsonData(result), 200);
    },
  );

  app.get(
    "/bodies/:governanceBodyId",
    requireAuth,
    requireAuthority("governance_body.read"),
    async (c) => {
      const appContext = c.get("appContext");
      const governanceBodyId = c.req.param("governanceBodyId");

      const result = await appContext.handlers.governanceAdmin.getBody.execute({
        governanceBodyId,
      });

      if (!result) {
        throw new NotFoundError(`Governance body ${governanceBodyId} was not found.`);
      }

      return c.json(jsonData(result), 200);
    },
  );

  app.post(
    "/bodies",
    requireAuth,
    requireAuthority("governance_body.create"),
    async (c) => {
      const appContext = c.get("appContext");
      const actor = c.get("requestContext").actor!;
      const body = await c.req.json<{
        code?: string;
        name?: string;
        description?: string;
        bodyType?: string;
      }>();

      const errors: Record<string, string[]> = {};
      if (!body.code?.trim()) errors.code = ["code is required."];
      if (!body.name?.trim()) errors.name = ["name is required."];
      if (!body.bodyType?.trim()) errors.bodyType = ["bodyType is required."];

      if (Object.keys(errors).length > 0) {
        throw new ValidationError("Create governance body request was invalid.", errors);
      }

      const requestPayload = {
        code: body.code!,
        name: body.name!,
        description: body.description,
        bodyType: body.bodyType!,
      };

      const result = await runIdempotentCommand({
        c,
        operationName: "governanceBody.create",
        requestBody: requestPayload,
        execute: async () =>
          appContext.handlers.governanceAdmin.createBody.execute({
            actor: { personId: actor.personId },
            input: requestPayload,
          }),
      });

      return c.json(jsonData(result), 201);
    },
  );

  app.get(
    "/offices",
    requireAuth,
    requireAuthority("office.read"),
    async (c) => {
      const appContext = c.get("appContext");
      const result = await appContext.handlers.governanceAdmin.listOffices.execute();
      return c.json(jsonData(result), 200);
    },
  );

  app.get(
    "/offices/:officeId",
    requireAuth,
    requireAuthority("office.read"),
    async (c) => {
      const appContext = c.get("appContext");
      const officeId = c.req.param("officeId");

      const result = await appContext.handlers.governanceAdmin.getOffice.execute({
        officeId,
      });

      if (!result) {
        throw new NotFoundError(`Office ${officeId} was not found.`);
      }

      return c.json(jsonData(result), 200);
    },
  );

  app.post(
    "/offices",
    requireAuth,
    requireAuthority("office.create"),
    async (c) => {
      const appContext = c.get("appContext");
      const actor = c.get("requestContext").actor!;
      const body = await c.req.json<{
        code?: string;
        name?: string;
        description?: string;
        governanceBodyId?: string;
      }>();

      const errors: Record<string, string[]> = {};
      if (!body.code?.trim()) errors.code = ["code is required."];
      if (!body.name?.trim()) errors.name = ["name is required."];

      if (Object.keys(errors).length > 0) {
        throw new ValidationError("Create office request was invalid.", errors);
      }

      const requestPayload = {
        code: body.code!,
        name: body.name!,
        description: body.description,
        governanceBodyId: body.governanceBodyId,
      };

      const result = await runIdempotentCommand({
        c,
        operationName: "office.create",
        requestBody: requestPayload,
        execute: async () =>
          appContext.handlers.governanceAdmin.createOffice.execute({
            actor: { personId: actor.personId },
            input: requestPayload,
          }),
      });

      return c.json(jsonData(result), 201);
    },
  );

  return app;
}
