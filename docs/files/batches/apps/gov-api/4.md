import { ConflictError } from "../../http/errors";
import type { CertificationRecord } from "./certification-record";

export function assertCertificationCanBeCreated(existing: CertificationRecord | null): void {
  if (existing) {
    throw new ConflictError("A certification record already exists for this ballot.", {
      code: "certification_already_exists",
    });
  }
}

export function assertCertificationCanBeCertified(record: CertificationRecord): void {
  if (!["PENDING", "UNDER_REVIEW"].includes(record.status)) {
    throw new ConflictError(
      `Certification cannot be completed from current status ${record.status}.`,
      {
        type: "https://gov-api.ardtiresociety.org/problems/invalid-state-transition",
        code: "invalid_state_transition",
      },
    );
  }
}

export function assertCertificationCanBeRejected(record: CertificationRecord): void {
  if (!["PENDING", "UNDER_REVIEW"].includes(record.status)) {
    throw new ConflictError(
      `Certification cannot be rejected from current status ${record.status}.`,
      {
        type: "https://gov-api.ardtiresociety.org/problems/invalid-state-transition",
        code: "invalid_state_transition",
      },
    );
  }
}

14. apps/gov-api/src/infrastructure/persistence/in-memory/in-memory-record-repository.ts

import type {
  ListRecordsInput,
  ListRecordsResult,
  OfficialRecordRepository,
  RecordVersionRepository,
} from "../../../domain/records/record-repository";
import type { OfficialRecord } from "../../../domain/records/official-record";
import type { RecordVersion } from "../../../domain/records/record-version";

export class InMemoryOfficialRecordRepository implements OfficialRecordRepository {
  private readonly records = new Map<string, OfficialRecord>();

  async createRecord(input: OfficialRecord): Promise<void> {
    this.records.set(input.id, structuredClone(input));
  }

  async updateRecord(input: OfficialRecord): Promise<void> {
    this.records.set(input.id, structuredClone(input));
  }

  async getRecordById(recordId: string): Promise<OfficialRecord | null> {
    const record = this.records.get(recordId);
    return record ? structuredClone(record) : null;
  }

  async listRecords(input: ListRecordsInput): Promise<ListRecordsResult> {
    let items = [...this.records.values()];

    if (input.recordType) {
      items = items.filter((item) => item.recordType === input.recordType);
    }

    if (input.status) {
      items = items.filter((item) => item.status === input.status);
    }

    items.sort((a, b) => b.createdAt.localeCompare(a.createdAt));

    const total = items.length;
    const totalPages = Math.max(1, Math.ceil(total / input.limit));
    const offset = (input.page - 1) * input.limit;
    const paged = items.slice(offset, offset + input.limit).map((item) => structuredClone(item));

    return {
      items: paged,
      page: input.page,
      limit: input.limit,
      total,
      totalPages,
    };
  }
}

export class InMemoryRecordVersionRepository implements RecordVersionRepository {
  private readonly versions = new Map<string, RecordVersion>();

  async createRecordVersion(input: RecordVersion): Promise<void> {
    this.versions.set(input.id, structuredClone(input));
  }

  async listRecordVersions(recordId: string): Promise<RecordVersion[]> {
    return [...this.versions.values()]
      .filter((item) => item.officialRecordId === recordId)
      .sort((a, b) => a.versionNumber - b.versionNumber)
      .map((item) => structuredClone(item));
  }
}

22. apps/gov-api/src/application/certifications/create-certification-handler.ts

import { NotFoundError } from "../../http/errors";
import { assertCertificationCanBeCreated } from "../../domain/certifications/certification-rules";
import type { BallotRepository } from "../../domain/ballots/ballot-repository";
import type { CertificationRepository } from "../../domain/certifications/certification-repository";
import type {
  CreateCertificationCommand,
  CreateCertificationCommandHandler,
  CertificationDetailDto,
} from "./create-certification";
import { toCertificationDetailDto } from "./certification-mappers";

export class CreateCertificationHandler implements CreateCertificationCommandHandler {
  constructor(
    private readonly ballotRepository: BallotRepository,
    private readonly certificationRepository: CertificationRepository,
  ) {}

  async execute(command: CreateCertificationCommand): Promise<CertificationDetailDto> {
    const ballot = await this.ballotRepository.getBallotById(command.input.ballotId);
    if (!ballot) {
      throw new NotFoundError(`Ballot ${command.input.ballotId} was not found.`);
    }

    const existing = await this.certificationRepository.findCertificationByBallotId(
      command.input.ballotId,
    );
    assertCertificationCanBeCreated(existing);

    const now = new Date().toISOString();

    const record = {
      id: crypto.randomUUID(),
      ballotId: command.input.ballotId,
      status: "PENDING" as const,
      notes: command.input.notes,
      quorumRuleVersionId: "stub-quorum-rule-version-id",
      thresholdRuleVersionId: "stub-threshold-rule-version-id",
      certificationRuleVersionId: "stub-certification-rule-version-id",
      createdAt: now,
      updatedAt: now,
    };

    await this.certificationRepository.createCertification(record);

    return toCertificationDetailDto(record);
  }
}

24. apps/gov-api/src/application/certifications/certify-result-handler.ts

import { NotFoundError } from "../../http/errors";
import { assertCertificationCanBeCertified } from "../../domain/certifications/certification-rules";
import type {
  CertificationRepository,
} from "../../domain/certifications/certification-repository";
import type { VoteTallyRepository } from "../../domain/ballots/ballot-repository";
import type {
  CertifyResultCommand,
  CertifyResultCommandHandler,
} from "./certify-result";
import type { CertificationDetailDto } from "./create-certification";
import { toCertificationDetailDto } from "./certification-mappers";

export class CertifyResultHandler implements CertifyResultCommandHandler {
  constructor(
    private readonly certificationRepository: CertificationRepository,
    private readonly voteTallyRepository: VoteTallyRepository,
  ) {}

  async execute(command: CertifyResultCommand): Promise<CertificationDetailDto> {
    const record = await this.certificationRepository.getCertificationById(command.certificationId);
    if (!record) {
      throw new NotFoundError(`Certification ${command.certificationId} was not found.`);
    }

    assertCertificationCanBeCertified(record);

    const tally = await this.voteTallyRepository.getTallyByBallotId(record.ballotId);
    if (!tally) {
      throw new NotFoundError(`No tally exists for ballot ${record.ballotId}.`);
    }

    record.status = "CERTIFIED";
    record.certifiedByPersonId = command.actor.personId;
    record.certifiedAt = new Date().toISOString();
    record.updatedAt = new Date().toISOString();

    await this.certificationRepository.updateCertification(record);

    return toCertificationDetailDto(record);
  }
}

      certificationId,

  });
